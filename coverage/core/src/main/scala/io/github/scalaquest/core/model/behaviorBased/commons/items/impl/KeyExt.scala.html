13 /**
14  * The trait makes possible to mix into a [[BehaviorBasedModel]] the <b>Key BehaviorBasedItem</b>.
15  */
16 trait KeyExt extends BehaviorBasedModel {
17 
18   /**
19    * A <b>BehaviorBasedItem</b> that should be used to open/close items with an <b>Openable</b>
20    * behavior.
21    */
22   trait Key extends BehaviorBasedItem {
23 
24     /**
25      * A disposable key is cancelled from the game after being used to open something.
26      * @return
27      *   True if key is cancelled after his use, False otherwise.
28      */
29     def disposable: Boolean
30   }
31 
32   /**
33    * Standard implementation of <b>Key</b>.
34    *
35    * @param description
36    *   An [[ItemDescription]] for the item.
37    * @param ref
38    *   A unique reference to the item.
39    * @param extraBehavBuilders
40    *   Additional behaviors associated to the item.
41    */
42   case class SimpleKey(
43     description: ItemDescription,
44     ref: ItemRef,
45     disposable: Boolean = true,
46     extraBehavBuilders: Seq[I => ItemBehavior] = Seq.empty
47   ) extends Key {
48     override val behaviors: Seq[ItemBehavior] = extraBehavBuilders.map(_(this))
49   }
50 
51   /**
52    * Companion object for <b>Key</b>.
53    */
54   object Key {
55 
56     /**
57      * Creates a standard <b>Key</b>.
58      * @param description
59      *   An [[ItemDescription]] for the item.
60      * @param extraBehavBuilders
61      *   Additional behaviors associated to the item.
62      * @return
63      *   An instance of a standard <b>Key</b>.
64      */
65     def apply(
66       description: ItemDescription,
67       disposable: Boolean = true,
68       extraBehavBuilders: Seq[I => ItemBehavior] = Seq.empty
69     ): Key = SimpleKey(description, ItemRef(description), disposable, extraBehavBuilders)
70   }
71 }
