14 /**
15  * The trait makes possible to add into the [[BehaviorBasedModel]] the <b>GenericItemBehavior</b>.
16  */
17 trait GenericItemBehaviorExt extends BehaviorBasedModel with CMessagesExt with CReactionsExt {
18 
19   /**
20    * An ItemBehavior with some triggers. It is useful in order to wrap <b>ItemBehavior</b> into a
21    * facility builder.
22    */
23   abstract class GenericItemBehavior extends ItemBehavior {
24 
25     /**
26      * <b>ItemTriggers</b> associated to the behavior.
27      * @return
28      *   Triggers associated to the behavior.
29      */
30     def triggers: ItemTriggers
31   }
32 
33   /**
34    * Standard implementation of <b>GenericItemBehavior</b>.
35    */
36   case class SimpleGenericItemBehavior(
37     override val triggers: ItemTriggers
38   )(implicit val subject: I)
39     extends GenericItemBehavior
40 
41   /**
42    * Companion object for <b>GenericItemBehavior</b>.
43    */
44   object GenericItemBehavior {
45 
46     /**
47      * A function that builds a <b>GenericBehavior</b> given a subject.
48      * @param triggers
49      *   The <b>ItemTriggers</b> associated to the subject.
50      * @return
51      *   A function that builds a <b>GenericBehavior</b> given a subject.
52      */
53     def builder(triggers: ItemTriggers = PartialFunction.empty): I => GenericItemBehavior =
54       SimpleGenericItemBehavior(triggers)(_)
55   }
56 }
