16 trait SimpleRoomExt extends Model {

20   /**
21    * A base implementation for [[Room]].
22    * @param name
23    *   The [[Room]] name.
24    * @param _items
25    *   the [[ItemRef]] that refers to the <b>Item</b> contained by this <b>SimpleRoom</b>.
26    * @param _neighbors
27    *   [[Map]] with neighbors of this <b>SimpleRoom</b>.
28    * @param ref
29    *   The [[RoomRef]] for this <b>SimpleRoom</b>.
30    */
31   case class SimpleRoom(
32     name: String,
33     _items: () => Set[ItemRef],
34     _neighbors: () => Map[Direction, RoomRef],
35     ref: RoomRef
36   ) extends Room {
37 
38     override def neighbor(direction: Direction)(implicit state: S): Option[RM] =
39       for {
40         refInDirection <- _neighbors().get(direction)
41         room           <- state.rooms.get(refInDirection)
42       } yield room
43 
44     override def items(implicit state: S): Set[I] = _items().flatMap(state.items.get)
45 
46     override def neighbors(implicit state: S): Map[Direction, RM] =
47       _neighbors().map(x => x._1 -> state.rooms(x._2))
48   }
49 
50   /**

52    */
53   object Room {
54 
55     /**
56      * A base implementation for <b>Room</b>.
57      * @param name
58      *   The <b>Room</b> name.
59      * @param neighbors
60      *   Map that define all <b>neighbors</b> of this room.
61      * @param items
62      *   All the <b>items</b> in this match.
63      */
64     def apply(
65       name: String,
66       neighbors: => Map[Direction, RoomRef] = Map.empty,
67       items: => Set[ItemRef] = Set.empty
68     ): RM = SimpleRoom(name, () => items, () => neighbors, RoomRef(name))
69   }
70
