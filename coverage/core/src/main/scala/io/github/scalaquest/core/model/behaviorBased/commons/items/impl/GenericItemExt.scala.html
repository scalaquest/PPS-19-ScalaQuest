14 /**
15  * The trait makes possible to mix into a [[BehaviorBasedModel]] the <b>GenericItem</b>.
16  */
17 trait GenericItemExt extends BehaviorBasedModel with GenericItemBehaviorExt {
18 
19   /**
20    * A standard <b>BehaviorBasedItem</b> completely and freely configurable, without a specific
21    * category.
22    */
23   trait GenericItem extends BehaviorBasedItem
24 
25   /**
26    * Standard implementation of <b>GenericItem</b>.
27    *
28    * @param description
29    *   An [[ItemDescription]] for the item.
30    * @param ref
31    *   A unique reference to the item.
32    * @param behaviorsBuilders
33    *   All the behaviors associated to the item.
34    */
35   case class SimpleGenericItem(
36     description: ItemDescription,
37     ref: ItemRef,
38     behaviorsBuilders: Seq[I => ItemBehavior] = Seq.empty
39   ) extends GenericItem {
40     override val behaviors: Seq[ItemBehavior] = behaviorsBuilders.map(_(this))
41   }
42 
43   /**
44    * Companion object for <b>GenericItem</b>.
45    */
46   object GenericItem {
47 
48     /**
49      * A standard <b>GenericItem</b>.
50      * @param description
51      *   An [[ItemDescription]] for the item.
52      * @param extraBehavBuilders
53      *   Builder of all the behaviors associated to the item.
54      * @return
55      *   An instance of a standard <b>GenericItem</b>.
56      */
57     def apply(
58       description: ItemDescription,
59       extraBehavBuilders: Seq[I => ItemBehavior] = Seq.empty
60     ): GenericItem = SimpleGenericItem(description, ItemRef(description), extraBehavBuilders)
61 
62     /**
63      * A <b>GenericItem</b> with a single behavior.
64      * @param description
65      *   An [[ItemDescription]] for the item.
66      * @param behavior
67      *   Builder for the single behavior associated to the item.
68      * @return
69      *   An instance of a <b>GenericItem</b> with a single behavior.
70      */
71     def withSingleBehavior(
72       description: ItemDescription,
73       behavior: I => ItemBehavior
74     ): GenericItem = SimpleGenericItem(description, ItemRef(description), Seq(behavior))
75 
76     /**
77      * An item with no behaviors.
78      * @param description
79      *   An [[ItemDescription]] for the item.
80      * @return
81      *   An item with no behaviors.
82      */
83     def empty(
84       description: ItemDescription
85     ): GenericItem = SimpleGenericItem(description, ItemRef(description), Seq())
86 
87     /**
88      * A <b>GenericItem</b> with a single behavior, created on-the-fly by some <b>ItemTriggers</b>.
89      * @param description
90      *   the generic item description.
91      * @param behaviorTriggers
92      *   Triggers for the single behavior associated to the item.
93      * @return
94      *   An instance of a <b>GenericItem</b> with a single behavior.
95      */
96     def withGenBehavior(
97       description: ItemDescription,
98       behaviorTriggers: ItemTriggers
99     ): GenericItem =
100       SimpleGenericItem(
101         description,
102         ItemRef(description),
103         Seq(GenericItemBehavior.builder(behaviorTriggers))
104       )
105   }
106 }
