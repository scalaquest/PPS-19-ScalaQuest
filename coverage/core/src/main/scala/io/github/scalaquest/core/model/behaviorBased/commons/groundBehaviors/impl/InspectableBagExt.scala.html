15 /**
16  * The trait makes possible to mix into a [[BehaviorBasedModel]] the <b>InspectableBag
17  * GroundBehavior</b>.
18  */
19 trait InspectableBagExt extends BehaviorBasedModel with CMessagesExt with CReactionsExt {
20 
21   /**
22    * A <b>GroundBehavior</b> that enables the possibility to know what items are contained into the
23    * bag.
24    */
25   abstract class InspectableBag extends GroundBehavior {
26 
27     /**
28      * A <b>Reaction</b> that communicates to the user the items contained into the bag.
29      * @return
30      *   A <b>Reaction</b> that communicates to the user the items contained into the bag.
31      */
32     def inspectBag: Reaction
33   }
34 
35   /**
36    * A standard implementation for <b>InspectableBag</b>.
37    *
38    * @param onInspectExtra
39    *   <b>Reaction</b> to be executed when the player successfully inspects the bag. It can be
40    *   omitted.
41    */
42   case class SimpleInspectableBag(onInspectExtra: Reaction = Reaction.empty)
43     extends InspectableBag {
44 
45     override def triggers: GroundTriggers = { case (InspectBag, _) => inspectBag }
46 
47     override def inspectBag: Reaction =
48       for {
49         s1 <- Reaction.empty
50         _  <- CReactions.addMessage(CMessages.InspectedBag(s1.bag))
51         s2 <- onInspectExtra
52       } yield s2
53   }
54 
55   /**
56    * Companion object for <b>InspectableBag</b>.
57    */
58   object InspectableBag {
59 
60     /**
61      * Shortcut for the standard implementation.
62      * @param onInspectExtra
63      *   An additional <b>Reaction</b> generated when user inspects the bag.
64      * @return
65      *   An instance of <b>SimpleInspectableBag</b>.
66      */
67     def apply(onInspectExtra: Reaction = Reaction.empty): InspectableBag =
68       SimpleInspectableBag(onInspectExtra)
69   }
70 }
