3 /**
4  * This package includes some facility methods in order to create terms in a Prolog-like syntax.
5  *
6  * If you include implicit conversions and [[dsl.CompoundBuilder]] you can create terms as in this
7  * example:
8  * {{{
9  *   val hello = CompoundBuilder("hello").constructor
10  *   val X = Variable("X")
11  *   (hello(X) --> ListP("hello", X)).generate
12  * }}}
13  *
14  * Should result in the following Prolog clause being printed:
15  * {{{
16  *   hello(X) --> ["hello", X].
17  * }}}
18  */
19 package object dsl {
20 
21   /** Enables the implicit conversion from `String` to `Atom` */
22   implicit def stringToAtom(name: String): Atom = Atom(name)
23 
24   /** Enables the implicit conversion from `Int` to `Number` */
25   implicit def intToNumber(n: Int): Number = Number(n)
26 
27   /** Enables the implicit conversion from `Term` to `Fact` */
28   implicit def termToFact(term: Term): Fact = Fact(term)
29 
30   /** Enables the implicit conversion fron `Seq[Term]` to `ListP` */
31   implicit def seqToListP(seq: Seq[Term]): ListP = ListP(seq: _*)
