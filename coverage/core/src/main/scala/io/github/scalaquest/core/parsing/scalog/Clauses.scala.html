10 /** Unit of information ending with the full-stop. */
11 sealed trait Clause extends CodeGen
12 
13 /** A special case of clause with empty body. */
14 case class Fact(head: Term) extends Clause {
15 
16   /**
17    * Allows the creation of a [[DCGRule]].
18    * @param right
19    *   the right side of the operation
20    * @return
21    *   the resulting DCG rule.
22    */
23   def -->(right: Term): DCGRule = DCGRule(head, right)
24 
25   /**
26    * Allows the creation of a [[Rule]].
27    * @param right
28    *   the right side of the operation
29    * @return
30    *   the resulting rule.
31    */
32   def :-(right: Term): Rule = Rule(head, right)
33 
34   override def generate: String = s"${head.generate}."
35 }

37 /**
38  * A horn clause. It is read `head` is true if `body` is true.
39  *
40  * The following example:
41  * {{{
42  *   Rule(Compound(Atom("hello"), Atom("world")), Atom("hello")).generate
43  * }}}
44  *
45  * Should result in the following Prolog clause being printed:
46  * {{{
47  *   hello(world) :- hello.
48  * }}}
49  *
50  * @param head
51  *   the left term
52  * @param body
53  *   the right term
54  */
55 case class Rule(head: Term, body: Term) extends Clause {
56   override def generate: String = s"${head.generate} :- ${body.generate}."
57 }
58 
59 /**
60  * A definite clause grammar rule.
61  *
62  * The following example:
63  * {{{
64  *   DCGRule(Compound(Atom("hello"), Atom("world")), ListP(Atom("hello"), Atom("world"))).generate
65  * }}}
66  *
67  * Should result in the following Prolog clause being printed:
68  * {{{
69  *   hello(world) --> ["hello","world"].
70  * }}}
71  *
72  * @param left
73  *   the left term
74  * @param right
75  *   the right term
76  */
77 case class DCGRule(left: Term, right: Term) extends Clause {
78   override def generate: String = s"${left.generate} --> ${right.generate}."
79 }
