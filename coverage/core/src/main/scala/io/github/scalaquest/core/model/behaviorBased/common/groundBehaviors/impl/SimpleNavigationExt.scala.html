12 trait SimpleNavigationExt extends CommonBase {
13 
14   /**
15    * Standard implementation of the Navigation GroundBehavior.
16    * @param onNavigateExtra
17    *   Reaction to be executed when the player successfully navigate in a new Room, using navigation
18    *   Actions after the standard [[Reaction]]. It can be omitted.
19    */
20   case class SimpleNavigation(onNavigateExtra: Option[Reaction] = None)(implicit
21     playerLocationLens: Lens[S, RoomRef],
22     messageLens: Lens[S, Seq[Message]]
23   ) extends Navigation {
24 
25     override def triggers: GroundTriggers = {
26       // "go <direction>"
27       case (Go(direction), state) if (for {
28             roomRef <- state.currentRoom.neighbor(direction)
29             room    <- state.roomFromRef(roomRef)
30           } yield movePlayer(room)).isDefined =>
31         (for {
32           roomRef <- state.currentRoom.neighbor(direction)
33           room    <- state.roomFromRef(roomRef)
34         } yield movePlayer(room)).getOrElse(state => state)
35     }
36 
37     def movePlayer(targetRoom: RM): Reaction =
38       state =>
39         state.applyReactions(
40           playerLocationLens.set(targetRoom.ref),
41           messageLens.modify(_ :+ Navigated(targetRoom)),
42           onNavigateExtra.getOrElse(s => s)
43         )
44   }
45 }
