14 /**
15  * The trait makes possible to mix into a [[BehaviorBasedModel]] the <b>Door BehaviorBasedItem</b>.
16  */
17 trait DoorExt extends BehaviorBasedModel with RoomLinkExt {
18 
19   /**
20    * A <b>BehaviorBasedItem</b> that should work as a link between two different rooms.
21    */
22   trait Door extends BehaviorBasedItem {
23 
24     /**
25      * The "openness" state of the item, as a boolean value. Initially closed.
26      * @return
27      *   True if item is already opened, False otherwise.
28      */
29     def isOpen: Boolean
30 
31     /**
32      * A <b>Door</b> has always an associated <b>RoomLink</b> behavior, that can be referenced from
33      * here.
34      * @return
35      *   The <b>RoomLink</b> behavior of the item.
36      */
37     def roomLink: RoomLink
38   }
39 
40   /**
41    * Standard implementation of <b>Door</b>.
42    *
43    * @param description
44    *   An [[ItemDescription]] for the item.
45    * @param ref
46    *   A unique reference to the item.
47    * @param roomLinkBuilder
48    *   A builder for the <b>RoomLink</b> behavior associated to the item.
49    * @param extraBehavBuilders
50    *   Additional behaviors associated to the item.
51    */
52   case class SimpleDoor(
53     description: ItemDescription,
54     ref: ItemRef,
55     roomLinkBuilder: I => RoomLink,
56     extraBehavBuilders: Seq[I => ItemBehavior] = Seq.empty
57   ) extends Door {
58     override val roomLink: RoomLink           = roomLinkBuilder(this)
59     override val behaviors: Seq[ItemBehavior] = roomLink +: extraBehavBuilders.map(_(this))
60     override def isOpen: Boolean              = roomLink.isOpen
61   }
62 
63   /**
64    * Companion object for <b>Door</b>.
65    */
66   object Door {
67 
68     /**
69      * Creates a standard <b>Door</b>.
70      * @param description
71      *   An [[ItemDescription]] for the item.
72      * @param roomLinkBuilder
73      *   A builder for the <b>RoomLink</b> behavior associated to the item.
74      * @param extraBehavBuilders
75      *   Additional behaviors associated to the item.
76      * @return
77      *   An instance of a standard <b>Door</b>.
78      */
79     def apply(
80       description: ItemDescription,
81       roomLinkBuilder: I => RoomLink,
82       extraBehavBuilders: Seq[I => ItemBehavior] = Seq.empty
83     ): Door = SimpleDoor(description, ItemRef(description), roomLinkBuilder, extraBehavBuilders)
84 
85     /**
86      * Creates a Door-Key pair, in which the given <b>Key</b> can be used to open the associated
87      * <b>Door</b>.
88      *
89      * @param key
90      *   The <b>Key</b> to use to open the <b>Door</b>.
91      * @param doorDesc
92      *   An [[ItemDescription]] for the <b>Door</b>.
93      * @param endRoom
94      *   The <b>Room</b> that will be set as the new location of the player, after entering the
95      *   <b>Door</b>.
96      * @param endRoomDirection
97      *   The [[Direction]] that will be associated to the end room, when available to enter.
98      * @param onOpenExtra
99      *   extra reaction generated when door is opened.
100      * @param onEnterExtra
101      *   An extra behavior generated when player enter the target room.
102      * @param extraBehavBuilders
103      *   An extra behavior generated when player opens the subject.
104      * @return
105      *   A tuple containing the generated <b>Door</b>, and the <b>Key</b> required to open it.
106      */
107     def createLockedWithKey(
108       key: Key,
109       doorDesc: ItemDescription,
110       endRoom: RM,
111       endRoomDirection: Direction,
112       onOpenExtra: Reaction = Reaction.empty,
113       onEnterExtra: Reaction = Reaction.empty,
114       extraBehavBuilders: Seq[I => ItemBehavior] = Seq.empty
115     ): (Door, Key) = {
116       val door = apply(
117         description = doorDesc,
118         RoomLink.closedLockedBuilder(
119           endRoom = endRoom,
120           endRoomDirection = endRoomDirection,
121           key = key,
122           onOpenExtra = onOpenExtra,
123           onEnterExtra = onEnterExtra
124         ),
125         extraBehavBuilders = extraBehavBuilders
126       )
127 
128       (door, key)
129     }
130 
131     /**
132      * Creates an opened door. It means that you can directly enter it without open.
133      * @param description
134      *   An [[ItemDescription]] for the <b>Door</b>.
135      * @param endRoom
136      *   The <b>Room</b> that will be set as the new location of the player, after entering the
137      *   <b>Door</b>.
138      * @param endRoomDirection
139      *   The [[Direction]] that will be associated to the end room, when available to enter.
140      * @param onEnterExtra
141      *   An extra behavior generated when player enter the target room.
142      * @param extraBehavBuilders
143      *   An extra behavior generated when player opens the subject.
144      * @return
145      *   Creates an opened door. It means that you can directly enter it without open.
146      */
147     def createOpened(
148       description: ItemDescription,
149       endRoom: RM,
150       endRoomDirection: Direction,
151       onEnterExtra: Reaction = Reaction.empty,
152       extraBehavBuilders: Seq[I => ItemBehavior] = Seq.empty
153     ): Door =
154       apply(
155         description,
156         RoomLink.openedBuilder(endRoom, endRoomDirection, onEnterExtra),
157         extraBehavBuilders
158       )
159 
160     /**
161      * Creates a closed unlocked door. It means that ite can be opened without key, then entered.
162      * @param description
163      *   An [[ItemDescription]] for the <b>Door</b>.
164      * @param endRoom
165      *   The <b>Room</b> that will be set as the new location of the player, after entering the
166      *   <b>Door</b>.
167      * @param endRoomDirection
168      *   The [[Direction]] that will be associated to the end room, when available to enter.
169      * @param onEnterExtra
170      *   An extra behavior generated when player enter the target room.
171      * @param extraBehavBuilders
172      *   An extra behavior generated when player opens the subject.
173      * @return
174      *   Creates a closed unlocked door. It means that ite can be opened without key, then entered.
175      */
176     def createClosedUnlocked(
177       description: ItemDescription,
178       endRoom: RM,
179       endRoomDirection: Direction,
180       onEnterExtra: Reaction = Reaction.empty,
181       extraBehavBuilders: Seq[I => ItemBehavior] = Seq.empty
182     ): Door =
183       apply(
184         description,
185         RoomLink.closedUnlockedBuilder(endRoom, endRoomDirection, onEnterExtra),
186         extraBehavBuilders
187       )
188   }
189 }
