18 /**
19  * Provides the theory and the structures needed by the application, given a set of items and a set
20  * of verbs.
21  */
22 trait DictionaryProvider[M0 <: Model] extends ApplicationStructure[M0] {
23   import io.github.scalaquest.core.dictionary.generators.implicits.listToMapGenerator
24   import io.github.scalaquest.core.dictionary.implicits.{itemToEntryGenerator, verbToEntryGenerator}
25 
26   /** The set of verbs used by the application. */
27   def verbs: Set[Verb]
28 
29   /** The set of items used by the application. */
30   def items: Set[I]
31 
32   /**
33    * The theory generated by the combination of static generic code and the dynamically generated
34    * one.
35    */
36   def baseTheory: String = programFromResource(baseTheoryPath)
37 
38   /** The path of the static theory. */
39   final def baseTheoryPath: String = "base.pl"
40 
41   /** The `Map` that associates every `ItemRef` to the item itself. */
42   final def refToItem: Map[ItemRef, I] = GeneratorK[List, I, Map[ItemRef, I]].generate(items.toList)
43 
44   /** The `Map` that associates every pair `(Verb, Preposition)` to its action. */
45   final def verbToAction: Map[VerbPrep, Action] =
46     GeneratorK[List, Verb, Map[VerbPrep, Action]].generate(verbs.toList)
47 
48   private final def programSource[F[_]: Functor](base: F[String]): F[String] =
49     ProgramFromDictionary(verbs, items).generateSource(base)
50 
51   private final def programFromResource(resourceName: String): String = {
52     type Id[X] = X
53     programSource[Id](Source.fromResource(resourceName).mkString)
54   }
55 }
