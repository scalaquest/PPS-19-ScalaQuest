11   /**
12    * A class that wrap a reaction and it map and flatmap methods.
13    * @param r
14    *   The wrapped <b>Reaction</b>.
15    */
16   implicit class EnhancedReaction(r: Reaction) {
17 
18     /**
19      * Use <b>Reaction.map</b> method with this and an other <b>Reaction</b>.
20      * @param g
21      *   The other <b>Reaction</b>.
22      * @return
23      *   A new <b>Reaction</b> that is the <b>Reaction::map</b> of the two reactions.
24      */
25     def map(g: S => S): Reaction = Reaction.map(g)(r)
26 
27     /**
28      * Use <b>Reaction::flatmap</b> method with this and an other <b>Reaction</b>.
29      * @param g
30      *   The other <b>Reaction</b>.
31      * @return
32      *   A new <b>Reaction</b> that is the <b>Reaction::flatmap</b> of the two reactions.
33      */
34     def flatMap(g: S => Reaction): Reaction = Reaction.flatMap(g)(r)
35   }
36 
37   /**
38    * Object with some methods to work on <b>Reaction</b>.
39    */
40   object Reaction {
41 
42     /**
43      * Append a specific [[Message]] to a <b>Reaction</b>.
44      * @param msg
45      *   The <b>message</b> to be appended.
46      * @param r
47      *   The <b>Reaction</b> that will have the message appended.
48      * @return
49      *   A new <b>Reaction</b> with the <b>message</b> appended.
50      */
51     def appendMessage(msg: Message)(r: Reaction): Reaction = appendMessages(Seq(msg))(r)
52 
53     /**
54      * Append some specific [[Message]] s to a <b>Reaction</b>.
55      * @param msgs
56      *   The <b>messages</b> to be appended.
57      * @param r
58      *   The <b>Reaction</b> that will have the messages appended.
59      * @return
60      *   A new <b>Reaction</b> with the <b>messages</b> appended.
61      */
62     def appendMessages(msgs: Seq[Message])(r: Reaction): Reaction =
63       s => {
64         val (s1, m1) = r(s)
65         (s1, m1 ++ msgs)
66       }
67 
68     /**
69      * Create a <b>Reaction</b> from a function <b>State => State</b> and some <b>messages</b>.
70      * @param f
71      *   A function <b>State => State</b> that giving a State provide an updated on it.
72      * @param messages
73      *   The <b>messages</b> of <b>Reaction</b>.
74      * @return
75      *   A new <b>Reaction</b> with as function <b>f</b> and for messages <b>messages</b>.
76      */
77     def apply(f: S => S, messages: Message*): Reaction = create(f, messages)
78 
79     /**
80      * Combine two <b>Reaction</b>s using the <b>Reaction::flatmap</b>.
81      * @param a
82      *   A <b>Reaction</b>.
83      * @param b
84      *   An other <b>Reaction</b>.
85      * @return
86      *   The new <b>Reaction</b>, result combining of the precedent ones.
87      */
88     def combine(a: Reaction, b: Reaction): Reaction = a.flatMap(_ => b)
89 
90     /**
91      * Create a <b>Reaction</b> from a function <b>State => State</b> and some <b>messages</b>.
92      * @param f
93      *   A function <b>State => State</b> that giving a State provide an updated on it.
94      * @param messages
95      *   The <b>messages</b> of <b>Reaction</b>.
96      * @return
97      *   The new <b>Reaction</b> as function <b>f</b> and for messages <b>messages</b>.
98      */
99     def create(f: S => S, messages: Seq[Message]): Reaction = s => (f(s), messages)
100 
101     /**
102      * Create an <b>empty Reaction</b>:
103      *   - no message.
104      *   - function identity [[S]] => [[S]]
105      * @return
106      *   The empty <b>Reaction</b>.
107      */
108     def empty: Reaction = create(identity[S], Seq.empty)
109 
110     /**
111      * The <b>flatmap</b> implemented for <b>Reactions</b>:
112      *   - product a <b>State</b> applying the two <b>Reactions</b>
113      *   - concatenate as message the two <b>messages</b> of the Reactions.
114      * @param g
115      *   A function <b>State => Reaction</b>.
116      * @param r
117      *   A <b>Reaction</b>.
118      * @return
119      *   A new <b>Reaction</b> combining the function <b>g</b> and the reaction <b>r</b>.
120      */
121     def flatMap(g: S => Reaction)(r: Reaction): Reaction = { s =>
122       val (s1, m1) = r(s)
123       val (s2, m2) = g(s1)(s1)
124       (s2, m1 ++ m2)
125     }
126 
127     /**
128      * The <b>map</b> implemented for <b>Reactions</b>:
129      *   - product a <b>State</b> applying a <b>Reaction</b> on a function <b>State => State</b>
130      *   - concatenate as message the two <b>messages</b> of the <b>Reactions</b>.
131      * @param g
132      *   A function <b>State => Reaction</b>.
133      * @param r
134      *   A <b>Reaction</b>.
135      * @return
136      *   A new <b>Reaction</b> combining the function <b>g</b> and the reaction <b>r</b>.
137      */
138     def map(g: S => S)(r: Reaction): Reaction =
139       s => {
140         val (s1, m1) = r(s)
141         (g(s1), m1)
142       }
143 
144     /**
145      * Add <b>messages</b> to an <b>empty Reaction</b>.
146      * @param msgs
147      *   <b>Messages</b> to append to <b>Reaction</b>.
148      * @return
149      *   The <b>empty Reaction</b> with the <b>messages</b>.
150      */
151     def messages(msgs: Message*): Reaction = appendMessages(msgs)(empty)
152 
153     /**
154      * A Reaction created with:
155      *   - a function <b>S => S</b> with any state to a specific State.
156      *   - empty messages.
157      * @param s
158      *   The State provided from any state.
159      * @return
160      *   A new <b>Reaction</b> as described above.
161      */
162     def pure(s: S): Reaction = create(_ => s, Seq.empty)
163 
164   }
165 
166   /**
167    * An object used to create the Reaction part composed solely by the function [[S]] => [[S]].
168    */
169   object Update {
170 
171     /**
172      * An Update [[U]] is the function [[S]] => [[S]].
173      */
174     private type U = S => S
175 
176     /**
177      * Concatenate more [[U]].
178      * @param fs
179      *   all the [[U]] to concatenate.
180      * @return
181      *   a [[U]] that is the reduction of [[U]] cited above, function identity of [[S]] if [[U]] are
182      *   not presents.
183      */
184     def apply(fs: U*): U = fs.reduceOption(_ andThen _) getOrElse identity[S]
185 
186     /**
187      * The [[U]] empty value.
188      * @return
189      *   the identity of [[S]].
190      */
191     def empty: U = identity[S]
192   }
