13 /**
14  * Extension for the model. Adds some common implementation of the [[BehaviorBasedModel]].
15  */
16 trait ReactionUtilsExt extends BehaviorBasedModel {
17 
18   /**
19    * A class that wrap a reaction and it map and flatmap methods.
20    * @param r
21    *   The wrapped <b>Reaction</b>.
22    */
23   implicit class EnhancedReaction(r: Reaction) {
24 
25     /**
26      * Use <b>Reaction.map</b> method with this and an other <b>Reaction</b>.
27      * @param g
28      *   The other <b>Reaction</b>.
29      * @return
30      *   A new <b>Reaction</b> that is the <b>Reaction::map</b> of the two reactions.
31      */
32     def map(g: S => S): Reaction = Reaction.map(g)(r)
33 
34     /**
35      * Use <b>Reaction::flatmap</b> method with this and an other <b>Reaction</b>.
36      * @param g
37      *   The other <b>Reaction</b>.
38      * @return
39      *   A new <b>Reaction</b> that is the <b>Reaction::flatmap</b> of the two reactions.
40      */
41     def flatMap(g: S => Reaction): Reaction = Reaction.flatMap(g)(r)
42   }
43 
44   /**
45    * Object with some methods to work on <b>Reaction</b>.
46    */
47   object Reaction {
48 
49     /**
50      * Append a specific [[Message]] to a <b>Reaction</b>.
51      * @param msg
52      *   The <b>message</b> to be appended.
53      * @param r
54      *   The <b>Reaction</b> that will have the message appended.
55      * @return
56      *   A new <b>Reaction</b> with the <b>message</b> appended.
57      */
58     def appendMessage(msg: Message)(r: Reaction): Reaction = appendMessages(Seq(msg))(r)
59 
60     /**
61      * Append some specific [[Message]] s to a <b>Reaction</b>.
62      * @param msgs
63      *   The <b>messages</b> to be appended.
64      * @param r
65      *   The <b>Reaction</b> that will have the messages appended.
66      * @return
67      *   A new <b>Reaction</b> with the <b>messages</b> appended.
68      */
69     def appendMessages(msgs: Seq[Message])(r: Reaction): Reaction =
70       s => {
71         val (s1, m1) = r(s)
72         (s1, m1 ++ msgs)
73       }
74 
75     /**
76      * Create a <b>Reaction</b> from a function <b>State => State</b> and some <b>messages</b>.
77      * @param f
78      *   A function <b>State => State</b> that giving a State provide an updated on it.
79      * @param messages
80      *   The <b>messages</b> of <b>Reaction</b>.
81      * @return
82      *   A new <b>Reaction</b> with as function <b>f</b> and for messages <b>messages</b>.
83      */
84     def apply(f: S => S, messages: Message*): Reaction = create(f, messages)
85 
86     /**
87      * Combine two <b>Reaction</b>s using the <b>Reaction::flatmap</b>.
88      * @param a
89      *   A <b>Reaction</b>.
90      * @param b
91      *   An other <b>Reaction</b>.
92      * @return
93      *   The new <b>Reaction</b>, result combining of the precedent ones.
94      */
95     def combine(a: Reaction, b: Reaction): Reaction = a.flatMap(_ => b)
96 
97     /**
98      * Create a <b>Reaction</b> from a function <b>State => State</b> and some <b>messages</b>.
99      * @param f
100      *   A function <b>State => State</b> that giving a State provide an updated on it.
101      * @param messages
102      *   The <b>messages</b> of <b>Reaction</b>.
103      * @return
104      *   The new <b>Reaction</b> as function <b>f</b> and for messages <b>messages</b>.
105      */
106     def create(f: S => S, messages: Seq[Message]): Reaction = s => (f(s), messages)
107 
108     /**
109      * Create an <b>empty Reaction</b>:
110      *   - no message.
111      *   - function identity [[S]] => [[S]]
112      * @return
113      *   The empty <b>Reaction</b>.
114      */
115     def empty: Reaction = create(identity[S], Seq.empty)
116 
117     /**
118      * The <b>flatmap</b> implemented for <b>Reactions</b>:
119      *   - product a <b>State</b> applying the two <b>Reactions</b>
120      *   - concatenate as message the two <b>messages</b> of the Reactions.
121      * @param g
122      *   A function <b>State => Reaction</b>.
123      * @param r
124      *   A <b>Reaction</b>.
125      * @return
126      *   A new <b>Reaction</b> combining the function <b>g</b> and the reaction <b>r</b>.
127      */
128     def flatMap(g: S => Reaction)(r: Reaction): Reaction = { s =>
129       val (s1, m1) = r(s)
130       val (s2, m2) = g(s1)(s1)
131       (s2, m1 ++ m2)
132     }
133 
134     /**
135      * The <b>map</b> implemented for <b>Reactions</b>:
136      *   - product a <b>State</b> applying a <b>Reaction</b> on a function <b>State => State</b>
137      *   - concatenate as message the two <b>messages</b> of the <b>Reactions</b>.
138      * @param g
139      *   A function <b>State => Reaction</b>.
140      * @param r
141      *   A <b>Reaction</b>.
142      * @return
143      *   A new <b>Reaction</b> combining the function <b>g</b> and the reaction <b>r</b>.
144      */
145     def map(g: S => S)(r: Reaction): Reaction =
146       s => {
147         val (s1, m1) = r(s)
148         (g(s1), m1)
149       }
150 
151     /**
152      * Add <b>messages</b> to an <b>empty Reaction</b>.
153      * @param msgs
154      *   <b>Messages</b> to append to <b>Reaction</b>.
155      * @return
156      *   The <b>empty Reaction</b> with the <b>messages</b>.
157      */
158     def messages(msgs: Message*): Reaction = appendMessages(msgs)(empty)
159 
160     /**
161      * A Reaction created with:
162      *   - a function <b>S => S</b> with any state to a specific State.
163      *   - empty messages.
164      * @param s
165      *   The State provided from any state.
166      * @return
167      *   A new <b>Reaction</b> as described above.
168      */
169     def pure(s: S): Reaction = create(_ => s, Seq.empty)
170 
171   }
172 
173   /**
174    * An object used to create the Reaction part composed solely by the function [[S]] => [[S]].
175    */
176   object Update {
177 
178     /**
179      * An Update [[U]] is the function [[S]] => [[S]].
180      */
181     private type U = S => S
182 
183     /**
184      * Concatenate more [[U]].
185      * @param fs
186      *   all the [[U]] to concatenate.
187      * @return
188      *   a [[U]] that is the reduction of [[U]] cited above, function identity of [[S]] if [[U]] are
189      *   not presents.
190      */
191     def apply(fs: U*): U = fs.reduceOption(_ andThen _) getOrElse identity[S]
192 
193     /**
194      * The [[U]] empty value.
195      * @return
196      *   the identity of [[S]].
197      */
198     def empty: U = identity[S]
199   }
