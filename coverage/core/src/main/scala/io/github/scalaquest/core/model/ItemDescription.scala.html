10 import scala.annotation.tailrec

12 /**
13  * A trait that is used for describe an Item. The description of the object should be univocal.
14  */
15 sealed trait ItemDescription
16 
17 /**
18  * The common name of the Item.
19  * @param name
20  *   of the Item.
21  */
22 final case class BaseItem(name: String) extends ItemDescription
23 
24 /**
25  * The description of the name.
26  * @param decoration
27  *   should be an adjective that adds specificity to the BaseItem name.
28  * @param item
29  *   could be:
30  *   - an other DecoratedItem to add some more specificity
31  *   - the BaseItem
32  */
33 final case class DecoratedItem(decoration: String, item: ItemDescription) extends ItemDescription
34 
35 /**
36  * Contain some methods to facilitates the build of an ItemDescription.
37  */
38 object ItemDescription {
39 
40   /**
41    * An object that contain some methods in order to add more declarativeness to the
42    * ItemDescription.
43    */
44   object dsl {
45 
46     /**
47      * A class with all the decorations for an ItemDescription.
48      * @param decorators
49      *   all the decorators
50      */
51     final case class DecoratorBuilder(decorators: List[String])
52 
53     /**
54      * Create a DecoratorBuilder with all the decorations. The first is mandatory.
55      * @param decorator0
56      *   the first decorator term.
57      * @param decorators
58      *   the other decorator terms.
59      * @return
60      */
61     def d(decorator0: String, decorators: String*): DecoratorBuilder =
62       DecoratorBuilder((decorator0 +: decorators).toList)
63 
64     /**
65      * Create an ItemDescription given the name and a DecoratorBuilder with all the decorations.
66      * @param decoratorBuilder
67      *   the decorationBuilder.
68      * @param base
69      *   the name of the ItemDescription.
70      * @return
71      *   the ItemDescription.
72      */
73     def i(decoratorBuilder: DecoratorBuilder, base: String): ItemDescription = {
74       def go(decorators: List[String]): ItemDescription =
75         decorators match {
76           case head :: next => DecoratedItem(head, go(next))
77           case Nil          => BaseItem(base)
78         }
79       go(decoratorBuilder.decorators)
80     }
81 
82     def i(base: String): ItemDescription = BaseItem(base)
83   }
84 
85   /**
86    * A method that facilitates the creation for ItemDescription avoiding the creation of all the
87    * Structures.
88    * @param base
89    *   the name for BaseItem.
90    * @param decorators
91    *   all the adjectives for
92    * @return
93    *   the ItemDescription.
94    */
95   def apply(base: String, decorators: String*): ItemDescription = {
96     def go(decorators: List[String]): ItemDescription =
97       decorators match {
98         case head :: next => DecoratedItem(head, go(next))
99         case Nil          => BaseItem(base)
100       }
101     go(decorators.toList)
102   }
103 
104   /**
105    * It individuates the base of an ItemDescription.
106    * @param d
107    *   the ItemDescription.
108    * @return
109    *   the BaseItem.
110    */
111   @tailrec
112   def base(d: ItemDescription): BaseItem =
113     d match {
114       case i: BaseItem         => i
115       case DecoratedItem(_, i) => base(i)
116     }
117 
118   /**
119    * It individuates all the decorations of an ItemDescription.
120    * @param d
121    *   the ItemDescription.
122    * @return
123    *   the all the adjectives.
124    */
125   def decorators(d: ItemDescription): Set[String] = {
126     @tailrec
127     def go(d: ItemDescription, acc: Set[String]): Set[String] =
128       d match {
129         case DecoratedItem(decoration, item) => go(item, acc + decoration)
130         case _                               => acc
131       }
132     go(d, Set())
133   }
134 
135   /**
136    * It individuates if two Items have a common part. In order to do this, there are two fundamental
137    * properties:
138    *   - items must have the same base.
139    *   - decorators must be subsets of each other.
140    * @param d1
141    *   first ItemDescription.
142    * @param d2
143    *   second ItemDescription.
144    * @return
145    */
146   def isSubset(d1: ItemDescription, d2: ItemDescription): Boolean =
147     base(d1) == base(d2) && decorators(d1).subsetOf(decorators(d2))
148 
149   /**
150    * Prepare in a String the description of an ItemDescription. Start with adjectives, end with
151    * BaseItem.
152    * @param itemDescription
153    *   the itemDescription that have to be printed.
154    * @param separator
155    *   the separation between each word printed.
156    * @return
157    *   the description.
158    */
159   def mkString(itemDescription: ItemDescription, separator: String = " "): String = {
160     def go: ItemDescription => String = {
161       case BaseItem(name)                 => name
162       case DecoratedItem(decorator, item) => s"${decorator}${separator}${go(item)}"
163     }
164     go(itemDescription)
165   }
166 
167   /**
168    * A class that add some more useful methods for ItemDescription.
169    * @param self
170    *   the ItemDescription wrapped into [[EnhancedItemDescription]].
171    */
172   implicit final class EnhancedItemDescription(self: ItemDescription) {
173 
174     /**
175      * Compare this ItemDescription with an other using ItemDescription method 'isSubset'.
176      * @param other
177      *   the other ItemDescription.
178      * @return
179      *   True if this is subset of the 'other' ItemDescription. False otherwise
180      */
181     def isSubset(other: ItemDescription): Boolean = ItemDescription.isSubset(self, other)
182 
183     /**
184      * The baseItem of this ItemDescription.
185      * @return
186      *   The baseItem of this ItemDescription.
187      */
188     def base: BaseItem = ItemDescription.base(self)
189 
190     /**
191      * The decorations of this ItemDescription.
192      * @return
193      *   The decorations of this ItemDescription.
194      */
195     def decorators: Set[String] = ItemDescription.decorators(self)
196 
197     /**
198      * The description of this ItemDescription.
199      * @param separator
200      *   used between each words.
201      * @return
202      *   the description.
203      */
204     def mkString(separator: String = " "): String = ItemDescription.mkString(self, separator)
205 
206     /**
207      * Used mkString with default separation.
208      * @return
209      *   The description of this ItemDescription.
210      */
211     def mkString: String = this.mkString()
212   }
