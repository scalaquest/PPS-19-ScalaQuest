8 /**
9  * The trait makes possible to mix into the StdCommonBehaviors the standard implementation of
10  * Openable.
11  */
12 trait SimpleOpenableExt extends CommonBase {
13 
14   /**
15    * Standard implementation of the common Openable behavior.
16    *
17    * This is a behavior associated to an Item that can be opened and closed. Open and close actions
18    * are mutually exclusive. The standard Reactions consists into changing an internal variable that
19    * indicates the "openness" of the Openable, when Statements "open Item (with something)", "close
20    * Item (with something)" are called.
21    * @param _isOpen
22    *   The initial openness state of the behavior. Default to closed False.
23    * @param requiredKey
24    *   The key required to open the Item. If not passed the Item can be open without Key.
25    * @param onOpenExtra
26    *   Reaction to be executed into the State when opened, after the standard Reaction. It can be
27    *   omitted.
28    * @param onCloseExtra
29    *   Reaction to be executed into the State when closed, after the standard Reaction. It can be
30    *   omitted.
31    */
32   case class SimpleOpenable(
33     var _isOpen: Boolean = false,
34     requiredKey: Option[Key] = None,
35     onOpenExtra: Option[Reaction] = None,
36     onCloseExtra: Option[Reaction] = None
37   )(implicit
38     messageLens: Lens[S, Seq[Message]]
39   ) extends Openable {
40 
41     override def isOpen: Boolean = _isOpen
42 
43     override def triggers: ItemTriggers = {
44       // "Open the item (with something)"
45       case (Open, item, maybeKey, state)
46           if state.isInCurrentRoom(item) && canBeOpened(state, maybeKey) && !isOpen =>
47         open(item)
48 
49       // "Close the item (with something)"
50       case (Close, item, maybeKey, state)
51           if state.isInCurrentRoom(item) && canBeOpened(state, maybeKey) && isOpen =>
52         close(item)
53     }
54 
55     /**
56      * Checks whether the Item can be opened with the current configuration. The usedKey is the
57      * eventual Key passed explicitly by the user, with a statement like "open the item with the
58      * key".
59      *
60      * @param state
61      *   The current State.
62      * @param usedKey
63      *   The eventual key passed explicitly by the user, e.g. , with a statement like "open the item
64      *   with the key".
65      * @return
66      *   True if:
67      *   - A key was passed explicitly, is effectively into the Bag of the Player (or in the current
68      *     room) and that key is effectively required to open the Item.
69      *   - No key was passed explicitly, is effectively into the Bag of the Player and that key is
70      *     effectively required to open the Item.
71      *   - No key was passed explicitly, and no key is required.
72      *
73      * False otherwise.
74      */
75     def canBeOpened(state: S, usedKey: Option[I] = None): Boolean = {
76       usedKey match {
77         case Some(key) => requiredKey.contains(key) && state.isInScope(key)
78         case None      => requiredKey.fold(true)(state.isInBag(_))
79       }
80     }
81 
82     /**
83      * Sets the Item in an open state and executes eventual extra actions.
84      * @return
85      *   A Reaction that sets the Item in an open state and executes eventual extra actions.
86      */
87     def open(item: I): Reaction =
88       state => {
89         _isOpen = true
90         state.applyReactions(
91           messageLens.modify(_ :+ Opened(item)),
92           onOpenExtra.getOrElse(state => state)
93         )
94       }
95 
96     /**
97      * Sets the Item in a closed state and executes eventual extra actions.
98      * @return
99      *   A Reaction that sets the Item in a closed state and executes eventual extra actions.
100      */
101     def close(item: I): Reaction =
102       state => {
103         _isOpen = false
104         state.applyReactions(
105           messageLens.modify(_ :+ Closed(item)),
106           onCloseExtra.getOrElse(state => state)
107         )
108       }
109   }
110 }
