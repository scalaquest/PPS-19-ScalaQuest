17 /**
18  * The trait makes possible to add into the [[BehaviorBasedModel]] the <b>Openable</b> behavior.
19  */
20 trait OpenableExt
21   extends BehaviorBasedModel
22   with KeyExt
23   with CMessagesExt
24   with CReactionsExt
25   with StateUtilsExt {
26 
27   /**
28    * An <b>ItemBehavior</b> associated to a <b>BehaviorBasedItem</b> that can be opened a single
29    * time.
30    */
31   abstract class Openable extends ItemBehavior {
32 
33     /**
34      * The "openness" state of the subject, as a boolean value. Initially closed.
35      * @return
36      *   True if subject is already opened, False otherwise.
37      */
38     def isOpen: Boolean
39 
40     /**
41      * The <b>Key</b> to be used to open the subject. If the subject can be opened without
42      * <b>Key</b>, it can be set to [[None]].
43      * @return
44      *   [[Some]] with the <b>Key</b> if key is required, [[None]] otherwise.
45      */
46     def requiredKey: Option[Key]
47 
48     /**
49      * Checks whether the subject can be opened with the current configuration.
50      *
51      * @param state
52      *   The current <b>State</b>.
53      * @param usedKey
54      *   The eventual key passed explicitly by the user, e.g. , with a statement like "open the item
55      *   with the key".
56      * @return
57      *   True if:
58      *   - A <b>Key</b> was passed explicitly, is effectively into the bag of the player (or in the
59      *     location) and that <b>Key</b> is effectively required to open the subject.
60      *   - No <b>Key</b> was passed explicitly, is effectively into the bag of the player and that
61      *     <b>Key</b> is effectively required to open the subject.
62      *   - No <b>Key</b> was passed explicitly, and no <b>Key</b> is required.
63      *
64      * False otherwise.
65      */
66     def canBeOpened(usedKey: Option[I])(implicit state: S): Boolean
67 
68     /**
69      * A <b>Reaction</b> that sets the subject in an open state.
70      * @return
71      *   A <b>Reaction</b> that sets the subject in an open state.
72      */
73     def open: Reaction
74   }
75 
76   /**
77    * Standard implementation of the <b>Openable</b> behavior.
78    *
79    * This is a behavior associated to an Item that can be opened a single time. The standard
80    * <b>Reaction</b> consists into changing an internal variable that indicates the "openness" of
81    * the subject, when <b>Statement</b> "open subject (with something)" is called.
82    * @param requiredKey
83    *   The <b>Key</b> required to open the Item. If not passed, the Item does not require a
84    *   <b>Key</b> to be opened.
85    * @param onOpenExtra
86    *   <b>Reaction</b> to be executed into the State when opened, after the standard
87    *   <b>Reaction</b>. It can be omitted.
88    */
89   case class SimpleOpenable(
90     requiredKey: Option[Key] = None,
91     onOpenExtra: Reaction = Reaction.empty
92   )(implicit val subject: I)
93     extends Openable {
94     var _isOpen: Boolean = false
95 
96     override def isOpen: Boolean = _isOpen
97 
98     override def triggers: ItemTriggers = {
99       case (Open, maybeKey, state)
100           if state.isInLocation(subject) && canBeOpened(maybeKey)(state) && !isOpen =>
101         open
102 
103       case (Open, _, _) =>
104         if (!isOpen)
105           CReactions.addMessage(CMessages.FailedToOpen(subject))
106         else
107           CReactions.addMessage(CMessages.AlreadyOpened(subject))
108     }
109 
110     override def canBeOpened(usedKey: Option[I] = None)(implicit state: S): Boolean = {
111       usedKey match {
112         case Some(key) => requiredKey.contains(key)
113         case None      => requiredKey.fold(true)(_ => false)
114       }
115     }
116 
117     override def open: Reaction = {
118       _isOpen = true
119 
120       for {
121         _ <-
122           if (requiredKey.exists(_.disposable)) CReactions.modifyBag(_ - requiredKey.get.ref)
123           else Reaction.empty
124         _ <-
125           if (requiredKey.exists(_.disposable))
126             CReactions.modifyLocationItems(_ - requiredKey.get.ref)
127           else Reaction.empty
128         _ <- CReactions.addMessage(CMessages.Opened(subject))
129         s <- onOpenExtra
130       } yield s
131     }
132   }
133 
134   /**
135    * Companion object for <b>Openable</b>.
136    */
137   object Openable {
138 
139     /**
140      * Builder for an <b>Openable</b> that requires a specific <b>Key</b> to open the subject.
141      * @param requiredKey
142      *   The key required in order to open the subject.
143      * @param onOpenExtra
144      *   An extra <b>ItemBehavior</b> generated when the subject is opened. It can be omitted.
145      * @return
146      *   A builder for an <b>Openable</b> that requirea a specific <b>Key</b> to open the subject.
147      */
148     def lockedBuilder(
149       requiredKey: Key,
150       onOpenExtra: Reaction = Reaction.empty
151     ): I => Openable = SimpleOpenable(Some(requiredKey), onOpenExtra)(_)
152 
153     /**
154      * Builder for an <b>Openable</b> that not requires a specific <b>Key</b> to open the subject.
155      * @param onOpenExtra
156      *   An extra <b>ItemBehavior</b> generated when the subject is opened. It can be omitted.
157      * @return
158      *   A builder for an <b>Openable</b> that not requires a specific <b>Key</b> to open the
159      *   subject.
160      */
161     def unlockedBuilder(
162       onOpenExtra: Reaction = Reaction.empty
163     ): I => Openable = SimpleOpenable(None, onOpenExtra)(_)
164   }
