31   /**
32    * Standard implementation of the common Openable behavior.
33    *
34    * This is a behavior associated to an Item that can be opened a single time. Open The standard
35    * [[Reaction]] consists into changing an internal variable that indicates the "openness" of the
36    * Openable, when Statement "open Item (with something)" ais called.
37    * @param requiredKey
38    *   The key required to open the Item. If not passed the Item can be opened without Key.
39    * @param onOpenExtra
40    *   Reaction to be executed into the State when opened, after the standard Reaction. It can be
41    *   omitted.
42    */
43   case class SimpleOpenable(
44     requiredKey: Option[Key] = None,
45     consumeKey: Boolean = false,
46     onOpenExtra: Option[Reaction] = None
47   )(implicit subject: I)
48     extends Openable {
49     var _isOpen: Boolean = false
50 
51     override def isOpen: Boolean = _isOpen
52 
53     override def triggers: ItemTriggers = {
54       // "Open the item (with something)"
55       case (Open, item, maybeKey, state)
56           if state.isInLocation(item) && canBeOpened(maybeKey)(state) && !isOpen =>
57         open
58     }
59 
60     /**
61      * Checks whether the Item can be opened with the current configuration. The usedKey is the
62      * eventual Key passed explicitly by the user, with a statement like "open the item with the
63      * key".
64      *
65      * @param state
66      *   The current State.
67      * @param usedKey
68      *   The eventual key passed explicitly by the user, e.g. , with a statement like "open the item
69      *   with the key".
70      * @return
71      *   True if:
72      *   - A key was passed explicitly, is effectively into the Bag of the Player (or in the current
73      *     room) and that key is effectively required to open the Item.
74      *   - No key was passed explicitly, is effectively into the Bag of the Player and that key is
75      *     effectively required to open the Item.
76      *   - No key was passed explicitly, and no key is required.
77      *
78      * False otherwise.
79      */
80     override def canBeOpened(usedKey: Option[I] = None)(implicit state: S): Boolean = {
81       usedKey match {
82         case Some(key) => requiredKey.contains(key) && state.isInScope(key)
83         case None      => requiredKey.fold(true)(state.isInBag(_))
84       }
85     }
86 
87     /**
88      * Sets the Item in an open state and executes eventual extra actions.
89      * @return
90      *   A Reaction that sets the Item in an open state and executes eventual extra actions.
91      */
92     def open: Reaction =
93       state => {
94         _isOpen = true
95 
96         state.applyReactions(
97           Reactions.open(subject, requiredKey, consumeKey),
98           onOpenExtra.getOrElse(Reactions.empty)
99         )
100       }
101   }
102 
103   /**
104    * Companion object for [[Openable]]. Shortcut for the standard implementation.
105    */
106   object Openable {
107 
108     def builder(
109       requiredKey: Option[Key] = None,
110       consumeKey: Boolean = true,
111       onOpenExtra: Option[Reaction] = None
112     ): I => Openable = SimpleOpenable(requiredKey, consumeKey, onOpenExtra)(_)
113   }
