28   /**
29    * Standard implementation of the common Openable behavior.
30    *
31    * This is a behavior associated to an Item that can be opened a single time. Open The standard
32    * [[Reaction]] consists into changing an internal variable that indicates the "openness" of the
33    * Openable, when Statement "open Item (with something)" ais called.
34    * @param requiredKey
35    *   The key required to open the Item. If not passed the Item can be opened without Key.
36    * @param onOpenExtra
37    *   Reaction to be executed into the State when opened, after the standard Reaction. It can be
38    *   omitted.
39    */
40   case class SimpleOpenable(
41     requiredKey: Option[Key] = None,
42     onOpenExtra: Reaction = Reaction.empty
43   )(implicit subject: I)
44     extends Openable {
45     var _isOpen: Boolean = false
46 
47     override def isOpen: Boolean = _isOpen
48 
49     override def triggers: ItemTriggers = {
50       case (Open, item, maybeKey, state)
51           if state.isInLocation(item) && canBeOpened(maybeKey)(state) && !isOpen =>
52         open
53 
54       case (Open, _, _, _) if !isOpen => failToOpen
55       case (Open, _, _, _) if isOpen  => alreadyOpened
56     }
57 
58     /**
59      * Checks whether the Item can be opened with the current configuration. The usedKey is the
60      * eventual Key passed explicitly by the user, with a statement like "open the item with the
61      * key".
62      *
63      * @param state
64      *   The current State.
65      * @param usedKey
66      *   The eventual key passed explicitly by the user, e.g. , with a statement like "open the item
67      *   with the key".
68      * @return
69      *   True if:
70      *   - A key was passed explicitly, is effectively into the Bag of the Player (or in the current
71      *     room) and that key is effectively required to open the Item.
72      *   - No key was passed explicitly, is effectively into the Bag of the Player and that key is
73      *     effectively required to open the Item.
74      *   - No key was passed explicitly, and no key is required.
75      *
76      * False otherwise.
77      */
78     override def canBeOpened(usedKey: Option[I] = None)(implicit state: S): Boolean = {
79       usedKey match {
80         case Some(key) => requiredKey.contains(key)
81         case None      => requiredKey.fold(true)(_ => false)
82       }
83     }
84 
85     /**
86      * Sets the Item in an open state and executes eventual extra actions.
87      * @return
88      *   A Reaction that sets the Item in an open state and executes eventual extra actions.
89      */
90     def open: Reaction = {
91       _isOpen = true
92 
93       Reaction.combine(
94         Reactions.open(subject, requiredKey),
95         onOpenExtra
96       )
97     }
98 
99     def failToOpen: Reaction = Reaction.messages(Messages.FailedToOpen(subject))
100 
101     def alreadyOpened: Reaction = Reaction.messages(Messages.AlreadyOpened(subject))
102   }
103 
104   /**
105    * Companion object for [[Openable]]. Shortcut for the standard implementation.
106    */
107   object Openable {
108 
109     def lockedBuilder(
110       requiredKey: Key,
111       onOpenExtra: Reaction = Reaction.empty
112     ): I => Openable = SimpleOpenable(Some(requiredKey), onOpenExtra)(_)
113 
114     def unlockedBuilder(
115       onOpenExtra: Reaction = Reaction.empty
116     ): I => Openable = SimpleOpenable(None, onOpenExtra)(_)
117   }
