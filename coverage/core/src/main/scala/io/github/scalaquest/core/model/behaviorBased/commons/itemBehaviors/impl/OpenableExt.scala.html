20   /**
21    * An <b>ItemBehavior</b> associated to a <b>BehaviorBasedItem</b> that can be opened a single
22    * time.
23    */
24   abstract class Openable extends ItemBehavior {
25 
26     /**
27      * The "openness" state of the subject, as a boolean value. Initially closed.
28      * @return
29      *   True if subject is already opened, False otherwise.
30      */
31     def isOpen: Boolean
32 
33     /**
34      * The <b>Key</b> to be used to open the subject. If the subject can be opened without
35      * <b>Key</b>, it can be set to [[None]].
36      * @return
37      *   [[Some]] with the <b>Key</b> if key is required, [[None]] otherwise.
38      */
39     def requiredKey: Option[Key]
40 
41     /**
42      * Checks whether the subject can be opened with the current configuration.
43      *
44      * @param state
45      *   The current <b>State</b>.
46      * @param usedKey
47      *   The eventual key passed explicitly by the user, e.g. , with a statement like "open the item
48      *   with the key".
49      * @return
50      *   True if:
51      *   - A <b>Key</b> was passed explicitly, is effectively into the bag of the player (or in the
52      *     location) and that <b>Key</b> is effectively required to open the subject.
53      *   - No <b>Key</b> was passed explicitly, is effectively into the bag of the player and that
54      *     <b>Key</b> is effectively required to open the subject.
55      *   - No <b>Key</b> was passed explicitly, and no <b>Key</b> is required.
56      *
57      * False otherwise.
58      */
59     def canBeOpened(usedKey: Option[I])(implicit state: S): Boolean
60 
61     /**
62      * A <b>Reaction</b> that sets the subject in an open state.
63      * @return
64      *   A <b>Reaction</b> that sets the subject in an open state.
65      */
66     def open: Reaction
67   }
68 
69   /**
70    * Standard implementation of the <b>Openable</b> behavior.
71    *
72    * This is a behavior associated to an Item that can be opened a single time. The standard
73    * <b>Reaction</b> consists into changing an internal variable that indicates the "openness" of
74    * the subject, when <b>Statement</b> "open subject (with something)" is called.
75    * @param requiredKey
76    *   The <b>Key</b> required to open the Item. If not passed, the Item does not require a
77    *   <b>Key</b> to be opened.
78    * @param onOpenExtra
79    *   <b>Reaction</b> to be executed into the State when opened, after the standard
80    *   <b>Reaction</b>. It can be omitted.
81    */
82   case class SimpleOpenable(
83     requiredKey: Option[Key] = None,
84     onOpenExtra: Reaction = Reaction.empty
85   )(implicit val subject: I)
86     extends Openable {
87     var _isOpen: Boolean = false
88 
89     override def isOpen: Boolean = _isOpen
90 
91     override def triggers: ItemTriggers = {
92       case (Open, maybeKey, state)
93           if state.isInLocation(subject) && canBeOpened(maybeKey)(state) && !isOpen =>
94         open
95 
96       case (Open, _, _) =>
97         if (!isOpen)
98           CReactions.addMessage(CMessages.FailedToOpen(subject))
99         else
100           CReactions.addMessage(CMessages.AlreadyOpened(subject))
101     }
102 
103     override def canBeOpened(usedKey: Option[I] = None)(implicit state: S): Boolean = {
104       usedKey match {
105         case Some(key) => requiredKey.contains(key)
106         case None      => requiredKey.fold(true)(_ => false)
107       }
108     }
109 
110     override def open: Reaction = {
111       _isOpen = true
112 
113       for {
114         _ <-
115           if (requiredKey.exists(_.disposable)) CReactions.modifyBag(_ - requiredKey.get.ref)
116           else Reaction.empty
117         _ <-
118           if (requiredKey.exists(_.disposable))
119             CReactions.modifyLocationItems(_ - requiredKey.get.ref)
120           else Reaction.empty
121         _ <- CReactions.addMessage(CMessages.Opened(subject))
122         s <- onOpenExtra
123       } yield s
124     }
125   }
126 
127   /**
128    * Companion object for <b>Openable</b>.
129    */
130   object Openable {
131 
132     /**
133      * Builder for an <b>Openable</b> that requires a specific <b>Key</b> to open the subject.
134      * @param requiredKey
135      *   The key required in order to open the subject.
136      * @param onOpenExtra
137      *   An extra <b>ItemBehavior</b> generated when the subject is opened. It can be omitted.
138      * @return
139      *   A builder for an <b>Openable</b> that requirea a specific <b>Key</b> to open the subject.
140      */
141     def lockedBuilder(
142       requiredKey: Key,
143       onOpenExtra: Reaction = Reaction.empty
144     ): I => Openable = SimpleOpenable(Some(requiredKey), onOpenExtra)(_)
145 
146     /**
147      * Builder for an <b>Openable</b> that not requires a specific <b>Key</b> to open the subject.
148      * @param onOpenExtra
149      *   An extra <b>ItemBehavior</b> generated when the subject is opened. It can be omitted.
150      * @return
151      *   A builder for an <b>Openable</b> that not requires a specific <b>Key</b> to open the
152      *   subject.
153      */
154     def unlockedBuilder(
155       onOpenExtra: Reaction = Reaction.empty
156     ): I => Openable = SimpleOpenable(None, onOpenExtra)(_)
157   }
