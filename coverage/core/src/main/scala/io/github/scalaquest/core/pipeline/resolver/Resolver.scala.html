7 /**
8  * A pipeline component that takes a [[AbstractSyntaxTree]] (wrapped into an [[ParserResult]] ) and
9  * returns a [[Statement]] wrapped into an [[ResolverResult]]. The execution may fail, when the
10  * given [[AbstractSyntaxTree]] has not a match with the [[Model]] components.
11  */
12 trait Resolver {
13 
14   /**
15    * Triggers the [[Resolver]] execution.
16    * @param parserResult
17    *   A wrapper for the input [[AbstractSyntaxTree]].
18    * @return
19    *   An [[Either]] describing the resolver result. If the [[Resolver]] fails, the result is a
20    *   [[Left]] describing what went wrong. Otherwise, it is a [[Right]] with the [[ResolverResult]]
21    *   (wrapper for a [[Statement]] ).
22    */
23   def resolve(parserResult: ParserResult): Either[String, ResolverResult]
24 }
25 
26 /**
27  * Companion object for the [[Resolver]] trait. It exposes the [[Resolver::apply()]] to instantiate
28  * the [[Resolver]] .
29  */
30 object Resolver {
31 
32   type Builder[S] = S => Resolver
33 
34   abstract class SimpleResolver extends Resolver {
35 
36     def actions: PartialFunction[VerbPrep, Action]
37 
38     def items: PartialFunction[ItemDescription, ItemRef]
39 
40     def retrieveAction(verbPrep: VerbPrep): Either[String, Action] =
41       actions lift verbPrep toRight s"Couldn't understand ${verbPrep._1}."
42 
43     def retrieveItem(name: ItemDescription): Either[String, ItemRef] =
44       items lift name toRight s"Couldn't understand $name"
45 
46     override def resolve(parserResult: ParserResult): Either[String, ResolverResult] = {
47       for {
48         statement <- parserResult.tree match {
49           case AbstractSyntaxTree.Intransitive(verb, prep, _) =>
50             for {
51               action <- retrieveAction((verb, prep))
52             } yield Statement.Intransitive(action)
53 
54           case AbstractSyntaxTree.Transitive(verb, prep, _, obj) =>
55             for {
56               action  <- retrieveAction((verb, prep))
57               itemRef <- retrieveItem(obj)
58             } yield Statement.Transitive(action, itemRef)
59 
60           case AbstractSyntaxTree.Ditransitive(verb, prep, _, directObj, indirectObj) =>
61             for {
62               action          <- retrieveAction((verb, prep))
63               directItemRef   <- retrieveItem(directObj)
64               indirectItemRef <- retrieveItem(indirectObj)
65             } yield Statement.Ditransitive(action, directItemRef, indirectItemRef)
66 
67           case _ => Left("The statement is wrong.")
68         }
69       } yield ResolverResult(statement)
70     }
71 
72   }
73 
74   def builder[M <: Model](implicit model: M): Builder[model.S] =
75     s =>
76       new SimpleResolver {
77 
78         override def actions: PartialFunction[VerbPrep, Action] = s.actions
79 
80         override def items: PartialFunction[ItemDescription, ItemRef] =
81           d =>
82             s.matchState.itemsInScope.filter(i => d.isSubset(i.description)).toList match {
83               case x :: Nil => x.ref
84             }
85       }
