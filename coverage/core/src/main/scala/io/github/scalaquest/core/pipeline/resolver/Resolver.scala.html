13 /**
14  * A pipeline component that takes a [[AbstractSyntaxTree]] (wrapped into an [[ParserResult]] ) and
15  * returns a [[Statement]] wrapped into an [[ResolverResult]]. The execution may fail, when the
16  * given [[AbstractSyntaxTree]] has not a match with the [[Model]] components.
17  */
18 trait Resolver {
19 
20   /**
21    * Triggers the [[Resolver]] execution.
22    * @param parserResult
23    *   A wrapper for the input [[AbstractSyntaxTree]].
24    * @return
25    *   An [[Either]] describing the resolver result. If the [[Resolver]] fails, the result is a
26    *   [[Left]] describing what went wrong. Otherwise, it is a [[Right]] with the [[ResolverResult]]
27    *   (wrapper for a [[Statement]] ).
28    */
29   def resolve(parserResult: ParserResult): Either[String, ResolverResult]
30 }
31 
32 /**
33  * Companion object for the [[Resolver]] trait. It exposes the [[Resolver::apply()]] to instantiate
34  * the [[Resolver]] .
35  */
36 object Resolver {
37 
38   type Builder[S] = S => Resolver
39 
40   def builder[M <: Model](implicit model: M): Builder[model.S] =
41     s =>
42       new AbstractSyntaxTreeResolver {
43 
44         override def actions(v: (String, Option[String])): Either[String, Action] =
45           s.actions get v toRight s"I couldn't understand what you said."
46 
47         override def items(d: ItemDescription): Either[String, ItemRef] =
48           s.scope.filter(i => d.isSubset(i.description)).toList match {
49             case x :: Nil => Right(x.ref)
50             case _ :: _   => Left(s"Which ${d.mkString} are you talking about?")
51             case _        => Left(s"I can't see any ${d.mkString}.")
52           }
53       }
54 }
