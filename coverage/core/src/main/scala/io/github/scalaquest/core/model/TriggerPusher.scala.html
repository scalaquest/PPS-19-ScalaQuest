22 /**
23  * A class that aims to interpret [[Message]] s and transform them in an output for the user.
24  *
25  * The output could be of any type in the base implementation. If [[Message]] is defined in
26  * <b>MessageTriggers</b> it will trigger the specific answer, otherwise return <b>not found</b>
27  * @tparam A
28  *   the concrete type that will shown to the user.
29  */
30 abstract class TriggerPusher[A] extends MessagePusher[A] {
31 
32   /**
33    * A [[PartialFunction]] used to find a match for the given input.
34    * @return
35    *   A [[PartialFunction]] used to find a match for the given input.
36    */
37   def triggers: MessageTriggers[A] = PartialFunction.empty
38 
39   /**
40    * An output returned if the message is not recognized.
41    * @return
42    *   An output returned if the message is not recognized.
43    */
44   def notFound: A
45 
46   /**
47    * Finds a match for the given [[Message]] analyzing the [[TriggerPusher::triggers]].
48    *
49    * @param input
50    *   the [[Message]] to control.
51    * @return
52    *   An output matching with the given message; [[TriggerPusher::notFound]] otherwise.
53    */
54   private final def eval(input: Message): Option[A] = {
55     triggers.lift(input)
56   }
57 
58   /**
59    * Define how two [[A]] have to be combined.
60    */
61   def combine(x: A, y: A): A
62 
63   override final def push(input: Seq[Message]): A =
64     input.flatMap(eval).reduceOption(combine).getOrElse(notFound)
65 }

83 /**

85  * [[MessageTriggers]].
86  *
87  * This is commonly used when it is needed to define some base extensible triggers, that should be
88  * easily extended letting the user defining the extras. When the triggers are effectively used,
89  * extras will be privileged, as the extra should always overwrite the base.

105 
106   /**
107    * All the triggers with the right order.
108    * @return
109    *   the chosen answer.
110    */
111   override def triggers: MessageTriggers[A] = Seq(extra, base).reduce(_ orElse _)
112 }
