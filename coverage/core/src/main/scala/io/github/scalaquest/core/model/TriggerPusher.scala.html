10 /**
11  * A class that aims to interpret [[Message]] s and transform them in an output for the user.
12  *
13  * The output could be of any type in the base implementation. If [[Message]] is defined in
14  * <b>MessageTriggers</b> it will trigger the specific answer, otherwise return <b>not found</b>
15  * @tparam A
16  *   the concrete type that will shown to the user.
17  */
18 abstract class TriggerPusher[A] extends MessagePusher[A] {
19 
20   /**
21    * A [[PartialFunction]] used to find a match for the given input.
22    * @return
23    *   A [[PartialFunction]] used to find a match for the given input.
24    */
25   def triggers: MessageTriggers[A] = PartialFunction.empty
26 
27   /**
28    * An output returned if the message is not recognized.
29    * @return
30    *   An output returned if the message is not recognized.
31    */
32   def notFound: A
33 
34   /**
35    * Finds a match for the given [[Message]] analyzing the [[TriggerPusher::triggers]].
36    *
37    * @param input
38    *   the [[Message]] to control.
39    * @return
40    *   An output matching with the given message; [[TriggerPusher::notFound]] otherwise.
41    */
42   private final def eval(input: Message): Option[A] = {
43     triggers.lift(input)
44   }
45 
46   def combine(x: A, y: A): A
47 
48   final def push(input: Seq[Message]): A =
49     input.flatMap(eval).reduceOption(combine).getOrElse(notFound)
50 }

68 /**
69  * A mixin for [[TriggerPusher]] s, that enables the possibility to define two levels of
70  * [[MessageTriggers]].
71  *
72  * This is commonly used when it is needed to define some base extensible triggers, that should be
73  * easily extended letting the user defining the extras. When the triggers are effectively used,
74  * extras will be privileged, as the extra should always overwrite the base.
75  */
76 trait Composable[A] extends TriggerPusher[A] {
77 
78   /**
79    * The extra [[MessageTriggers]] of the pusher.
80    * @return
81    *   The extra [[MessageTriggers]].
82    */
83   def extra: MessageTriggers[A] = PartialFunction.empty
84 
85   /**
86    * The base [[MessageTriggers]] of the pusher.
87    * @return
88    */
89   def base: MessageTriggers[A]
90 
91   /**
92    * All the triggers with the right order.
93    * @return
94    *   the chosen answer.
95    */
96   override def triggers: MessageTriggers[A] = Seq(extra, base).reduce(_ orElse _)
97 }
