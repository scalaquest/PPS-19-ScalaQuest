5 /**
6  * Integrates some additional functionalities for state inspection and re-generation, by the use of
7  * [[monocle.Lens]].

9 trait StateUtilsExt extends Model {
10 
11   /**
12    * Add some utility methods for [[State]] implementation.
13    * @param state
14    *   A state ([[S]])
15    */
16   implicit class StateUtils(state: S) {
17     implicit val s: S = state
18 
19     /**
20      * Check if an [[Item]] is in Bag.
21      * @param item
22      *   The item ([[I]]) to check.
23      * @return
24      *   True if item is in bag, false otherwise.
25      */
26     def isInBag(item: I): Boolean = state.bag.contains(item)
27 
28     /**
29      * Check if an [[Item]] is in the actual player location [[Room]].
30      * @param item
31      *   The item ([[I]]) to check.
32      * @return
33      *   True if item is in location, false otherwise.
34      */
35     def isInLocation(item: I): Boolean = state.location.items.contains(item)
36 
37     /**
38      * Check if an [[Item]] is visible. Two possible conditions:
39      *   - item is in the actual player location [[Room]].
40      *   - item is in bag.
41      * @param item
42      *   the item ([[I]]) to check.
43      * @return
44      *   True if item is in location or in bag, false otherwise.
45      */
46     def isInScope(item: I): Boolean = state.isInLocation(item) || state.isInBag(item)
47 
48     /**
49      * Check if player have a neighbor [[Room]] for a given Direction.
50      * @param direction
51      *   The given [[Direction]].
52      * @return
53      *   [[Option]] that contain the neighbor Room ([[RM]]) if present, [[None]] otherwise.
54      */
55     def locationNeighbor(direction: Direction): Option[RM] = state.location.neighbor(direction)
56 
57     /**
58      * Responds with all the [[Item]] s present in the player location [[Room]].
59      * @return
60      *   All the [[Item]] s present in the player location [[Room]].
61      */
62     def locationItems: Set[I] = state.location.items
63 
64     /**
65      * Copy the State [[S]] and add an item to the player location Room.
66      * @param item
67      *   The <b>item</b> [[I]] to add.
68      * @return
69      *   A new <b>State</b> with the <b>item</b> added into the player's location.
70      */
71     def copyWithItemInLocation(item: I): S = {
72       val stateWithTarget    = itemsLens.modify(_ + (item.ref -> item))(state)
73       val currRoomWithTarget = roomItemsLens.modify(_ + item.ref)(state.location)
74       roomsLens.modify(_ + (currRoomWithTarget.ref -> currRoomWithTarget))(stateWithTarget)
75     }
76 
77     /**
78      * Copy the State [[S]] and add an item to the player's bag.
79      * @param item
80      *   The <b>item</b> [[I]] to add to the State.
81      * @return
82      *   The new <b>State</b> with the <b>item</b> in the player's <b>bag</b>.
83      */
84     def copyWithItemInBag(item: I): S = {
85       val stateWithTarget = itemsLens.modify(_ + (item.ref -> item))(state)
86       bagLens.modify(_ + item.ref)(stateWithTarget)
87     }
88   }
89 }
