29       val stateWithTarget    = itemsLens.modify(_ + (item.ref -> item))(state)
30       val currRoomWithTarget = roomItemsLens.modify(_ + item.ref)(state.location)
31       roomsLens.modify(_ + (currRoomWithTarget.ref -> currRoomWithTarget))(stateWithTarget)
32     }
33 
34     def copyWithItemInBag(item: I): S = {
35       val stateWithTarget = itemsLens.modify(_ + (item.ref -> item))(state)
36       bagLens.modify(_ + item.ref)(stateWithTarget)
37     }
38 
39     def applyReactions(reactions: Reaction*): S = Function.chain(reactions.toList)(this.state)

41   }
