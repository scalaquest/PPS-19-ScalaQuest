5 /**
6  * A class that aims to interpret [[Message]] s and transform them in an output for the user.
7  *
8  * The output could be of any type in the base implementation. If [[Message]] is defined in
9  * <b>MessageTriggers</b> it will trigger the specific answer, otherwise return <b>not found</b>
10  * @tparam A
11  *   the concrete type that will shown to the user.
12  */
13 abstract class MessagePusher[A] {
14 
15   /**
16    * A [[PartialFunction]] used to find a match for the given input.
17    * @return
18    *   A [[PartialFunction]] used to find a match for the given input.
19    */
20   def triggers: MessageTriggers[A]
21 
22   /**
23    * An output returned if the message is not recognized.
24    * @return
25    *   An output returned if the message is not recognized.
26    */
27   def notFound: A
28 
29   /**
30    * Finds a match for the given [[Message]] analyzing the [[MessagePusher::triggers]].
31    * @param input
32    *   the [[Message]] to control.
33    * @return
34    *   An output matching with the given message; [[MessagePusher::notFound]] otherwise.
35    */
36   def push(input: Message): A = {
37     triggers.lift(input).getOrElse(notFound)
38   }
39 
40   def push(input: Seq[Message]): A
41 }

59 /**
60  * A mixin for [[MessagePusher]] s, that enables the possibility to define two levels of
61  * [[MessageTriggers]].
62  *
63  * This is commonly used when it is needed to define some base extensible triggers, that should be
64  * easily extended letting the user defining the extras. When the triggers are effectively used,
65  * extras will be privileged, as the extra should always overwrite the base.
66  */
67 trait Composable[A] extends MessagePusher[A] {
68 
69   /**
70    * The extra [[MessageTriggers]] of the pusher.
71    * @return
72    *   The extra [[MessageTriggers]].
73    */
74   def extra: MessageTriggers[A] = PartialFunction.empty
75 
76   /**
77    * The base [[MessageTriggers]] of the pusher.
78    * @return
79    */
80   def base: MessageTriggers[A]
81 
82   /**
83    * All the triggers with the right order.
84    * @return
85    *   the chosen answer.
86    */
87   override def triggers: MessageTriggers[A] = Seq(extra, base).reduce(_ orElse _)
88 }
