16 /**
17  * The trait makes possible to add into the [[BehaviorBasedModel]] the <b>Eatable</b> Behavior.
18  */
19 trait EatableExt
20   extends BehaviorBasedModel
21   with CMessagesExt
22   with CReactionsExt
23   with StateUtilsExt {
24 
25   /**
26    * An <b>ItemBehavior</b> associated to an <b>BehaviorBasedItem</b> that can be eaten. After an
27    * item is eaten, it should be removed from the player bag (or from the current room, if it was
28    * there).
29    */
30   abstract class Eatable extends ItemBehavior {
31 
32     /**
33      * A Reaction that should remove the Item from the player bag (or from the location).
34      * @return
35      *   A Reaction that should remove the Item from the player bag (or from the location).
36      */
37     def eat: Reaction
38   }
39 
40   /**
41    * Standard implementation of the Eatable behavior.
42    *
43    * This is a behavior associated to an Item that can be eaten, if present in the bag or in the
44    * room.
45    * @param onEatExtra
46    *   Reaction to be executed when the item has been successfully eaten, after the standard
47    *   <b>Reaction</b>. It can be omitted.
48    */
49   case class SimpleEatable(onEatExtra: Reaction = Reaction.empty)(implicit val subject: I)
50     extends Eatable {
51 
52     override def triggers: ItemTriggers = {
53       case (Eat, None, state) if state.isInScope(subject) => eat
54     }
55 
56     override def eat: Reaction =
57       for {
58         _ <- CReactions.modifyLocationItems(_ - subject.ref)
59         _ <- CReactions.modifyBag(_ - subject.ref)
60         _ <- CReactions.addMessage(CMessages.Eaten(subject))
61         s <- onEatExtra
62       } yield s
63   }
64 
65   /**
66    * Companion object for [[Eatable]].
67    */
68   object Eatable {
69 
70     /**
71      * A function that builds an Eatable behavior given a subject Item.
72      * @param onEatExtra
73      *   Reaction to be executed when the item has been successfully eaten, after the standard
74      *   Reaction. It can be omitted.
75      * @return
76      *   A function that builds an Eatable behavior given a subject Item.
77      */
78     def builder(onEatExtra: Reaction = Reaction.empty): I => Eatable = SimpleEatable(onEatExtra)(_)
79   }
80 }
