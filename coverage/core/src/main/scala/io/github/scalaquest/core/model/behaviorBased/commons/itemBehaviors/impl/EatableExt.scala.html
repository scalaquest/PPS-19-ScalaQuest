9 /**
10  * The trait makes possible to mix into [[BehaviorBasedModel]] the Eatable behavior.
11  */
12 trait EatableExt
13   extends BehaviorBasedModel
14   with StateUtilsExt
15   with CommonMessagesExt
16   with CommonReactionsExt {
17 
18   /**
19    * A [[ItemBehavior]] associated to an [[Item]] that can be eaten. After an item is eaten, it
20    * should be removed from the player bag (or from the current room, if it was there).
21    */
22   abstract class Eatable extends ItemBehavior {
23     def eat: Reaction
24   }
25 
26   /**
27    * Standard implementation of the Eatable behavior.
28    *
29    * This is a behavior associated to an Item that can be eaten, if present in the bag or in the
30    * room.
31    * @param onEatExtra
32    *   Reaction to be executed when the item has been successfully eaten, after the standard
33    *   [[Reaction]]. It can be omitted.
34    */
35   case class SimpleEatable(onEatExtra: Option[Reaction] = None)(implicit subject: I)
36     extends Eatable {
37 
38     override def triggers: ItemTriggers = {
39       // "Eat the item"
40       case (Eat, item, None, state) if state.isInScope(item) => eat
41     }
42 
43     def eat: Reaction =
44       _.applyReactions(
45         Reactions.eat(subject),
46         onEatExtra.getOrElse(Reactions.empty)
47       )
48   }
49 
50   /**
51    * Companion object for [[Eatable]]. Shortcut for the standard implementation.
52    */
53   object Eatable {
54     def builder(onEatExtra: Option[Reaction] = None): I => Eatable = SimpleEatable(onEatExtra)(_)
55   }
