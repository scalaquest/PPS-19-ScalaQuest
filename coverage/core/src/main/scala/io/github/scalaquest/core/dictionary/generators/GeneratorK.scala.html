12 /**
13  * A `Generator` that operates on kinds. It is used to wrap functions from `F[A]` to `B`.
14  *
15  * This type class is useful to work with `A`s that live in a context such as `Option` or `List`.
16  *
17  * It is necessary that `B` is a monoid instance, because its `empty` and `combine` are used to fold
18  * on the `F`.
19  */

21   extends Generator[F[A], B] {
22 
23   override def generate(a: F[A]): B = {
