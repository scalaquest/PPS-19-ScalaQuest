13  * The trait makes possible to mix into the [[SimpleCommonBehaviors]] the standard implementation of

16 trait SimpleTakeableExt extends CommonBase {
17 
18   /**
19    * Standard implementation of the Takeable.
20    *
21    * The behavior of an Item that could be put into the Bag of the player from the current room.
22    * @param onTakeExtra
23    *   Reaction to be executed into the State when taken, after the standard Reaction. It can be
24    *   omitted.
25    */
26   case class SimpleTakeable(onTakeExtra: Option[Reaction] = None)(implicit
27     playerBagLens: Lens[S, Set[ItemRef]],
28     matchRoomsLens: Lens[S, Map[RoomRef, RM]],
29     roomLens: Lens[RM, Set[ItemRef]],
30     messageLens: Lens[S, Seq[Message]]
31   ) extends Takeable {
32 
33     override def triggers: ItemTriggers = {
34       // "Take the item"
35       case (Take, item, None, state) if state.isInCurrentRoom(item) => take(item)
36     }
37 
38     /**
39      * Returns a Reaction that removes the item from the current room, put it into the bag, executes
40      * the eventual extra reaction.
41      * @param item
42      *   The item to be taken.
43      * @return
44      *   A Reaction that removes the item from the current room, put it into the bag, executes the
45      *   eventual extra reaction.
46      */
47     def take(item: I): Reaction =
48       state => {
49         val updCurrRoom = roomLens.modify(_ - item.ref)(state.currentRoom)
50         state.applyReactions(
51           matchRoomsLens.modify(_ + (updCurrRoom.ref -> updCurrRoom)),
52           playerBagLens.modify(_ + item.ref),
53           messageLens.modify(_ :+ Taken(item)),
54           onTakeExtra.getOrElse(state => state)
55         )
56       }
57   }
58 }
