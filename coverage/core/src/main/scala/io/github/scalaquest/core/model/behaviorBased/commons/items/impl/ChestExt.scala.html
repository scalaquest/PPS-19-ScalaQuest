14 /**
15  * The trait makes possible to mix into a [[BehaviorBasedModel]] the <b>Chest BehaviorBasedItem</b>.
16  */
17 trait ChestExt extends BehaviorBasedModel with ContainerExt {
18 
19   /**
20    * A <b>BehaviorBasedItem</b> that can be opened, and reveals a set of items when opened.
21    */
22   trait Chest extends BehaviorBasedItem {
23 
24     /**
25      * The "openness" state of the item, as a boolean value. Initially closed.
26      * @return
27      *   True if item is already opened, False otherwise.
28      */
29     def isOpen: Boolean
30 
31     /**
32      * A <b>Chest</b> has always an associated <b>Container</b> behavior, that can be referenced
33      * from here.
34      * @return
35      *   The <b>Container</b> behavior of the item.
36      */
37     def container: Container
38   }
39 
40   /**
41    * A standard implementation for <b>Chest</b>.
42    * @param description
43    *   An [[ItemDescription]] for the item.
44    * @param ref
45    *   A unique reference to the item.
46    * @param containerBuilder
47    *   A builder for the <b>Container</b> behavior associated to the item.
48    * @param extraBehavBuilders
49    *   Additional behaviors associated to the item.
50    */
51   case class SimpleChest(
52     description: ItemDescription,
53     ref: ItemRef,
54     containerBuilder: I => Container,
55     extraBehavBuilders: Seq[I => ItemBehavior] = Seq.empty
56   ) extends Chest {
57     override val container: Container         = containerBuilder(this)
58     override val behaviors: Seq[ItemBehavior] = container +: extraBehavBuilders.map(_(this))
59     override def isOpen: Boolean              = container.isOpen
60   }
61 
62   /**
63    * Companion object for <b>Chest</b>.
64    */
65   object Chest {
66 
67     /**
68      * Creates a standard <b>Chest</b>.
69      * @param description
70      *   An [[ItemDescription]] for the item.
71      * @param containerBuilder
72      *   A builder for the <b>Container</b> behavior associated to the item.
73      * @param extraBehavBuilder
74      *   Additional behaviors associated to the item.
75      * @return
76      *   An instance of a standard <b>Chest</b>.
77      */
78     def apply(
79       description: ItemDescription,
80       containerBuilder: I => Container,
81       extraBehavBuilder: Seq[I => ItemBehavior] = Seq.empty
82     ): Chest = SimpleChest(description, ItemRef(description), containerBuilder, extraBehavBuilder)
83 
84     /**
85      * A chest that to be opened doesn't need any key.
86      * @param description
87      *   the [[ItemDescription]] of the chest.
88      * @param items
89      *   the items contained in a chest
90      * @param extraBehavBuilder
91      *   possible extra behavior when is open the container item.
92      * @return
93      *   an instance that implements chest trait.
94      */
95     def createUnlocked(
96       description: ItemDescription,
97       items: Set[I] = Set.empty,
98       extraBehavBuilder: Seq[I => ItemBehavior] = Seq.empty
99     ): Chest =
100       apply(
101         description,
102         Container.unlockedBuilder(items),
103         extraBehavBuilder
104       )
105 
106     /**
107      * A chest that to be opened require a key.
108      * @param description
109      *   the [[ItemDescription]] of the chest.
110      * @param items
111      *   the items contained in a chest.
112      * @param key
113      *   the key required to open the chest.
114      * @param extraBehavBuilder
115      *   possible extra behavior when is open the container item.
116      * @return
117      *   an instance that implements chest trait.
118      * @return
119      */
120     def createLocked(
121       description: ItemDescription,
122       items: Set[I] = Set.empty,
123       key: Key,
124       extraBehavBuilder: Seq[I => ItemBehavior] = Seq.empty
125     ): Chest =
126       apply(
127         description,
128         Container.lockedBuilder(items, key),
129         extraBehavBuilder
130       )
131   }
132 }
