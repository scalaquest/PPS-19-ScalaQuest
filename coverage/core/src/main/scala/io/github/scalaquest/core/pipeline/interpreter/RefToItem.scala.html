10 import io.github.scalaquest.core.model.{ItemRef, Model}

12 /**
13  * An utility to extract an [[Model.Item]], given its [[ItemRef]]. The [[RefToItem::unapply()]]
14  * functionality is exploited, as this utility is used in match cases.
15  * @tparam I
16  *   The concrete type of the [[Model.Item]] in use. It should be a subtype ofderived from
17  *   [[Model.Item]].
18  */
19 trait RefToItem[I] {
20   def unapply(ref: ItemRef): Option[I]
21 }
22 
23 /**
24  * Companion object for the [[RefToItem]] trait. It exposes an [[RefToItem::apply()]] to build the
25  * [[RefToItem]] instance from a dictionary of [[Model.Item]] s.
26  */
27 object RefToItem {
28 
29   /**
30    * Used to build the [[RefToItem]] instance from a dictionary of [[Model.Item]], making
31    * transparent the presence of it during its use.
32    * @param model
33    *   The concrete instance of the [[Model]] in use.
34    * @param itemsDict
35    *   A [[Map]] that indicates the [[ItemRef]] of each ot the [[Model.Item]] in use. The
36    *   [[Model.Item]] type must be derived from the `model` passed as parameter.
37    * @tparam M
38    *   The concrete type of the [[Model]] in use.
39    * @return
40    *   An instance of [[RefToItem]], using transparently the given item dictionary to match
41    *   [[ItemRef]] - [[Model.Item]] links.
42    */
43   def apply[M <: Model](model: M)(itemsDict: Map[ItemRef, model.I]): RefToItem[model.I] = {
44     case class SimpleRefToItem(itemsDict: Map[ItemRef, model.I]) extends RefToItem[model.I] {
45       override def unapply(ref: ItemRef): Option[model.I] = itemsDict get ref
46     }
47     SimpleRefToItem(itemsDict)
48   }
49 }
