10 import io.github.scalaquest.core.model.Model

12 /**
13  * A wrapper for the output of the [[Interpreter]] execution. It should contain a [[Model.Reaction]]
14  * instance.
15  * @tparam R
16  *   The [[Model.Reaction]] concrete type.
17  */
18 trait InterpreterResult[R] {
19   def reaction: R
20 }
21 
22 /**
23  * A companion object for the [[InterpreterResult]] trait. It exposes an
24  * [[InterpreterResult::apply()]] to instantiate the [[InterpreterResult]] with the right
25  * constraints between types.
26  */
27 object InterpreterResult {
28 
29   /**
30    * Builds an [[InterpreterResult]], with the right constraints between types.
31    * @param model
32    *   The concrete instance of the [[Model]] in use.
33    * @param reaction
34    *   The [[Model.Reaction]] instance to be wrapped. The [[Model.Reaction]] type must be derived
35    *   from the `model` passed as parameter.
36    * @tparam M
37    *   The concrete type of the [[Model]] in use.
38    * @return
39    *   An [[InterpreterResult]], with the right type constraints.
40    */
41   def apply[M <: Model](model: M)(reaction: model.Reaction): InterpreterResult[model.Reaction] = {
42     case class SimpleInterpreterResult(reaction: model.Reaction)
43       extends InterpreterResult[model.Reaction]
44     SimpleInterpreterResult(reaction)
45   }
46 }
