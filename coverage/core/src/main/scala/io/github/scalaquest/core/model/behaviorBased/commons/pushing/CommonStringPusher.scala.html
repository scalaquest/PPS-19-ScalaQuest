48   @tailrec
49   private def printItems(tail: List[CommonMessagesExt#I], acc: String = ""): String = {
50     tail match {
51       case ::(head, Nil)  => s"${acc}a ${head.toString}."
52       case ::(head, next) => printItems(next, s"${acc}a ${head.toString}, ")
53       case Nil            => "nothing."
54     }
55   }
56 
57   private def printNeighbors(neighbors: Map[Direction, CommonMessagesExt#RM]): String = {
58     neighbors match {
59       case ns if ns.isEmpty => "\nYou cannot go anywhere now."
60       case ns =>
61         ns.map(n => s"There is a ${n._2.toString} in direction ${n._1.toString}.")
62           .fold("")(_ + "\n" + _)
63     }
64   }
