59  */
60 abstract class PrologParser extends Parser with Helpers {
61 
62   /** The [[io.github.scalaquest.core.parsing.engine.Engine]] used. */
63   protected def engine: Engine
64 
65   override def parse(lexerResult: LexerResult): Option[ParserResult] = {
66     import dsl._
67     import io.github.scalaquest.core.parsing.scalog.dsl.seqToListP
68     val tokens = lexerResult.tokens.map(Atom)
69     val query  = phrase(imp(X), tokens)
70 
71     for {
72       r <- engine.solve(query).headOption
73       x <- r.getVariable(X)
74       ast <- x match {
75         case sentence(`/`(verb, p(prep)), Atom(subject)) =>
76           Some(AbstractSyntaxTree.Intransitive(verb, prep, subject))
77         case sentence(`/`(verb, p(prep)), Atom(subject), i(obj)) =>
78           Some(AbstractSyntaxTree.Transitive(verb, prep, subject, obj))
79         case sentence(
80               `/`(verb, p(prep)),
81               Atom(subject),
82               i(directObj),
83               i(indirectObj)
84             ) =>
85           Some(AbstractSyntaxTree.Ditransitive(verb, prep, subject, directObj, indirectObj))
86         case _ => None
87       }
88     } yield SimpleParserResult(ast)
89   }
