63 abstract class PrologParser extends Parser with Helpers {
64 
65   /** The [[io.github.scalaquest.core.parsing.engine.Engine]] used. */
66   protected def engine: Engine

68   override def parse(lexerResult: LexerResult): Option[ParserResult] = {
69     import dsl._
70     import io.github.scalaquest.core.parsing.scalog.dsl.seqToListP
71     val tokens = lexerResult.tokens.map(Atom)
72     val query  = phrase(imp(X), tokens)
73 
74     for {
75       r <- engine.solve(query).headOption
76       x <- r.getVariable(X)
77       ast <- x match {
78         case sentence(`/`(verb, p(prep)), Atom(subject)) =>
79           Some(AbstractSyntaxTree.Intransitive(verb, prep, subject))
80         case sentence(`/`(verb, p(prep)), Atom(subject), i(obj)) =>
81           Some(AbstractSyntaxTree.Transitive(verb, prep, subject, obj))
82         case sentence(
83               `/`(verb, p(prep)),
84               Atom(subject),
85               i(directObj),
86               i(indirectObj)
87             ) =>
88           Some(AbstractSyntaxTree.Ditransitive(verb, prep, subject, directObj, indirectObj))
89         case _ => None
90       }
91     } yield SimpleParserResult(ast)
92   }
