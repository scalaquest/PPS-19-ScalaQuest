19 trait Helpers {
20 
21   object dsl {
22     import io.github.scalaquest.core.parsing.scalog.dsl._
23 
24     val X         = Variable("X")
25     val imp       = CompoundBuilder("imp").constructor
26     val phrase    = CompoundBuilder("phrase").constructor
27     val sentence  = CompoundBuilder("sentence").extractor.toTerms
28     val `/`       = CompoundBuilder("/").extractor.toStrings
29     val decorated = CompoundBuilder("decorated").extractor.toTerms
30   }
31 
32   object ItemDescription {
33     import dsl.decorated
34 
35     @nowarn("msg=not all missing cases are reported")
36     @nowarn("msg=match.*exhaustive")
37     private def toItemDescription(term: Term): ItemDescription =
38       term match {
39         case Atom(name) => BaseItem(name)
40         case decorated(Atom(decoration), t) =>
41           DecoratedItem(decoration, toItemDescription(t))
42       }
43 
44     def unapply(t: Term): Option[ItemDescription] =
45       t match {
46         case obj @ (_: Compound | _: Term) => Some(toItemDescription(obj))
47         case _                             => None
48       }
49 
50   }
51 
52   object Preposition {
53     private final val empty = "{}"
54 
55     def unapply(prep: String): Option[Option[String]] =
56       prep match {
57         case `empty` => Some(None)
58         case x       => Some(Some(x))
59       }
60   }
61 
62   val i = ItemDescription
63   val p = Preposition
64 }

66 /**
67  * A [[Parser]] implementation that takes advantage of an
68  * [[io.github.scalaquest.core.parsing.engine.Engine]] in order to perform the syntactical analysis.
69  */
70 abstract class PrologParser extends Parser with Helpers {
71 
72   /** The [[io.github.scalaquest.core.parsing.engine.Engine]] used. */
73   protected def engine: Engine
74 
75   override def parse(lexerResult: LexerResult): Option[ParserResult] = {
76     import dsl._
77     import io.github.scalaquest.core.parsing.scalog.dsl.seqToListP
78     val tokens = lexerResult.tokens.map(Atom)
79     val query  = phrase(imp(X), tokens)
80 
81     for {
82       r <- engine.solve(query).headOption
83       x <- r.getVariable(X)
84       ast <- x match {
85         case sentence(`/`(verb, p(prep)), Atom(subject)) =>
86           Some(AbstractSyntaxTree.Intransitive(verb, prep, subject))
87         case sentence(`/`(verb, p(prep)), Atom(subject), i(obj)) =>
88           Some(AbstractSyntaxTree.Transitive(verb, prep, subject, obj))
89         case sentence(
90               `/`(verb, p(prep)),
91               Atom(subject),
92               i(directObj),
93               i(indirectObj)
94             ) =>
95           Some(AbstractSyntaxTree.Ditransitive(verb, prep, subject, directObj, indirectObj))
96         case _ => None
97       }
98     } yield SimpleParserResult(ast)
99   }
