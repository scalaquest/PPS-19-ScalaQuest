27   /**
28    * A standard implementation for <b>InspectableLocation</b>.
29    * @param onInspectExtra
30    *   <b>Reaction</b> to be executed when the player successfully inspect the location. It can be
31    *   omitted.
32    */
33   case class SimpleInspectableLocation(onInspectExtra: Reaction = Reaction.empty)
34     extends InspectableLocation {
35 
36     override def triggers: GroundTriggers = { case (Inspect, _) => inspectLocation }
37 
38     override def inspectLocation: Reaction =
39       for {
40         s1 <- Reaction.empty
41         _ <- CReactions.addMessage(
42           CMessages.Inspected(s1.location, s1.location.items(s1), s1.location.neighbors(s1))
43         )
44         s2 <- onInspectExtra
45       } yield s2
46   }
47 
48   /**
49    * Companion object for <b>InspectableLocation</b>.
50    */
51   object InspectableLocation {
52 
53     /**
54      * Shortcut for the standard implementation.
55      * @param onInspectExtra

57      *   omitted.
58      * @return
59      *   the instance of <b>InspectableLocation</b>.
60      */
61     def apply(onInspectExtra: Reaction = Reaction.empty): InspectableLocation =
62       SimpleInspectableLocation(onInspectExtra)
63   }
