18 
19   /**
20    * A <b>GroundBehavior</b> that enables the possibility to describe the location.
21    */
22   abstract class InspectableLocation extends GroundBehavior {
23 
24     /**
25      * A <b>Reaction</b> that communicates to the user a description of the current location.
26      * @return
27      *   A <b>Reaction</b> that communicates to the user a description of the current location.
28      */
29     def inspectLocation: Reaction
30   }
31 
32   /**
33    * A standard implementation for <b>InspectableLocation</b>.
34    * @param onInspectExtra
35    *   <b>Reaction</b> to be executed when the player successfully inspect the location. It can be
36    *   omitted.
37    */
38   case class SimpleInspectableLocation(onInspectExtra: Reaction = Reaction.empty)
39     extends InspectableLocation {
40 
41     override def triggers: GroundTriggers = { case (Inspect, _) => inspectLocation }
42 
43     override def inspectLocation: Reaction =
44       for {
45         s1 <- Reaction.empty
46         _ <- CReactions.addMessage(
47           CMessages.Inspected(s1.location, s1.locationItems, s1.location.neighbors(s1))
48         )
49         s2 <- onInspectExtra
50       } yield s2
51   }
52 
53   /**
54    * Companion object for <b>InspectableLocation</b>.
55    */
56   object InspectableLocation {
57 
58     /**
59      * Shortcut for the standard implementation.
60      * @param onInspectExtra
61      *   <b>Reaction</b> to be executed when the player successfully inspect the location. It can be
62      *   omitted.
63      * @return
64      *   the instance of <b>InspectableLocation</b>.
65      */
66     def apply(onInspectExtra: Reaction = Reaction.empty): InspectableLocation =
67       SimpleInspectableLocation(onInspectExtra)
68   }
69 }
