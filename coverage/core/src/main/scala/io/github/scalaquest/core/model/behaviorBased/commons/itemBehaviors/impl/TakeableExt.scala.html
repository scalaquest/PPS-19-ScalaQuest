16 /**
17  * The trait makes possible to add into the [[BehaviorBasedModel]] the <b>Takeable</b> behavior.
18  */
19 trait TakeableExt
20   extends BehaviorBasedModel
21   with CMessagesExt
22   with CReactionsExt
23   with StateUtilsExt {
24 
25   /**
26    * An <b>ItemBehavior</b> aassociated to a <b>BehaviorBasedItem</b> that can be taken and put away
27    * into the bag of the player.
28    */
29   abstract class Takeable extends ItemBehavior {
30 
31     /**
32      * A <b>Reaction</b> that moves the subject from the location to the bag.
33      * @return
34      *   A <b>Reaction</b> that moves the subject from the location to the bag.
35      */
36     def take: Reaction
37   }
38 
39   /**
40    * Standard implementation of <b>Takeable</b>.
41    *
42    * @param onTakeExtra
43    *   <b>Reaction</b> to be executed after after the standard take <b>Reaction</b>. It can be
44    *   omitted.
45    */
46   case class SimpleTakeable(onTakeExtra: Reaction = Reaction.empty)(implicit val subject: I)
47     extends Takeable {
48 
49     override def triggers: ItemTriggers = {
50       case (Take, None, state) if state.isInLocation(subject) => take
51     }
52 
53     override def take: Reaction =
54       for {
55         _ <- CReactions.modifyLocationItems(_ - subject.ref)
56         _ <- CReactions.modifyBag(_ + subject.ref)
57         _ <- CReactions.addMessage(CMessages.Taken(subject))
58         s <- onTakeExtra
59       } yield s
60   }
61 
62   /**
63    * Companion object for <b>Takeable</b>.
64    */
65   object Takeable {
66 
67     /**
68      * Builder that returns a standard <b>Takeable</b> behavior given a subject.
69      *
70      * @param onTakeExtra
71      *   <b>Reaction</b> to be executed after after the standard take <b>Reaction</b>. It can be
72      *   omitted.
73      * @return
74      *   Builder that returns a standard <b>Takeable</b> behavior given a subject.
75      */
76     def builder(onTakeExtra: Reaction = Reaction.empty): I => Takeable =
77       SimpleTakeable(onTakeExtra)(_)
78   }
79 }
