9 /**
10  * The trait makes possible to add into the [[BehaviorBasedModel]] the <b>Takeable</b> behavior.
11  */
12 trait TakeableExt
13   extends BehaviorBasedModel
14   with CMessagesExt
15   with CReactionsExt
16   with StateUtilsExt {
17 
18   /**
19    * An <b>ItemBehavior</b> aassociated to a <b>BehaviorBasedItem</b> that can be taken and put away
20    * into the bag of the player.
21    */
22   abstract class Takeable extends ItemBehavior {
23 
24     /**
25      * A <b>Reaction</b> that moves the subject from the location to the bag.
26      * @return
27      *   A <b>Reaction</b> that moves the subject from the location to the bag.
28      */
29     def take: Reaction
30   }
31 
32   /**
33    * Standard implementation of <b>Takeable</b>.
34    *
35    * @param onTakeExtra
36    *   <b>Reaction</b> to be executed after after the standard take <b>Reaction</b>. It can be
37    *   omitted.
38    */
39   case class SimpleTakeable(onTakeExtra: Reaction = Reaction.empty)(implicit val subject: I)
40     extends Takeable {
41 
42     override def triggers: ItemTriggers = {
43       case (Take, None, state) if state.isInLocation(subject) => take
44     }
45 
46     override def take: Reaction =
47       for {
48         _ <- CReactions.modifyLocationItems(_ - subject.ref)
49         _ <- CReactions.modifyBag(_ + subject.ref)
50         _ <- CReactions.addMessage(CMessages.Taken(subject))
51         s <- onTakeExtra
52       } yield s
53   }
54 
55   /**
56    * Companion object for <b>Takeable</b>.
57    */
58   object Takeable {
59 
60     /**
61      * Builder that returns a standard <b>Takeable</b> behavior given a subject.
62      *
63      * @param onTakeExtra
64      *   <b>Reaction</b> to be executed after after the standard take <b>Reaction</b>. It can be
65      *   omitted.
66      * @return
67      *   Builder that returns a standard <b>Takeable</b> behavior given a subject.
68      */
69     def builder(onTakeExtra: Reaction = Reaction.empty): I => Takeable =
70       SimpleTakeable(onTakeExtra)(_)
71   }
