26   /**
27    * Standard implementation of the Takeable.
28    *
29    * The behavior of an Item that could be put into the Bag of the player from the current room.
30    * @param onTakeExtra
31    *   Reaction to be executed into the State when taken, after the standard Reaction. It can be
32    *   omitted.
33    */
34   case class SimpleTakeable(onTakeExtra: Option[Reaction] = None)(implicit subject: I)
35     extends Takeable {
36 
37     override def triggers: ItemTriggers = {
38       // "Take the item"
39       case (Take, item, None, state) if state.isInLocation(item) => take
40     }
41 
42     /**
43      * Returns a Reaction that removes the item from the current room, put it into the bag, executes
44      * the eventual extra reaction.
45      * @return
46      *   A Reaction that removes the item from the current room, put it into the bag, executes the
47      *   eventual extra reaction.
48      */
49     def take: Reaction =
50       _.applyReactions(
51         Reactions.take(subject),
52         onTakeExtra.getOrElse(Reactions.empty)
53       )
54 
55   }
56 
57   /**
58    * Companion object for [[Takeable]]. Shortcut for the standard implementation.
59    */
60   object Takeable {
61 
62     def builder(onTakeExtra: Option[Reaction] = None): I => Takeable =
63       SimpleTakeable(onTakeExtra)(_)
64   }
