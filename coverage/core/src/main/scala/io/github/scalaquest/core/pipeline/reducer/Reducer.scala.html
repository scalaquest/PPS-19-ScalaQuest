8 package io.github.scalaquest.core.pipeline.reducer

10 import io.github.scalaquest.core.model.Model
11 import io.github.scalaquest.core.pipeline.interpreter.InterpreterResult
12 
13 /**
14  * A pipeline component that takes a [[Model.Reaction]] (wrapped into an [[InterpreterResult]] ) and
15  * returns a [[Model.State]] wrapped into an [[ReducerResult]]. The execution cannot fail.
16  * @tparam M
17  *   The concrete type of the [[Model]] in use.
18  * @tparam S
19  *   The concrete type of the [[Model.State]] in use. It should be derived from [[M]].
20  * @tparam R
21  *   The concrete type of the [[Model.Reaction]] in use. It should be derived from [[M]].
22  */
23 trait Reducer[M <: Model, S, R] {
24 
25   /**
26    * Triggers the [[Reducer]] execution.
27    * @param interpreterResult
28    *   A wrapper for the input [[Model.Reaction]].
29    * @return
30    *   A [[ReducerResult]] (wrapper for the updated [[Model.State]] ).
31    */
32   def reduce(interpreterResult: InterpreterResult[R]): ReducerResult[S]
33 }
34 
35 /**
36  * Companion object for the [[Reducer]] trait. It exposes some utilities to instantiate the
37  * [[Reducer]] with the right constraints between types.
38  *
39  * It exposes an [[Reducer::apply()]] to build an [[Reducer]] <b>with the right types</b>, as some
40  * constraints between the used types are defined. In addition, it provides a [[Reducer::builder()]]
41  * utility used into the pipeline to inject the [[Model.State]] in a separate phase, compared to
42  * passing the state directly to [[Reducer::reduce()]].
43  */
44 object Reducer {
45 
46   /**
47    * A generator of [[Reducer]] instances, giving the possibility to inject the [[Model.State]]
48    * separately from the [[Reducer::reduce()]] call.
49    * @tparam M
50    *   The concrete type of the [[Model]] in use.
51    * @tparam S
52    *   The concrete type of the [[Model.State]] in use. It should be derived from [[M]].
53    * @tparam R
54    *   The concrete type of the [[Model.Reaction]] in use. It should be derived from [[M]].
55    */
56   type Builder[M <: Model, S, R] = S => Reducer[M, S, R]
57 
58   /**
59    * It generates a [[Builder]] with the right type constraints.
60    * @param model
61    *   The concrete instance of the [[Model]] in use.
62    * @tparam M
63    *   The concrete type of the [[Model]] in use.
64    * @return
65    *   A [[Builder]] of [[Reducer]] instances, with the right type constraints.
66    */
67   def builder[M <: Model](implicit model: M): Builder[model.type, model.S, model.Reaction] =
68     // shortcut for implementing the Reducer, as it is a single method trait
69     state =>
70       (interpreterResult: InterpreterResult[model.Reaction]) => {
71         val (updState, messages) = interpreterResult.reaction(state)
72         ReducerResult(model)(updState, messages)
73       }
74 }
