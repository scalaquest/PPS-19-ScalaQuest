6 /**
7  * A pipeline component that takes a [[Model.Reaction]] (wrapped into an [[InterpreterResult]] ) and
8  * returns a [[Model.State]] wrapped into an [[ReducerResult]]. The execution cannot fail.
9  * @tparam M
10  *   The concrete type of the [[Model]] in use.
11  * @tparam S
12  *   The concrete type of the [[Model.State]] in use. It should be derived from [[M]].
13  * @tparam R
14  *   The concrete type of the [[Model.Reaction]] in use. It should be derived from [[M]].
15  */
16 trait Reducer[M <: Model, S, R] {
17 
18   /**
19    * Triggers the [[Reducer]] execution.
20    * @param interpreterResult
21    *   A wrapper for the input [[Model.Reaction]].
22    * @return
23    *   A [[ReducerResult]] (wrapper for the updated [[Model.State]] ).
24    */
25   def reduce(interpreterResult: InterpreterResult[R]): ReducerResult[S]
26 }
27 
28 /**
29  * Companion object for the [[Reducer]] trait. It exposes some utilities to instantiate the
30  * [[Reducer]] with the right constraints between types.
31  *
32  * It exposes an [[Reducer::apply()]] to build an [[Reducer]] <b>with the right types</b>, as some
33  * constraints between the used types are defined. In addition, it provides a [[Reducer::builder()]]
34  * utility used into the pipeline to inject the [[Model.State]] in a separate phase, compared to
35  * passing the state directly to [[Reducer::reduce()]].
36  */
37 object Reducer {
38 
39   /**
40    * A generator of [[Reducer]] instances, giving the possibility to inject the [[Model.State]]
41    * separately from the [[Reducer::reduce()]] call.
42    * @tparam M
43    *   The concrete type of the [[Model]] in use.
44    * @tparam S
45    *   The concrete type of the [[Model.State]] in use. It should be derived from [[M]].
46    * @tparam R
47    *   The concrete type of the [[Model.Reaction]] in use. It should be derived from [[M]].
48    */
49   type Builder[M <: Model, S, R] = S => Reducer[M, S, R]
50 
51   /**
52    * It generates a [[Builder]] with the right type constraints.
53    * @param model
54    *   The concrete instance of the [[Model]] in use.
55    * @tparam M
56    *   The concrete type of the [[Model]] in use.
57    * @return
58    *   A [[Builder]] of [[Reducer]] instances, with the right type constraints.
59    */
60   def builder[M <: Model](implicit model: M): Builder[model.type, model.S, model.Reaction] =
61     // shortcut for implementing the Reducer, as it is a single method trait
62     state =>
63       (interpreterResult: InterpreterResult[model.Reaction]) => {
64         val (updState, messages) = interpreterResult.reaction(state)
65         ReducerResult(model)(updState, messages)
66       }
67 }
