10 /** A Prolog data object. */
11 sealed trait Term extends CodeGen {
12 
13   /**
14    * Template method for all infix operators.
15    * @note
16    *   All operators that use this method should be right associative.
17    */
18   protected def infixOp(op: String)(left: Term): Compound = Compound(Atom(op), left, List(this))
19 
20   /** Implementation of the infix `^` Prolog operator. */
21   def ^:(left: Term): Compound = infixOp("^")(left)
22 
23   /** Implementation of the infix `/` Prolog operator. */
24   def /:(left: Term): Compound = infixOp("/")(left)
25 }
26 
27 /** A Prolog atom. */
28 case class Atom(name: String) extends Term {
29   override def generate: String = name
30 }
31 
32 /** A Prolog number */
33 case class Number(n: Int) extends Term {

35 }
36 
37 /** A Prolog variable. */
38 case class Variable(name: String) extends Term {
39   override def generate: String = name
40 }
41 
42 /**
43  * A Prolog compound term.
44  * @param functor
45  *   the atom used as a functor
46  * @param arg1
47  *   the first mandatory term
48  * @param args
49  *   the other optional terms
50  */
51 case class Compound(functor: Atom, arg1: Term, args: List[Term] = List()) extends Term {
52 
53   override def generate: String =
54     s"${functor.generate}(${(arg1 +: args).map(_.generate).mkString(",")})"
55 }
56 
57 /** A Prolog list. */
58 case class ListP(terms: Term*) extends Term {
59   override def generate: String = terms.map(_.generate).mkString("[", ",", "]")
