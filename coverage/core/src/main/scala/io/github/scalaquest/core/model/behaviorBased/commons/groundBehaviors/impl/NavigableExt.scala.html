22   /**
23    * Standard implementation of [[Navigable]].
24    *
25    * @param onNavigateExtra
26    *   [[Reaction]] to be executed when the player successfully navigate in a new Room, using
27    *   navigation Actions after the standard [[Reaction]]. It can be omitted.
28    */
29   case class SimpleNavigable(onNavigateExtra: Reaction = Reaction.empty) extends Navigable {
30 
31     override def triggers: GroundTriggers = { case (Go(d), s) =>
32       s.locationNeighbor(d) map movePlayer getOrElse failedToNavigate(d)
33     }
34 
35     override def movePlayer(targetRoom: RM): Reaction =
36       Reaction.combine(
37         Reactions.navigate(targetRoom),
38         onNavigateExtra
39       )
40 
41     def failedToNavigate(direction: Direction): Reaction =
42       Reaction.appendMessage(Messages.FailedToNavigate(direction))(Reaction.empty)
43   }
44 
45   /**
46    * Companion object for [[Navigable]]. Shortcut for the standard implementation.
47    */
48   object Navigable {
49 
50     def apply(onNavigateExtra: Reaction = Reaction.empty): Navigable =
51       SimpleNavigable(onNavigateExtra)
52   }
