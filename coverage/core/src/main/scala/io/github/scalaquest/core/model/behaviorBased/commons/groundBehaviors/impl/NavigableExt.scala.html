24   /**
25    * Standard implementation of [[Navigable]].
26    *
27    * @param onNavigateExtra
28    *   [[Reaction]] to be executed when the player successfully navigate in a new Room, using
29    *   navigation Actions after the standard [[Reaction]]. It can be omitted.
30    */
31   case class SimpleNavigable(onNavigateExtra: Option[Reaction] = None) extends Navigable {
32 
33     override def triggers: GroundTriggers = {
34       // "go <direction>"
35       case (Go(direction), state) if state.locationNeighbor(direction).isDefined =>
36         movePlayer(state.locationNeighbor(direction).get)
37     }
38 
39     def movePlayer(targetRoom: RM): Reaction =
40       _.applyReactions(
41         Reactions.navigate(targetRoom),
42         onNavigateExtra.getOrElse(Reactions.empty)
43       )
44   }

48    */
