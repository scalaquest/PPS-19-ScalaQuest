17 /**
18  * The trait makes possible to mix into a [[BehaviorBasedModel]] the <b>Navigable
19  * GroundBehavior</b>.
20  */
21 trait NavigableExt
22   extends BehaviorBasedModel
23   with CMessagesExt
24   with CReactionsExt
25   with StateUtilsExt {
26 
27   /**
28    * A <b>GroundBehavior</b> that enables the possibility to change the location using
29    * <b>Directions<b>.
30    */
31   abstract class Navigable extends GroundBehavior {
32 
33     /**
34      * Sets the location of the player to the given <b>Room</b>.
35      * @param targetRoom
36      *   The <b>Room</b> into which move the player.
37      * @return
38      *   A <b>Reaction</b> that sets the location of the player to the given <b>Room</b>.
39      */
40     def movePlayer(targetRoom: RM): Reaction
41   }
42 
43   /**
44    * Standard implementation of <b>Navigable</b>.
45    *
46    * @param onNavigateExtra
47    *   <b>Reaction</b> to be executed when the player successfully navigate in a new <b>Room</b>. It
48    *   can be omitted.
49    */
50   case class SimpleNavigable(onNavigateExtra: Reaction = Reaction.empty) extends Navigable {
51 
52     override def triggers: GroundTriggers = { case (Go(d), s) =>
53       s.locationNeighbor(d) map movePlayer getOrElse failedToNavigate(d)
54     }
55 
56     override def movePlayer(targetRoom: RM): Reaction =
57       for {
58         _ <- CReactions.switchLocation(targetRoom)
59         _ <- CReactions.addMessage(CMessages.Navigated(targetRoom))
60         s <- onNavigateExtra
61       } yield s
62 
63     def failedToNavigate(direction: Direction): Reaction =
64       Reaction.appendMessage(CMessages.FailedToNavigate(direction))(Reaction.empty)
65   }
66 
67   /**
68    * Companion object for <b>Navigable</b>.
69    */
70   object Navigable {
71 
72     /**
73      * Shortcut for the standard implementation.
74      * @param onNavigateExtra
75      *   <b>Reaction<b> to be executed when the player successfully navigate in a new <b>Room</b>.
76      *   It can be omitted.
77      * @return
78      *   The instance of <b>Navigable</b>.
79      */
80     def apply(onNavigateExtra: Reaction = Reaction.empty): Navigable =
81       SimpleNavigable(onNavigateExtra)
82   }
83 }
