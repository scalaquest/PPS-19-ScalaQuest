10 /**
11  * The trait makes possible to mix into a [[BehaviorBasedModel]] the <b>Navigable

13  */
14 trait NavigableExt
15   extends BehaviorBasedModel
16   with CMessagesExt
17   with CReactionsExt
18   with StateUtilsExt {
19 
20   /**
21    * A <b>GroundBehavior</b> that enables the possibility to change the location using
22    * <b>Directions<b>.
23    */
24   abstract class Navigable extends GroundBehavior {
25 
26     /**
27      * Sets the location of the player to the given <b>Room</b>.
28      * @param targetRoom
29      *   The <b>Room</b> into which move the player.
30      * @return
31      *   A <b>Reaction</b> that sets the location of the player to the given <b>Room</b>.
32      */
33     def movePlayer(targetRoom: RM): Reaction
34   }
35 
36   /**
37    * Standard implementation of <b>Navigable</b>.
38    *
39    * @param onNavigateExtra
40    *   <b>Reaction</b> to be executed when the player successfully navigate in a new <b>Room</b>. It
41    *   can be omitted.
42    */
43   case class SimpleNavigable(onNavigateExtra: Reaction = Reaction.empty) extends Navigable {
44 
45     override def triggers: GroundTriggers = { case (Go(d), s) =>
46       s.locationNeighbor(d) map movePlayer getOrElse failedToNavigate(d)
47     }
48 
49     override def movePlayer(targetRoom: RM): Reaction =
50       for {
51         _ <- CReactions.switchLocation(targetRoom)
52         _ <- CReactions.addMessage(CMessages.Navigated(targetRoom))
53         s <- onNavigateExtra
54       } yield s
55 
56     def failedToNavigate(direction: Direction): Reaction =
57       Reaction.appendMessage(CMessages.FailedToNavigate(direction))(Reaction.empty)
58   }
59 
60   /**
61    * Companion object for <b>Navigable</b>.
62    */
63   object Navigable {
64 
65     /**
66      * Shortcut for the standard implementation.
67      * @param onNavigateExtra
68      *   <b>Reaction<b> to be executed when the player successfully navigate in a new <b>Room</b>.
69      *   It can be omitted.
70      * @return
71      *   The instance of <b>Navigable</b>.
72      */
73     def apply(onNavigateExtra: Reaction = Reaction.empty): Navigable =
74       SimpleNavigable(onNavigateExtra)
75   }
76 }
