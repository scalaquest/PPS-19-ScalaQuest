30   /**
31    * Standard implementation of the RoomLink.
32    *
33    * This is a behavior associated to an item that is a link between two rooms (a door, for
34    * instance), and that could be opened to pass into it. The user can "Enter" into it, resulting
35    * into move it into the connected Room. another room.
36    * @param endRoom
37    *   The room into which the user is projected after entering the object.
38    * @param openable
39    *   The Openable behavior associated to the roomLink. If not passed, the item can be entered
40    *   without opening.
41    * @param onEnterExtra
42    *   Reaction to be executed into the State when entered, after the standard Reaction. It can be
43    *   omitted.
44    */
45   case class SimpleRoomLink(
46     endRoom: RM,
47     endRoomDirection: Direction,
48     openable: Option[Openable] = None,
49     onEnterExtra: Option[Reaction] = None
50   )(implicit subject: I)
51     extends RoomLink
52     with Delegate {
53 
54     /**
55      * If an openable is passed, it is passed as father behavior.
56      * @return
57      *   The father behavior triggers.
58      */
59     override def delegateTriggers: ItemTriggers =
60       openable.fold(PartialFunction.empty: ItemTriggers)(_.triggers)
61 
62     override def receiverTriggers: ItemTriggers = {
63       // "Open the item (with something)"
64       case (Open, i, maybeKey, s)
65           if s.isInLocation(i) && openable.fold(true)(_.canBeOpened(maybeKey)(s))
66             && !isOpen && openable.isDefined =>
67         open
68 
69       // "Enter the item"
70       case (Enter, i, None, s) if s.isInLocation(i) && openable.fold(true)(_.isOpen) => enter
71     }
72 
73     override def enter: Reaction =
74       _.applyReactions(
75         Reactions.enter(endRoom),
76         onEnterExtra.getOrElse(Reactions.empty)
77       )
78 
79     def open: Reaction =
80       state => {
81         val addDirection = roomsLens.modify(
82           _.updatedWith(state.location.ref) {
83             case Some(room) =>
84               Some(roomDirectionsLens.modify(_ + (endRoomDirection -> endRoom.ref))(room))
85             case _ => None
86           }
87         )
88 
89         state.applyReactions(
90           openable.fold(Reactions.empty)(o => o.open),
91           addDirection
92         )
93       }
94 
95     override def isOpen: Boolean = openable.fold(true)(_.isOpen)
96   }
97 
98   /**
99    * Companion object for [[RoomLink]]. Shortcut for the standard implementation.
100    */
101   object RoomLink {
102

107       onEnterExtra: Option[Reaction] = None
108     ): I => RoomLink =
109       item =>
110         SimpleRoomLink(endRoom, endRoomDirection, openableBuilder.map(_(item)), onEnterExtra)(item)
111   }
