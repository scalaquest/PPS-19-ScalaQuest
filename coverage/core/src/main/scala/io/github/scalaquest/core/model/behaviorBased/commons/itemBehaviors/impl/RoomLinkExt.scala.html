15 /**
16  * The trait makes possible to add into the [[BehaviorBasedModel]] the <b>RoomLink</b> behavior.
17  */
18 trait RoomLinkExt extends BehaviorBasedModel with OpenableExt with CReactionsExt {
19 
20   /**
21    * An <b>ItemBehavior</b> associated to a <b>BehaviorBasedItem</b> that enables the possibility to
22    * change the player location when entered, like a <b>Door</b>. Conceptually, an subject that
23    * exposes a <b>RoomLink</b> behavior could also be opened before permitting to the player to
24    * enter.
25    */
26   abstract class RoomLink extends ItemBehavior {
27 
28     /**
29      * The "openness" state of the subject, as a boolean value. An open subject is conceptually
30      * "ready-to-use", as it can be entered. Always open, if the subject is not openable.
31      * @return
32      *   True if RoomLink is open, false otherwise.
33      */
34     def isOpen: Boolean
35 
36     /**
37      * If the subject have to be opened before entering, the <b>RoomLink</b> should include [[Some]]
38      * <b>Openable</b> behavior here. If the subject is set as always open, it can be set to
39      * [[None]].
40      * @return
41      *   [[Some]] with an <b>Openable</b> behavior if the subject should be opened before entering,
42      *   [[None]] otherwise.
43      */
44     def openable: Option[Openable]
45 
46     /**
47      * The <b>Room</b> that will be set as the new location of the player, after entering the
48      * subject.
49      * @param s
50      *   The current <b>State</b>.
51      * @return
52      *   The <b>Room</b> that will be set as the new location of the player, after entering the
53      *   subject.
54      */
55     def endRoom(implicit s: S): RM
56 
57     /**
58      * A <b>Reaction</b> that sets the subject in an open state. Internally, it should exploit the
59      * <b>Openable::open</b> <b>Reaction</b>, if an <b>Openable</b> is passed.
60      * @return
61      *   A <b>Reaction</b> that sets the subject in an open state.
62      */
63     def open: Reaction
64 
65     /**
66      * A <b>Reaction</b> that sets the new location of the player to <b>RoomLink::endRoom</b>, and
67      * notifies the move to the user.
68      * @return
69      *   A <b>Reaction</b> that sets the new location of the player to <b>RoomLink::endRoom</b>, and
70      *   notifies the move to the user.
71      */
72     def enter: Reaction
73   }
74 
75   /**
76    * Standard implementation of <b>RoomLink</b>.
77    *
78    * @param endRoomRef
79    *   The [[RoomRef]] of the <b>Room</b> that will be set as the new location of the player, after
80    *   entering the subject.
81    * @param endRoomDirection
82    *   The [[Direction]] that will be associated to the end room, when available to enter.
83    * @param openable
84    *   If the subject have to be opened before entering, the <b>RoomLink</b> should include [[Some]]
85    *   <b>Openable</b> behavior here. If the subject is set as always open, it can be set to
86    *   [[None]].
87    * @param onEnterExtra
88    *   <b>Reaction</b> to be executed into the <b>State</b> when entered, after the standard
89    *   <b>Reaction</b>. It can be omitted.
90    */
91   case class SimpleRoomLink(
92     endRoomRef: RoomRef,
93     endRoomDirection: Direction,
94     openable: Option[Openable] = None,
95     onEnterExtra: Reaction = Reaction.empty
96   )(implicit val subject: I)
97     extends RoomLink
98     with Delegate {
99 
100     /**
101      * If the subject can also be opened, che associated <b>Openable</b> triggers are used as a
102      * delegate.
103      * @return
104      *   The <b>Openable</b> triggers, if the openable is passed; an empty [[PartialFunction]]
105      *   otherwise.
106      */
107     override def delegateTriggers: ItemTriggers =
108       openable map (_.triggers) getOrElse PartialFunction.empty: ItemTriggers
109 
110     override def receiverTriggers: ItemTriggers = {
111       case (Open, maybeKey, s)
112           if s.isInLocation(subject) && openable.forall(_.canBeOpened(maybeKey)(s))
113             && !isOpen && openable.isDefined =>
114         open
115 
116       case (Enter, None, s) if s.isInLocation(subject) && openable.forall(_.isOpen) =>
117         enter
118 
119       case (Enter, _, _) =>
120         Reaction.messages(CMessages.FailedToEnter(subject))
121     }
122 
123     override def enter: Reaction =
124       for {
125         s1 <- Reaction.empty
126         _  <- CReactions.switchLocation(endRoomRef)
127         _  <- CReactions.addMessage(CMessages.Navigated(endRoom(s1)))
128         s2 <- onEnterExtra
129       } yield s2
130 
131     override def open: Reaction =
132       for {
133         s1 <- openable map (_.open) getOrElse Reaction.empty
134         s2 <- CReactions.addDirectionToLocation(endRoomDirection, endRoom(s1))
135       } yield s2
136 
137     override def endRoom(implicit state: S): RM = state.rooms(endRoomRef)
138 
139     override def isOpen: Boolean = openable forall (_.isOpen)
140   }
141 
142   /**
143    * Companion object for <b>RoomLink</b>.
144    */
145   object RoomLink {
146 
147     /**
148      * Builder to generate a simple <b>RoomLink</b>, given a subject.
149      *
150      * @param endRoom
151      *   The <b>Room</b> that will be set as the new location of the player, after entering the
152      *   subject.
153      * @param endRoomDirection
154      *   The [[Direction]] that will be associated to the end room, when available to enter.
155      * @param openableBuilder
156      *   A function that returns the <b>Openable</b> behavior associated to the subject, when the
157      *   subject is passed.
158      * @param onEnterExtra
159      *   An extra behavior generated when player enter the target room.
160      * @return
161      *   Builder to generate a simple <b>RoomLink</b>, given a subject.
162      */
163     def builder(
164       endRoom: RM,
165       endRoomDirection: Direction,
166       openableBuilder: Option[I => Openable] = None,
167       onEnterExtra: Reaction = Reaction.empty
168     ): I => RoomLink =
169       i =>
170         SimpleRoomLink(
171           endRoom.ref,
172           endRoomDirection,
173           openableBuilder.map(_(i)),
174           onEnterExtra
175         )(i)
176 
177     /**
178      * Builder to generate a <b>RoomLink</b> (given a subject) that is open yet, an can be entered
179      * directly.
180      *
181      * @param endRoom
182      *   The <b>Room</b> that will be set as the new location of the player, after entering the
183      *   subject.
184      * @param endRoomDirection
185      *   The [[Direction]] that will be associated to the end room, when available to enter.
186      * @param onEnterExtra
187      *   An extra behavior generated when player enter the target room.
188      * @return
189      *   Builder to generate a simple <b>RoomLink</b>, given a subject.
190      */
191     def openedBuilder(
192       endRoom: RM,
193       endRoomDirection: Direction,
194       onEnterExtra: Reaction = Reaction.empty
195     ): I => RoomLink = builder(endRoom, endRoomDirection, None, onEnterExtra)
196 
197     /**
198      * Builder to generate a <b>RoomLink</b> (given a subject) that is closed without <b>Key</b>,
199      * and can be entered after being opened.
200      *
201      * @param endRoom
202      *   The <b>Room</b> that will be set as the new location of the player, after entering the
203      *   subject.
204      * @param endRoomDirection
205      *   The [[Direction]] that will be associated to the end room, when available to enter.
206      * @param onEnterExtra
207      *   An extra behavior generated when player enter the target room.
208      * @param onOpenExtra
209      *   An extra behavior generated when player opens the subject.
210      * @return
211      *   Builder to generate a simple <b>RoomLink</b>, given a subject.
212      */
213     def closedUnlockedBuilder(
214       endRoom: RM,
215       endRoomDirection: Direction,
216       onEnterExtra: Reaction = Reaction.empty,
217       onOpenExtra: Reaction = Reaction.empty
218     ): I => RoomLink =
219       builder(endRoom, endRoomDirection, Some(Openable.unlockedBuilder(onOpenExtra)), onEnterExtra)
220 
221     /**
222      * Builder to generate a <b>RoomLink</b> (given a subject) that is closed with a <b>Key</b>, and
223      * can be entered after being opened.
224      *
225      * @param key
226      *   The <b>Key</b> to be used to open the subject.
227      * @param endRoom
228      *   The <b>Room</b> that will be set as the new location of the player, after entering the
229      *   subject.
230      * @param endRoomDirection
231      *   The [[Direction]] that will be associated to the end room, when available to enter.
232      * @param onEnterExtra
233      *   An extra behavior generated when player enter the target room.
234      * @param onOpenExtra
235      *   An extra behavior generated when player opens the subject.
236      * @return
237      *   Builder to generate a simple <b>RoomLink</b>, given a subject.
238      */
239     def closedLockedBuilder(
240       key: Key,
241       endRoom: RM,
242       endRoomDirection: Direction,
243       onEnterExtra: Reaction = Reaction.empty,
244       onOpenExtra: Reaction = Reaction.empty
245     ): I => RoomLink =
246       builder(
247         endRoom,
248         endRoomDirection,
249         Some(Openable.lockedBuilder(key, onOpenExtra)),
250         onEnterExtra
251       )
252   }
