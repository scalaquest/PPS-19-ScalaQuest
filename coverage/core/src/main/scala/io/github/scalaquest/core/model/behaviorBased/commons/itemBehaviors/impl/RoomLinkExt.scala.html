13   /**
14    * An <b>ItemBehavior</b> associated to a <b>BehaviorBasedItem</b> that enables the possibility to
15    * change the player location when entered, like a <b>Door</b>. Conceptually, an subject that
16    * exposes a <b>RoomLink</b> behavior could also be opened before permitting to the player to
17    * enter.
18    */
19   abstract class RoomLink extends ItemBehavior {
20 
21     /**
22      * The "openness" state of the subject, as a boolean value. An open subject is conceptually
23      * "ready-to-use", as it can be entered. Always open, if the subject is not openable.
24      * @return
25      *   True if RoomLink is open, false otherwise.
26      */
27     def isOpen: Boolean
28 
29     /**
30      * If the subject have to be opened before entering, the <b>RoomLink</b> should include [[Some]]
31      * <b>Openable</b> behavior here. If the subject is set as always open, it can be set to
32      * [[None]].
33      * @return
34      *   [[Some]] with an <b>Openable</b> behavior if the subject should be opened before entering,
35      *   [[None]] otherwise.
36      */
37     def openable: Option[Openable]
38 
39     /**
40      * The <b>Room</b> that will be set as the new location of the player, after entering the
41      * subject.
42      * @param s
43      *   The current <b>State</b>.
44      * @return
45      *   The <b>Room</b> that will be set as the new location of the player, after entering the
46      *   subject.
47      */
48     def endRoom(implicit s: S): RM
49 
50     /**
51      * A <b>Reaction</b> that sets the subject in an open state. Internally, it should exploit the
52      * <b>Openable::open</b> <b>Reaction</b>, if an <b>Openable</b> is passed.
53      * @return
54      *   A <b>Reaction</b> that sets the subject in an open state.
55      */
56     def open: Reaction
57 
58     /**
59      * A <b>Reaction</b> that sets the new location of the player to <b>RoomLink::endRoom</b>, and
60      * notifies the move to the user.
61      * @return
62      *   A <b>Reaction</b> that sets the new location of the player to <b>RoomLink::endRoom</b>, and
63      *   notifies the move to the user.
64      */
65     def enter: Reaction
66   }
67 
68   /**
69    * Standard implementation of <b>RoomLink</b>.
70    *
71    * @param endRoomRef
72    *   The [[RoomRef]] of the <b>Room</b> that will be set as the new location of the player, after
73    *   entering the subject.
74    * @param endRoomDirection
75    *   The [[Direction]] that will be associated to the end room, when available to enter.
76    * @param openable
77    *   If the subject have to be opened before entering, the <b>RoomLink</b> should include [[Some]]
78    *   <b>Openable</b> behavior here. If the subject is set as always open, it can be set to
79    *   [[None]].
80    * @param onEnterExtra
81    *   <b>Reaction</b> to be executed into the <b>State</b> when entered, after the standard
82    *   <b>Reaction</b>. It can be omitted.
83    */
84   case class SimpleRoomLink(
85     endRoomRef: RoomRef,
86     endRoomDirection: Direction,
87     openable: Option[Openable] = None,
88     onEnterExtra: Reaction = Reaction.empty
89   )(implicit val subject: I)
90     extends RoomLink
91     with Delegate {
92 
93     /**
94      * If the subject can also be opened, che associated <b>Openable</b> triggers are used as a
95      * delegate.
96      * @return
97      *   The <b>Openable</b> triggers, if the openable is passed; an empty [[PartialFunction]]
98      *   otherwise.
99      */
100     override def delegateTriggers: ItemTriggers =
101       openable map (_.triggers) getOrElse PartialFunction.empty: ItemTriggers
102 
103     override def receiverTriggers: ItemTriggers = {
104       case (Open, maybeKey, s)
105           if s.isInLocation(subject) && openable.forall(_.canBeOpened(maybeKey)(s))
106             && !isOpen && openable.isDefined =>
107         open
108 
109       case (Enter, None, s) if s.isInLocation(subject) && openable.forall(_.isOpen) =>
110         enter
111 
112       case (Enter, _, _) =>
113         Reaction.messages(CMessages.FailedToEnter(subject))
114     }
115 
116     override def enter: Reaction =
117       for {
118         s1 <- Reaction.empty
119         _  <- CReactions.switchLocation(endRoomRef)
120         _  <- CReactions.addMessage(CMessages.Navigated(endRoom(s1)))
121         s2 <- onEnterExtra
122       } yield s2
123 
124     override def open: Reaction =
125       for {
126         s1 <- openable map (_.open) getOrElse Reaction.empty
127         s2 <- CReactions.addDirectionToLocation(endRoomDirection, endRoom(s1))
128       } yield s2
129 
130     override def endRoom(implicit state: S): RM = state.rooms(endRoomRef)
131 
132     override def isOpen: Boolean = openable forall (_.isOpen)
133   }

140     /**
141      * Builder to generate a simple <b>RoomLink</b>, given a subject.
142      *
143      * @param endRoom
144      *   The <b>Room</b> that will be set as the new location of the player, after entering the
145      *   subject.
146      * @param endRoomDirection
147      *   The [[Direction]] that will be associated to the end room, when available to enter.
148      * @param openableBuilder
149      *   A function that returns the <b>Openable</b> behavior associated to the subject, when the
150      *   subject is passed.
151      * @param onEnterExtra
152      *   An extra behavior generated when player enter the target room.
153      * @return
154      *   Builder to generate a simple <b>RoomLink</b>, given a subject.
155      */
156     def builder(
157       endRoom: RM,
158       endRoomDirection: Direction,
159       openableBuilder: Option[I => Openable] = None,
160       onEnterExtra: Reaction = Reaction.empty
161     ): I => RoomLink =
162       i =>
163         SimpleRoomLink(
164           endRoom.ref,
165           endRoomDirection,
166           openableBuilder.map(_(i)),
167           onEnterExtra
168         )(i)
169 
170     /**
171      * Builder to generate a <b>RoomLink</b> (given a subject) that is open yet, an can be entered
172      * directly.
173      *
174      * @param endRoom
175      *   The <b>Room</b> that will be set as the new location of the player, after entering the
176      *   subject.
177      * @param endRoomDirection
178      *   The [[Direction]] that will be associated to the end room, when available to enter.
179      * @param onEnterExtra
180      *   An extra behavior generated when player enter the target room.
181      * @return
182      *   Builder to generate a simple <b>RoomLink</b>, given a subject.
183      */
184     def openedBuilder(
185       endRoom: RM,
186       endRoomDirection: Direction,
187       onEnterExtra: Reaction = Reaction.empty
188     ): I => RoomLink = builder(endRoom, endRoomDirection, None, onEnterExtra)
189 
190     /**
191      * Builder to generate a <b>RoomLink</b> (given a subject) that is closed without <b>Key</b>,
192      * and can be entered after being opened.
193      *
194      * @param endRoom
195      *   The <b>Room</b> that will be set as the new location of the player, after entering the
196      *   subject.
197      * @param endRoomDirection
198      *   The [[Direction]] that will be associated to the end room, when available to enter.
199      * @param onEnterExtra
200      *   An extra behavior generated when player enter the target room.
201      * @param onOpenExtra
202      *   An extra behavior generated when player opens the subject.
203      * @return
204      *   Builder to generate a simple <b>RoomLink</b>, given a subject.
205      */
206     def closedUnlockedBuilder(
207       endRoom: RM,
208       endRoomDirection: Direction,
209       onEnterExtra: Reaction = Reaction.empty,
210       onOpenExtra: Reaction = Reaction.empty
211     ): I => RoomLink =
212       builder(endRoom, endRoomDirection, Some(Openable.unlockedBuilder(onOpenExtra)), onEnterExtra)
213 
214     /**
215      * Builder to generate a <b>RoomLink</b> (given a subject) that is closed with a <b>Key</b>, and
216      * can be entered after being opened.
217      *
218      * @param key
219      *   The <b>Key</b> to be used to open the subject.
220      * @param endRoom
221      *   The <b>Room</b> that will be set as the new location of the player, after entering the
222      *   subject.
223      * @param endRoomDirection
224      *   The [[Direction]] that will be associated to the end room, when available to enter.
225      * @param onEnterExtra
226      *   An extra behavior generated when player enter the target room.
227      * @param onOpenExtra
228      *   An extra behavior generated when player opens the subject.
229      * @return
230      *   Builder to generate a simple <b>RoomLink</b>, given a subject.
231      */
232     def closedLockedBuilder(
233       key: Key,
234       endRoom: RM,
235       endRoomDirection: Direction,
236       onEnterExtra: Reaction = Reaction.empty,
237       onOpenExtra: Reaction = Reaction.empty
238     ): I => RoomLink =
239       builder(
240         endRoom,
241         endRoomDirection,
242         Some(Openable.lockedBuilder(key, onOpenExtra)),
243         onEnterExtra
244       )
245   }
