26   /**
27    * Standard implementation of the RoomLink.
28    *
29    * This is a behavior associated to an item that is a link between two rooms (a door, for
30    * instance), and that could be opened to pass into it. The user can "Enter" into it, resulting
31    * into move it into the connected Room. another room.
32    * @param endRoomRef
33    *   The room into which the user is projected after entering the object.
34    * @param openable
35    *   The Openable behavior associated to the roomLink. If not passed, the item can be entered
36    *   without opening.
37    * @param onEnterExtra
38    *   Reaction to be executed into the State when entered, after the standard Reaction. It can be
39    *   omitted.
40    */
41   case class SimpleRoomLink(
42     endRoomRef: RoomRef,
43     endRoomDirection: Direction,
44     openable: Option[Openable] = None,
45     onEnterExtra: Reaction = Reaction.empty
46   )(implicit subject: I)
47     extends RoomLink
48     with Delegate {
49 
50     /**
51      * If an openable is passed, it is passed as father behavior.
52      * @return
53      *   The father behavior triggers.
54      */
55     override def delegateTriggers: ItemTriggers =
56       openable map (_.triggers) getOrElse PartialFunction.empty: ItemTriggers
57 
58     override def receiverTriggers: ItemTriggers = {
59       case (Open, _, maybeKey, s)
60           if s.isInLocation(subject) && openable.fold(true)(_.canBeOpened(maybeKey)(s))
61             && !isOpen && openable.isDefined =>
62         open
63 
64       case (Enter, _, None, s) if s.isInLocation(subject) && openable.fold(true)(_.isOpen) =>
65         enter
66       case (Enter, _, _, _) =>
67         Reaction.messages(Messages.FailedToEnter(subject))
68     }
69 
70     override def enter: Reaction =
71       s =>
72         Reaction.combine(
73           Reactions.enter(endRoom(s)),
74           onEnterExtra
75         )(s)
76 
77     override def open: Reaction =
78       Reaction.combine(
79         openable map (_.open) getOrElse Reaction.empty,
80         Reaction(
81           (locationRoomLens composeLens roomDirectionsLens)
82             .modify(_ + (endRoomDirection -> endRoomRef))
83         )
84       )
85 
86     override def endRoom(implicit s: S): RM = s.rooms(endRoomRef)
87 
88     override def isOpen: Boolean = openable forall (_.isOpen)
89   }
