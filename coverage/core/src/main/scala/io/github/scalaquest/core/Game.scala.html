13 /**
14  * A wrapper for the main game iteration, that given an input textual command and a State, it tries
15  * to interpret it and modify the state accordingly.
16  * @param model
17  *   The concrete type of the [[Model]] in use.
18  */
19 abstract class Game[M <: Model](val model: M) {
20 
21   /**
22    * The main game iteration. Given an input textual command and a State, it tries to interpret it
23    * and modify the state accordingly.
24    * @param input
25    *   A textual command from the user.
26    * @param state
27    *   The current [[Model.State]] of the game.
28    * @return
29    *   [[Right]] with the updated state and the resulting message, if no errors have been occurred;
30    *   [[Left]] with a string message error, instead.
31    */
32   def send(input: String)(state: model.S): Either[String, (model.S, Seq[Message])]
33 }
34 
35 /**
36  * Contains a default builder for [[Game]], using an external instance of Pipeline.
37  */
38 object Game {
39 
40   class GameBuilder[M <: Model](val model: M) {
41 
42     /**
43      * A partial implementation of [[Game]], using an external instance of Pipeline.
44      */
45     abstract class GameWithPipeline extends Game[model.type](model) {
46 
47       def partialBuilder: PartialBuilder[model.S, model.type]
48 
49       override def send(input: String)(state: model.S): Either[String, (model.S, Seq[Message])] =
50         partialBuilder(state) run input
51     }
52 
53     /**
54      * Builds a default instance of [[Game]], using an external instance of Pipeline.
55      */
56     def build(
57       _partialBuilder: PartialBuilder[model.S, model.type]
58     ): Game[model.type] =
59       new GameWithPipeline {
60         override def partialBuilder: PartialBuilder[model.S, model.type] = _partialBuilder
61       }
62   }
63 
64   /**
65    * Generates the default builder.
66    */
67   def builderFrom[M <: Model](model: M): GameBuilder[M] = new GameBuilder(model)
68 }
