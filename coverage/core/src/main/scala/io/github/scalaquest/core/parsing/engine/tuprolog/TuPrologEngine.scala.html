20 /**
21  * Engine implementation that uses a tuProlog engine under the hood.
22  * @note
23  *   this implementation computes solutions in a lazy way
24  */
25 private class TuPrologEngine(prolog: TuProlog) extends Engine {
26   import implicits._
27 
28   private def exploreSolutions(goal: TuPrologTerm): Seq[SolveInfo] = {
29     def go(info: SolveInfo): LazyList[SolveInfo] = {
30       val current =
31         if (info.isSuccess) LazyList(info)
32         else LazyList.empty
33       lazy val next = if (prolog.hasOpenAlternatives) go(prolog.solveNext()) else LazyList.empty
34       current #::: next
35     }
36 
37     go(prolog.solve(goal))
38   }
39 
40   override def solve(term: Term): Seq[Solution] =
41     exploreSolutions(term.toTuPrologTerm).map(TuPrologSolution)
42 }

44 object TuPrologEngine {
45   import implicits._
46 
47   /**
48    * Creates an [[Engine]] with the provided theory and libraries using tuProlog under the hood.
49    * @note
50    *   this implementation computes solutions lazily.
51    */
52   def apply(theory: Theory, libraries: Set[Library] = Set()): Engine =
53     new TuPrologEngine(createTuProlog(theory.toTuProlog, libraries))
54 
55   private def getArgs(struct: Struct): Seq[Term] = {
56     (0 until struct.getArity)
57       .map(i => struct.getTerm(i))
58       .map(_.toTerm)
59   }
60 
61   @nowarn("msg=exhaustive")
62   /**
63    * @note
64    *   this will fail if called with a Struct with no arguments.
65    */
66   private[tuprolog] def buildCompound(struct: Struct): Compound =
67     getArgs(struct).toList match {
68       case h :: t =>
69         Compound(Atom(struct.getName), h, t)
70     }
71 
72   /**
73    * @note
74    *   this might fail if not called with an Integer, but for now we don't use any other Number
75    *   implementation
76    */
77   private[tuprolog] def buildNumber(number: tuprolog.Number): scalog.Number =
78     Number(number.intValue)
79 
80   /**
81    * @note
82    *   this might throw if called with an invalid theory
83    */
84   private def createTuProlog(theory: TuPrologTheory, libraries: Set[Library]): TuProlog = {
85     val prolog = new TuProlog
86     try {
87       prolog.setTheory(theory)
88     } catch {
89       case e: alice.tuprolog.InvalidTheoryException => throw InvalidTheoryException(e)
90     }
91     libraries.foreach(l => prolog.loadLibrary(l.toTuProlog))
92     prolog
93   }
94 }
