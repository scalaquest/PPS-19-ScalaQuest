18 /**
19  * Engine implementation that uses a tuProlog engine under the hood.
20  * @note
21  *   this implementation computes solutions in a lazy way
22  */
23 private class TuPrologEngine(prolog: TuProlog) extends Engine {
24   import implicits._
25 
26   private def exploreSolutions(goal: TuPrologTerm): Seq[SolveInfo] = {
27     def go(info: SolveInfo): LazyList[SolveInfo] = {
28       val current =
29         if (info.isSuccess) LazyList(info)
30         else LazyList.empty
31       lazy val next = if (prolog.hasOpenAlternatives) go(prolog.solveNext()) else LazyList.empty
32       current #::: next
33     }
34 
35     go(prolog.solve(goal))
36   }
37 
38   override def solve(term: Term): Seq[Solution] =
39     exploreSolutions(term.toTuPrologTerm).map(TuPrologSolution)
40 }

42 object TuPrologEngine {
43   import implicits._
44 
45   /**
46    * Creates an [[Engine]] with the provided theory and libraries using tuProlog under the hood.
47    * @note
48    *   this implementation computes solutions lazily.
49    */
50   def apply(theory: Theory, libraries: Set[Library] = Set()): Engine =
51     new TuPrologEngine(createTuProlog(theory.toTuProlog, libraries))
52 
53   private def getArgs(struct: Struct): Seq[Term] = {
54     (0 until struct.getArity)
55       .map(i => struct.getTerm(i))
56       .map(_.toTerm)
57   }
58 
59   /**
60    * @note
61    *   this will fail if called with a Struct with no arguments.
62    */
63   private[tuprolog] def buildCompound(struct: Struct): Compound =
64     getArgs(struct).toList match {
65       case h :: t =>
66         Compound(Atom(struct.getName), h, t)
67     }
68 
69   /**
70    * @note
71    *   this might fail if not called with an Integer, but for now we don't use any other Number
72    *   implementation
73    */
74   private[tuprolog] def buildNumber(number: tuprolog.Number): scalog.Number =
75     Number(number.intValue)
76 
77   /**
78    * @note
79    *   this might throw if called with an invalid theory
80    */
81   private def createTuProlog(theory: TuPrologTheory, libraries: Set[Library]): TuProlog = {
82     val prolog = new TuProlog
83     try {
84       prolog.setTheory(theory)
85     } catch {
86       case e: alice.tuprolog.InvalidTheoryException => throw InvalidTheoryException(e)
87     }
88     libraries.foreach(l => prolog.loadLibrary(l.toTuProlog))
89     prolog
90   }
91 }
