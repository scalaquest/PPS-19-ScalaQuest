27 /**
28  * Engine implementation that uses a tuProlog engine under the hood.
29  * @note
30  *   this implementation computes solutions in a lazy way
31  */
32 private class TuPrologEngine(prolog: TuProlog) extends Engine {
33   import implicits._
34 
35   private def exploreSolutions(goal: TuPrologTerm): Seq[SolveInfo] = {
36     def go(info: SolveInfo): LazyList[SolveInfo] = {
37       val current =
38         if (info.isSuccess) LazyList(info)
39         else LazyList.empty
40       lazy val next = if (prolog.hasOpenAlternatives) go(prolog.solveNext()) else LazyList.empty
41       current #::: next
42     }
43 
44     go(prolog.solve(goal))
45   }
46 
47   override def solve(term: Term): Seq[Solution] =
48     exploreSolutions(term.toTuPrologTerm).map(TuPrologSolution)
49 }

51 object TuPrologEngine {
52   import implicits._
53 
54   /**
55    * Creates an [[Engine]] with the provided theory and libraries using tuProlog under the hood.
56    * @note
57    *   this implementation computes solutions lazily.
58    */
59   def apply(theory: Theory, libraries: Set[Library] = Set()): Engine =
60     new TuPrologEngine(createTuProlog(theory.toTuProlog, libraries))
61 
62   private def getArgs(struct: Struct): Seq[Term] = {
63     (0 until struct.getArity)
64       .map(i => struct.getTerm(i))
65       .map(_.toTerm)
66   }
67 
68   @nowarn("msg=exhaustive")
69   /**
70    * @note
71    *   this will fail if called with a Struct with no arguments.
72    */
73   private[tuprolog] def buildCompound(struct: Struct): Compound =
74     getArgs(struct).toList match {
75       case h :: t =>
76         Compound(Atom(struct.getName), h, t)
77     }
78 
79   /**
80    * @note
81    *   this might fail if not called with an Integer, but for now we don't use any other Number
82    *   implementation
83    */
84   private[tuprolog] def buildNumber(number: tuprolog.Number): scalog.Number =
85     Number(number.intValue)
86 
87   /**
88    * @note
89    *   this might throw if called with an invalid theory
90    */
91   private def createTuProlog(theory: TuPrologTheory, libraries: Set[Library]): TuProlog = {
92     val prolog = new TuProlog
93     try {
94       prolog.setTheory(theory)
95     } catch {
96       case e: alice.tuprolog.InvalidTheoryException => throw InvalidTheoryException(e)
97     }
98     libraries.foreach(l => prolog.loadLibrary(l.toTuProlog))
99     prolog
100   }
101 }

103 private[tuprolog] object implicits {
104 
105   implicit class EnhancedTerm(term: Term) {
106     def toTuPrologTerm: TuPrologTerm = TuPrologTerm.createTerm(term.generate)
107   }
108 
109   implicit class EnhancedTuPrologTerm(tuPrologTerm: TuPrologTerm) {
110 
111     @nowarn("msg=match.*exhaustive")
112     def toTerm: Term =
113       tuPrologTerm match {
114         case number: tuprolog.Number => buildNumber(number)
115         case struct: Struct =>
116           struct match {
117             case s if s.isCompound => buildCompound(s)
118             case a                 => Atom(a.getName)
119           }
120         case _var: Var => Variable(_var.getName)
121       }
122   }
