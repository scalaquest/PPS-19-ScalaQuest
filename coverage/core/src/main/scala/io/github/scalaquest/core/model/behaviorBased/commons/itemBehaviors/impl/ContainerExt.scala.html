17 /**
18  * The trait makes possible to add into the [[BehaviorBasedModel]] the <b>Container</b> behavior.
19  */
20 trait ContainerExt
21   extends BehaviorBasedModel
22   with CMessagesExt
23   with CReactionsExt
24   with StateUtilsExt
25   with OpenableExt {
26 
27   /**
28    * An <b>ItemBehavior</b> that enables the possibility to open a <b>BehaviorBasedItem</b>, and to
29    * reveal a set of items into the location when opened.
30    */
31   abstract class Container extends ItemBehavior {
32 
33     /**
34      * <b>BehaviorBasedItem</b>s that will be added to the location after opening the subject.
35      * @param state
36      *   The actual game state. Used to extract the actual <b>BehaviorBasedItem</b>s from their
37      *   refs.
38      * @return
39      *   <b>BehaviorBasedItem</b>s that will be added to the location after opening the subject.
40      */
41     def items(implicit state: S): Set[I]
42 
43     /**
44      * <b>Reaction</b> triggered when the subject is opened: it reveals a set of
45      * <b>BehaviorBasedItem</b>s into the current Room when opened.
46      * @return
47      *   <b>Reaction</b> triggered when the subject is opened.
48      */
49     def revealItems: Reaction
50 
51     /**
52      * The <b>Openable</b> behavior associated to the subject.
53      * @return
54      *   The <b>Openable</b> behavior associated to the subject.
55      */
56     def openable: Openable
57 
58     /**
59      * The "openness" state of the Item, initially closed.
60      * @return
61      *   True if the container is open, False otherwise.
62      */
63     def isOpen: Boolean
64   }
65 
66   /**
67    * The standard implementation for <b>Container</b>.
68    * @param itemRefs
69    *   Refs to the Items that will be added to the location after opening the subject.
70    * @param openable
71    *   The Openable behavior associated to the subject Item.
72    * @param subject
73    *   The item that owes the Container behavior.
74    */
75   case class SimpleContainer(
76     itemRefs: Set[ItemRef],
77     openable: Openable
78   )(implicit val subject: I)
79     extends Container
80     with Delegate {
81 
82     override def delegateTriggers: ItemTriggers = openable.triggers
83 
84     override def receiverTriggers: ItemTriggers = {
85       case (Open, maybeKey, s)
86           if s.isInLocation(subject) && openable.canBeOpened(maybeKey)(s)
87             && !openable.isOpen =>
88         revealItems
89     }
90 
91     override def items(implicit state: S): Set[I] = itemRefs.map(state.items(_))
92 
93     override def isOpen: Boolean = openable.isOpen
94 
95     override def revealItems: Reaction =
96       for {
97         s1 <- openable.open
98         _  <- CReactions.modifyLocationItems(_ ++ items(s1).map(_.ref))
99         s2 <- CReactions.addMessage(CMessages.ReversedIntoLocation(items(s1)))
100       } yield s2
101   }
102 
103   /**
104    * A companion object used to create <b>Container</b> builders.
105    */
106   object Container {
107 
108     /**
109      * A builder for a container that does not require a <b>Key</b> to be opened.
110      * @param items
111      *   Items that will be added to the location after opening the subject.
112      * @return
113      *   A builder for a container that does not require a Key to be opened.
114      */
115     def unlockedBuilder(
116       items: Set[I]
117     ): I => Container = i => SimpleContainer(items.map(_.ref), Openable.unlockedBuilder()(i))(i)
118 
119     /**
120      * A builder for a container that requires a <b>Key</b> to be opened.
121      * @param items
122      *   Items that will be added to the location after opening the subject.
123      * @param key
124      *   The Key required to open the containerized item.
125      * @return
126      *   A builder for a container that requires a Key to be opened.
127      */
128     def lockedBuilder(
129       items: Set[I],
130       key: Key
131     ): I => Container = i => SimpleContainer(items.map(_.ref), Openable.lockedBuilder(key)(i))(i)
132   }
133 }
