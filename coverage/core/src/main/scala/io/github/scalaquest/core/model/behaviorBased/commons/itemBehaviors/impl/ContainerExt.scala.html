20   /**
21    * An <b>ItemBehavior</b> that enables the possibility to open a <b>BehaviorBasedItem</b>, and to
22    * reveal a set of items into the location when opened.
23    */
24   abstract class Container extends ItemBehavior {
25 
26     /**
27      * <b>BehaviorBasedItem</b>s that will be added to the location after opening the subject.
28      * @param state
29      *   The actual game state. Used to extract the actual <b>BehaviorBasedItem</b>s from their
30      *   refs.
31      * @return
32      *   <b>BehaviorBasedItem</b>s that will be added to the location after opening the subject.
33      */
34     def items(implicit state: S): Set[I]
35 
36     /**
37      * <b>Reaction</b> triggered when the subject is opened: it reveals a set of
38      * <b>BehaviorBasedItem</b>s into the current Room when opened.
39      * @return
40      *   <b>Reaction</b> triggered when the subject is opened.
41      */
42     def revealItems: Reaction
43 
44     /**
45      * The <b>Openable</b> behavior associated to the subject.
46      * @return
47      *   The <b>Openable</b> behavior associated to the subject.
48      */
49     def openable: Openable
50 
51     /**
52      * The "openness" state of the Item, initially closed.
53      * @return
54      *   True if the container is open, False otherwise.
55      */
56     def isOpen: Boolean
57   }
58 
59   /**
60    * The standard implementation for <b>Container</b>.
61    * @param itemRefs
62    *   Refs to the Items that will be added to the location after opening the subject.
63    * @param openable
64    *   The Openable behavior associated to the subject Item.
65    * @param subject
66    *   The item that owes the Container behavior.
67    */
68   case class SimpleContainer(
69     itemRefs: Set[ItemRef],
70     openable: Openable
71   )(implicit val subject: I)
72     extends Container
73     with Delegate {
74 
75     override def delegateTriggers: ItemTriggers = openable.triggers
76 
77     override def receiverTriggers: ItemTriggers = {
78       case (Open, maybeKey, s)
79           if s.isInLocation(subject) && openable.canBeOpened(maybeKey)(s)
80             && !openable.isOpen =>
81         revealItems
82     }
83 
84     override def items(implicit state: S): Set[I] = itemRefs.map(state.items(_))
85 
86     override def isOpen: Boolean = openable.isOpen
87 
88     override def revealItems: Reaction =
89       for {
90         s1 <- openable.open
91         _  <- CReactions.modifyLocationItems(_ ++ items(s1).map(_.ref))
92         s2 <- CReactions.addMessage(CMessages.ReversedIntoLocation(items(s1)))
93       } yield s2
94   }
95 
96   /**
97    * A companion object used to create <b>Container</b> builders.
98    */
99   object Container {
100 
101     /**
102      * A builder for a container that does not require a <b>Key</b> to be opened.
103      * @param items
104      *   Items that will be added to the location after opening the subject.
105      * @return
106      *   A builder for a container that does not require a Key to be opened.
107      */
108     def unlockedBuilder(
109       items: Set[I]
110     ): I => Container = i => SimpleContainer(items.map(_.ref), Openable.unlockedBuilder()(i))(i)
111 
112     /**
113      * A builder for a container that requires a <b>Key</b> to be opened.
114      * @param items
115      *   Items that will be added to the location after opening the subject.
116      * @param key
117      *   The Key required to open the containerized item.
118      * @return
119      *   A builder for a container that requires a Key to be opened.
120      */
121     def lockedBuilder(
122       items: Set[I],
123       key: Key
124     ): I => Container = i => SimpleContainer(items.map(_.ref), Openable.lockedBuilder(key)(i))(i)
125   }
