17 /**
18  * A basic game iteration, from a command to a modified state. The concept of Pipeline implicitly
19  * comprehends the initial State.
20  */
21 abstract class Pipeline[M <: Model](val model: M) {
22   def run(rawSentence: String): Either[String, (model.S, Seq[Message])]
23 }
24 
25 /**
26  * Contains a default builder for [[Pipeline]], splitting it into its components.
27  */
28 object Pipeline {
29 
30   type PartialBuilder[S <: Model#State, M <: Model] = S => Pipeline[M]
31 
32   def builderFrom[M <: Model](model: M) = new PipelineBuilder[M](model)
33 
34   class PipelineBuilder[M <: Model](val model: M) {
35 
36     def build(
37       lexer: Lexer,
38       parser: Parser,
39       resolver: Resolver.Builder[model.S],
40       interpreterBuilder: Interpreter.Builder[model.type, model.S, model.Reaction],
41       reducerBuilder: Reducer.Builder[model.type, model.S, model.Reaction]
42     )(state: model.S): Pipeline[model.type] =
43       new Pipeline[model.type](model) {
44 
45         override def run(rawSentence: String): Either[String, (model.S, Seq[Message])] =
46           for {
47             lr  <- Right(lexer tokenize rawSentence)
48             pr  <- (parser parse lr) toRight "Couldn't understand input."
49             rr  <- resolver(state) resolve pr
50             ir  <- interpreterBuilder(state) interpret rr
51             rdr <- Right(reducerBuilder(state) reduce ir)
52           } yield (rdr.state, rdr.message)
53       }
54   }
55 
56 }
