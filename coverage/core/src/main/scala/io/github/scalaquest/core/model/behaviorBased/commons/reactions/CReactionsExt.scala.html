15 /**
16  * A mixable trait that contains some common <b>Reactions</b>.
17  */
18 trait CReactionsExt extends BehaviorBasedModel with ReactionUtilsExt with CMessagesExt {
19 
20   /**
21    * Companion Object with the effective implementation of the common [[Reaction]] s.
22    */
23   object CReactions {
24 
25     /**
26      * <b>Reaction</b> that modifies the content of the player's bag, following the instructions
27      * from the given function.
28      * @param modifier
29      *   A function that indicates how to modify the bag.
30      * @return
31      *   <b>Reaction</b> that modifies the content of the player's bag.
32      */
33     def modifyBag(modifier: Set[ItemRef] => Set[ItemRef]): Reaction =
34       Reaction(bagLens.modify(modifier))
35 
36     /**
37      * <b>Reaction</b> that modifies the items contained into the player's location, following the
38      * instructions from the given function.
39      * @param modifier
40      *   A function that indicates how to modify the items contained into the player's location.
41      * @return
42      *   <b>Reaction</b> that modifies the items contained into the player's location.
43      */
44     def modifyLocationItems(modifier: Set[ItemRef] => Set[ItemRef]): Reaction =
45       s => modifyRoomItems(s.location.ref, modifier)(s)
46 
47     /**
48      * <b>Reaction</b> that changes the location of the player, given a <b>RoomRef</b>.
49      * @param location
50      *   The new location of the player, as a <b>RoomRef</b>.
51      * @return
52      *   <b>Reaction</b> that switches the location of the player.
53      */
54     def switchLocation(location: RoomRef): Reaction = Reaction(locationLens.set(location))
55 
56     /**
57      * <b>Reaction</b> that changes the location of the player.
58      * @param location
59      *   The new location of the player.
60      * @return
61      *   <b>Reaction</b> that switches the location of the player.
62      */
63     def switchLocation(location: RM): Reaction = switchLocation(location.ref)
64 
65     /**
66      * <b>Reaction</b> that adds a <b>Message</b> to the ones to be returned.
67      * @param message
68      *   The new <b>Message</b>.
69      * @return
70      *   <b>Reaction</b> that adds a <b>Message</b> to the ones to be returned.
71      */
72     def addMessage(message: Message): Reaction = Reaction.messages(message)
73 
74     /**
75      * Modifies the set of items of the given room.
76      * @param roomRef
77      *   The ref of the room to modify.
78      * @param itemsModifier
79      *   A function that indicates how to modify the items contained into the room.
80      * @return
81      *   Modifies the set of items of the given room.
82      */
83     def modifyRoomItems(roomRef: RoomRef, itemsModifier: Set[ItemRef] => Set[ItemRef]): Reaction =
84       s => {
85         val modifiedRoom = roomItemsLens.modify(itemsModifier)(roomsLens.get(s)(roomRef))
86         val react        = Reaction(roomsLens.modify(_ + (roomRef -> modifiedRoom)))
87         react(s)
88       }
89 
90     /**
91      * <b>Reaction</b> that adds a neighbor <b>Room</b> to the actual player location.
92      * @param direction
93      *   the new visible direction.
94      * @param room
95      *   the new visible room.
96      * @return
97      *   the Reaction described above.
98      */
99     def addDirectionToLocation(direction: Direction, room: RM): Reaction =
100       Reaction(
101         (locationRoomLens composeLens roomDirectionsLens).modify(_ + (direction -> room.ref))
102       )
103 
104     /**
105      * <b>Reaction</b> that ends the match in two possible scenario:
106      *   - The player wins
107      *   - The player loses
108      * @param win
109      *   True if game is finished with a victory, False for a defeat.
110      * @return
111      *   the Reaction cited above.
112      */
113     def finishGame(win: Boolean): Reaction =
114       for {
115         _ <- Reaction(matchEndedLens.set(true))
116         s <- Reaction.messages(if (win) CMessages.Won else CMessages.Lost)
117       } yield s
118   }
119 }
