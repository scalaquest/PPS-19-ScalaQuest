67     /**

69      * @param roomRef
70      *   The ref of the room to modify.
71      * @param itemsModifier
72      *   A function that indicates how to modify the items contained into the room.
73      * @return
74      *   Modifies the set of items of the given room.
75      */
76     def modifyRoomItems(roomRef: RoomRef, itemsModifier: Set[ItemRef] => Set[ItemRef]): Reaction =
77       s => {
78         val modifiedRoom = roomItemsLens.modify(itemsModifier)(roomsLens.get(s)(roomRef))
79         val react        = Reaction(roomsLens.modify(_ + (roomRef -> modifiedRoom)))
80         react(s)
81       }
82 
83     /**
84      * <b>Reaction</b> that adds a neighbor <b>Room</b> to the actual player location.
85      * @param direction
86      *   the new visible direction.
87      * @param room
88      *   the new visible room.
89      * @return
90      *   the Reaction described above.
91      */
92     def addDirectionToLocation(direction: Direction, room: RM): Reaction =
93       Reaction(
94         (locationRoomLens composeLens roomDirectionsLens).modify(_ + (direction -> room.ref))
95       )
96 
97     /**
98      * <b>Reaction</b> that ends the match in two possible scenario:
99      *   - The player wins
100      *   - The player loses
101      * @param win
102      *   True if game is finished with a victory, False for a defeat.
103      * @return
104      *   the Reaction cited above.
105      */
106     def finishGame(win: Boolean): Reaction =
107       for {
108         _ <- Reaction(matchEndedLens.set(true))
109         s <- Reaction.messages(if (win) CMessages.Won else CMessages.Lost)
110       } yield s
111   }
