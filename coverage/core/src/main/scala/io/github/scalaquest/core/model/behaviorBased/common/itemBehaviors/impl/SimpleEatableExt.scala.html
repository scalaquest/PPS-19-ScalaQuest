12 
13   /**
14    * Standard implementation of the Eatable behavior.
15    *
16    * This is a behavior associated to an Item that can be eaten, if present in the bag or in the
17    * room.
18    * @param onEatExtra
19    *   Reaction to be executed when the item has been successfully eaten, after the standard
20    *   [[Reaction]]. It can be omitted.
21    */
22   case class SimpleEatable(onEatExtra: Option[Reaction] = None)(implicit
23     playerBagLens: Lens[S, Set[ItemRef]],
24     matchRoomsLens: Lens[S, Map[RoomRef, RM]],
25     roomLens: Lens[RM, Set[ItemRef]],
26     messageLens: Lens[S, Seq[Message]]
27   ) extends Eatable {
28 
29     override def triggers: ItemTriggers = {
30       // "Eat the item"
31       case (Eat, item, None, state) if state.isInScope(item) => eat(item)
32     }
33 
34     def eat(item: I): Reaction =
35       state => {
36         val updCurrRoom = roomLens.modify(_ - item.ref)(state.currentRoom)
37         state.applyReactions(
38           matchRoomsLens.modify(_ + (updCurrRoom.ref -> updCurrRoom)),
39           playerBagLens.modify(_ - item.ref),
40           messageLens.modify(_ :+ Eaten(item)),
41           onEatExtra.getOrElse(state => state)
42         )
43       }
44   }
45 }
