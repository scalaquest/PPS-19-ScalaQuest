17 /**
18  * Extension for the model. Adds a base implementation of the [[Model.State]].
19  */
20 trait SimpleStateExt extends BehaviorBasedModel with CGroundExt {
21 
22   override type S = SimpleState
23 
24   /**
25    * Class that implement the [[Model.State]] interface.
26    * @param actions
27    *   [[Map]] that found for each <b>verb</b> the specific [[Action]].
28    * @param rooms
29    *   [[Map]] that found for each <b>room</b> reference the updated [[Room]].
30    * @param items
31    *   [[Map]] that found for each <b>item</b> reference the updated [[Item]].
32    * @param ground
33    *   <b>Ground</b> implementation used in this <b>State</b>.
34    * @param _bag
35    *   Player's <b>bag</b> reference.
36    * @param _location
37    *   <b>Room</b> reference where player start the match.
38    * @param ended
39    *   True if game is ended, false otherwise.
40    */
41   case class SimpleState(
42     actions: Map[VerbPrep, Action],
43     rooms: Map[RoomRef, RM],
44     items: Map[ItemRef, I],
45     ground: G,
46     _bag: Set[ItemRef],
47     _location: RoomRef,
48     ended: Boolean = false
49   ) extends State {
50     override def bag: Set[I]  = _bag.flatMap(items.get)
51     override def location: RM = rooms(_location)
52   }
53 
54   /**
55    * Companion object with apply for SimpleState.
56    */
57   object State {
58 
59     /**
60      * Facility methods for SimpleState.
61      * @param actions
62      *   [[Map]] that found for each <b>verb</b> the specific [[Action]].
63      * @param rooms
64      *   [[Map]] that found for each <b>room</b> reference the updated [[Room]].
65      * @param items
66      *   [[Map]] that found for each <b>item</b> reference the updated [[Item]].
67      * @param ground
68      *   <b>Ground</b> implementation used in this <b>State</b>.
69      * @param ended
70      *   True if game is ended, false otherwise.
71      */
72     def apply(
73       actions: Map[VerbPrep, Action],
74       rooms: Map[RoomRef, RM],
75       items: Map[ItemRef, I],
76       ground: G = CGround(),
77       bag: Set[ItemRef] = Set.empty,
78       location: RoomRef,
79       ended: Boolean = false
80     ): S = SimpleState(actions, rooms, items, ground, bag, location, ended)
81   }
82 
83   override def roomsLens: Lens[S, Map[RoomRef, RM]] = GenLens[S](_.rooms)
84   override def itemsLens: Lens[S, Map[ItemRef, I]]  = GenLens[S](_.items)
85   override def matchEndedLens: Lens[S, Boolean]     = GenLens[S](_.ended)
86   override def bagLens: Lens[S, Set[ItemRef]]       = GenLens[S](_._bag)
87   override def locationLens: Lens[S, RoomRef]       = GenLens[S](_._location)
88 }
