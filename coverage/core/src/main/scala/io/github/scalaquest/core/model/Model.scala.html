6 /**
7  * A way to represent the basic linked concepts of the story, in an extendible way. Usage example:
8  * {{{
9  *
10  *   object MyModel extends Model with SimpleState with SimpleGround with MyItems
11  *   ...
12  *   val myItemImpl = MyModel.MySword("a sword!", Weapon())
13  *   ...
14  * }}}
15  */
16 trait Model {
17   type S <: State
18   type I <: Item
19   type G <: Ground
20   type RM <: Room
21 
22   type Reaction = S => S
23 
24   /**
25    * Represents a snapshot of the current game, at an higher level in comparison to [[MatchState]].
26    * The key fact is that this level of abstraction can handle also [[Message]] s, that is a
27    * representation of the output to render to the user at the end of the pipeline round, and all
28    * the possible [[Action]] s.
29    */
30   abstract class State { self: S =>
31 
32     /**
33      * All the possible [[Action]] s that can be used into the match, associated with the Verb that
34      * triggers it.
35      * @return
36      *   All the possible [[Action]] s, as a [[Map]].
37      */
38     def actions: Map[VerbPrep, Action]
39 
40     /**
41      * The state of the game, in a vision scoped to the player capabilities.
42      * @return
43      *   The [[MatchState]] of the match.
44      */
45     def matchState: MatchState[I, RM]
46 
47     /**
48      * A representation of the occurred events in a given the pipeline round. The storyteller can
49      * then decide what to show for each one.
50      * @return
51      *   A [[Seq]] of [[Message]] s.
52      */
53     def messages: Seq[Message]
54   }
55 
56   /**
57    * Represents a single object against which the [[Player]] can interact.
58    */
59   abstract class Item { item: I =>
60 
61     /**
62      * Semantic description of the object.
63      * @return
64      *   An [[ItemDescription]] describing the semantic of the object.
65      */
66     def description: ItemDescription
67 
68     /**
69      * A textual representation of the [[Item]].
70      * @return
71      */
72     def name: String = description.mkString
73 
74     /**
75      * The unique identifier of the [[Item]]. This is necessary, as passing from a state to another,
76      * the reference to an object changes, the [[State]] works in an immutable fashion.
77      */
78     def ref: ItemRef
79 
80     /**
81      * Define a way make the item interact with the [[State]]. The interaction is founded into the
82      * [[Action]] (ex. 'open the door'), but it can include also an additional [[Item]] ('open the
83      * door with the key'). The item should react to each combination with an appropriate
84      * [[Reaction]].
85      * @param action
86      *   An action from a statement (ex. <b>open</b> the door with a key).
87      * @param state
88      *   the starting state, useful to give a context to the decision to take.
89      * @param sideItem
90      *   A side [[Item]] from a statement (ex. open the door with <b>a key</b>).
91      * @return
92      *   The resulting [[Reaction]] from the combination, or a [[None]] if the match fails.
93      */
94     def use(action: Action, state: S, sideItem: Option[I] = None): Option[Reaction]
95   }
96 
97   /**
98    * The ground is a sort of "virtual" item, that handles the intransitive [[Action]] s linked to
99    * the match.
100    */
101   abstract class Ground { ground: G =>
102 
103     /**
104      * Define a way make the given [[Action]] interacts with the [[State]]. The interaction is
105      * founded into a single [[Action]] value (ex. 'inspect'). The ground should react to each
106      * combination with an appropriate [[Reaction]].
107      * @param action
108      *   An action from a statement (ex. <b>inspect</b> ).
109      * @param state
110      *   the starting state, useful to give a context to the decision to take.
111      * @return
112      *   The resulting [[Reaction]] from the combination, or a [[None]] if the match fails.
113      */
114     def use(action: Action, state: S): Option[Reaction]
115   }
116 
117   /**
118    * A geographical portion of the match map.
119    *
120    * This is one of the basic block for the story build by the storyteller, as it is used to
121    * identify [[Player]] 's and [[Model.Item]] s' location, in a given moment of the story, and
122    * navigate across the match geography.
123    */
124   abstract class Room { room: RM =>
125 
126     /**
127      * A textual identifier for the room.
128      * @return
129      *   A textual identifier for the room.
130      */
131     def name: String
132 
133     /**
134      * The unique identifier of the [[Room]]. This is necessary, as passing from a state to another,
135      * the reference to an object changes, the [[State]] works in an immutable fashion.
136      */
137     def ref: RoomRef
138 
139     /**
140      * Identifies [[Room]] s near to the current one, at the cardinal points.
141      */
142     def neighbor(direction: Direction): Option[RoomRef]
143 
144     /**
145      * Identifies the [[Item]] s positioned into the current [[Room]].
146      * @return
147      *   [[Item]] s positioned into the [[Room]] as a [[Set]]
148      */
149     def items: Set[ItemRef]
150   }
151 
152   /**
153    * Lens used to empty [[Message]] s after each pipeline round.
154    * @return
155    */
156   implicit def messageLens: Lens[S, Seq[Message]]
157 }
