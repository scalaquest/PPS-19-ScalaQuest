6 /**
7  * A way to represent the basic linked concepts of the story, in an extendible way. Usage example:
8  * {{{
9  *
10  *   object MyModel extends Model with SimpleState with SimpleGround with MyItems
11  *   ...
12  *   val myItemImpl = MyModel.MySword("a sword!", Weapon())
13  *   ...
14  * }}}
15  */

23   /**
24    * Represents a snapshot of the current game, at an higher level in comparison to [[MatchState]].
25    * The key fact is that this level of abstraction can handle also [[Message]] s, that is a
26    * representation of the output to render to the user at the end of the pipeline round, and all
27    * the possible [[Action]] s.
28    */
29   abstract class State { self: S =>

38     def actions: Map[VerbPrep, Action]

40     /**

42      * @return

44      */
45     def rooms: Map[RoomRef, RM]

118     /**
119      * Define a way make the item interact with the [[State]]. The interaction is founded into the
