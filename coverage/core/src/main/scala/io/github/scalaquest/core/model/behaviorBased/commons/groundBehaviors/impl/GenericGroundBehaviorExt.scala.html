14 /**
15  * The trait makes possible to mix into a [[BehaviorBasedModel]] the <b>GenericGroundBehavior
16  * GroundBehavior</b>.
17  */
18 trait GenericGroundBehaviorExt extends BehaviorBasedModel with CMessagesExt with CReactionsExt {
19 
20   /**
21    * A GroundBehavior with some triggers. It is useful in order to wrap <b>GroundBehavior</b> into a
22    * facility builder.
23    */
24   abstract class GenericGroundBehavior extends GroundBehavior {
25 
26     /**
27      * <b>GroundTriggers</b> associated to the behavior.
28      * @return
29      *   Triggers associated to the behavior.
30      */
31     def triggers: GroundTriggers
32   }
33 
34   /**
35    * Standard implementation of <b>GenericGroundBehavior</b>.
36    */
37   case class SimpleGenericGroundBehavior(
38     override val triggers: GroundTriggers
39   ) extends GenericGroundBehavior
40 
41   /**
42    * Companion object for <b>GenericGroundBehavior</b>.
43    */
44   object GenericGroundBehavior {
45 
46     /**
47      * A <b>GenericBehavior</b> built by its triggers.
48      * @param triggers
49      *   The <b>GroundTriggers</b> associated to the ground.
50      * @return
51      *   A <b>GenericBehavior</b> built by its triggers.
52      */
53     def apply(triggers: GroundTriggers = PartialFunction.empty): GenericGroundBehavior =
54       SimpleGenericGroundBehavior(triggers)
55   }
56 }
