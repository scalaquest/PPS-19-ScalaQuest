13 /**
14  * This package provides two type classes `Generator` and `GeneratorK`. They are composable wrappers
15  * around functions. In addition, there are facility methods that allow the creation of these type
16  * classes.
17  *
18  * For example:
19  * {{{
20  *   implicit val generator = Generator.makeEntry((x: Int) => x -> Math.pow(x, 2))
21  *   val generatorK = new GeneratorK[List, Int, Map[Int, Double]]
22  * }}}
23  *
24  * A [[GeneratorK]] can create a Map from a List.
25  * {{{
26  *   generatorK.generate(List(1, 2, 3))
27  * }}}
28  *
29  * Should result in the following being printed:
30  * {{{
31  *   Map(
32  *     1 -> 1,
33  *     2 -> 4,
34  *     3 -> 9
35  *   )
36  * }}}
37  */
38 package object generators extends GeneratorImplicits {

40   /**
41    * Allows to fold a `F[A]`.
42    *
43    * Inspired by:
44    * [[https://typelevel.org/cats/typeclasses/monoid.html#example-usage-collapsing-a-list]].
45    */
46   def combineAll[F[_]: Foldable, A: Monoid](as: F[A]): A = Foldable[F].fold(as)
47 
48   /**
49    * Variadic argument function that wraps the non-variadic `combineAll` one.
50    */
51   def combineAll[A: Monoid](a: A, as: A*): A = combineAll((a +: as).toList)
