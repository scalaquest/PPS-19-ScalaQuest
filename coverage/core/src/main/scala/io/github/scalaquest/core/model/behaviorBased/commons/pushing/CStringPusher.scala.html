51   @tailrec
52   private def printItems(tail: List[CMessagesExt#I], acc: String = ""): String = {
53     tail match {
54       case ::(head, Nil)  => s"${acc}a ${head.toString}."
55       case ::(head, next) => printItems(next, s"${acc}a ${head.toString}, ")
56       case Nil            => "nothing."
57     }
58   }
59 
60   private def printNeighbors(neighbors: Map[Direction, CMessagesExt#RM]): String = {
61     neighbors match {
62       case ns if ns.isEmpty => "\nYou cannot go anywhere now."
63       case ns =>
64         ns.map(n => s"There is a ${n._2.toString} in direction ${n._1.toString}.")
65           .fold("")(_ + "\n" + _)
66     }
67   }
