16 /**
17  * A [[ComposableStringPusher]] that have can handle the messages contained into [[CMessagesExt]].
18  *
19  * @param model
20  *   contains the Messages to interpret.
21  */
22 abstract class CStringPusher(model: BehaviorBasedModel with CMessagesExt)
23   extends ComposableStringPusher {
24   import model.CMessages._
25 
26   /**
27    * Recognize every common message and output its content.
28    * @return
29    */
30   final def base: StringMessageTriggers = {
31     case Inspected(room, items, neighbors) =>
32       val ordItems = items.toList.sortWith(_.toString < _.toString)
33       s"The ${room.name} contains ${printItems(ordItems)}${printNeighbors(neighbors)}"
34 
35     case InspectedBag(items) =>
36       val ordItems = items.toList.sortWith(_.toString < _.toString)
37       s"The bag contains ${printItems(ordItems)}"
38 
39     case ReversedIntoLocation(items) =>
40       val ordItems = items.toList.sortWith(_.toString < _.toString)
41       s"Current room now contains also ${printItems(ordItems)}!"
42 
43     case Eaten(item)  => s"The $item has been eaten!"
44     case Taken(item)  => s"The $item has been taken!"
45     case Opened(item) => s"The $item has been opened!"
46 
47     case AlreadyOpened(item)         => s"The $item is already opened!"
48     case FailedToOpen(item)          => s"The $item failed to open!"
49     case FailedToEnter(item)         => s"The $item is locked! You cannot enter now."
50     case Navigated(room)             => s"You entered $room!"
51     case FailedToNavigate(direction) => s"You cannot go $direction from here!"
52     case Print(msg)                  => msg
53     case Welcome(msg)                => msg
54     case Won                         => "You win!"
55     case Lost                        => "You lose!"
56   }
57 
58   @tailrec
59   private def printItems(tail: List[CMessagesExt#I], acc: String = ""): String = {
60     tail match {
61       case ::(head, Nil)  => s"${acc}a ${head.toString}."
62       case ::(head, next) => printItems(next, s"${acc}a ${head.toString}, ")
63       case Nil            => "nothing."
64     }
65   }
66 
67   private def printNeighbors(neighbors: Map[Direction, CMessagesExt#RM]): String = {
68     neighbors match {
69       case ns if ns.isEmpty => "\nYou cannot go anywhere now."
70       case ns =>
71         ns.map(n => s"There is a ${n._2.toString} in direction ${n._1.toString}.")
72           .fold("")(_ + "\n" + _)
73     }
74   }
75 }

77 /**
78  * Object with a useful construct for create a [[CStringPusher]].
79  */
80 object CStringPusher {
81 
82   /**
83    * Create the pusher giving him only the game's model.
84    * @param model
85    *   the actual model game.
86    * @param _additionalTriggers
87    *   add some customized triggers with regard to the base ones.
88    * @return
89    *   an instance of CommonStringPusher.
90    */
91   def apply(
92     model: CMessagesExt,
93     _additionalTriggers: MessageTriggers[String] = PartialFunction.empty
94   ): CStringPusher = {
95     new CStringPusher(model) {
96       override def extra: MessageTriggers[String] = _additionalTriggers
97     }
98   }
99 }
