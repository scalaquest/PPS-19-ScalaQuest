5 /**
6  * This is a partial implementation of [[Model]], following a <b>behavior-based fashion</b>. This is
7  * a style that enables the "decoration" of [[Model.Item]] s and the [[Model.Ground]] by the use of
8  * the <b>Behavior</b>: a sort of composable plugin that matches dynamically various
9  * action-item-state scenarios.
10  */
11 abstract class BehaviorBasedModel extends Model {
12   override type I = BehaviorBasedItem
13 
14   /**
15    * A [[Item]] capable to process [[ItemBehavior]] s to react to specific action-item-state
16    * combinations.
17    *
18    * Items that exploit the behavioral mechanism should extend this partial implementation. This
19    * because the handling of use* functions following this style (by exploring the available
20    * triggers) is implemented here.
21    */
22   abstract class BehaviorBasedItem extends Item {
23 
24     /**
25      * [[ItemBehavior]] s associated to the Item.
26      *
27      * @return
28      *   [[ItemBehavior]] s associated to the Item.
29      */
30     def behaviors: Seq[ItemBehavior] = Seq.empty
31 
32     override def use(action: Action, maybeSideItem: Option[I])(implicit
33       state: S
34     ): Option[Reaction] =
35       behaviors
36         .map(_.triggers)
37         .reduceOption(_ orElse _)
38         .getOrElse(PartialFunction.empty)
39         .lift((action, this, maybeSideItem, state))
40   }
41 
42   /**
43    * A [[PartialFunction]], that makes possible to react to specific combinations
44    * action-item-sideItem-state (or action-item-state) responding with a [[Reaction]]. The
45    * [[ItemBehavior]] is based into this construct.
46    */
47   type ItemTriggers = PartialFunction[(Action, I, Option[I], S), Reaction]
48 
49   /**
50    * Makes a [[BehaviorBasedItem]] react to specific [[ItemTriggers]] with a [[Reaction]]. The
51    * [[BehaviorBasedModel]] is based into this construct.
52    */
53   abstract class ItemBehavior {
54     def triggers: ItemTriggers = PartialFunction.empty
55   }
56 
57   /**
58    * A mixin for [[ItemBehavior]], that enables the possibility implement the delegation pattern for
59    * them. It means that the receiver behavior (the one that mixes in [[Delegate]] ) can owe
60    * internally another one, inheriting its [[ItemTriggers]]. If the receiver overwrites some of the
61    * delegate triggers, the receiver ones have the priority, and the delegate ones are not executed.
62    */
63   trait Delegate extends ItemBehavior {
64 
65     /**
66      * Should be overrode with the delegate [[ItemBehavior]] 's [[ItemTriggers]].
67      *
68      * @return
69      *   The delegate [[ItemBehavior]] 's [[ItemTriggers]].
70      */
71     def delegateTriggers: ItemTriggers
72 
73     /**
74      * Should be overrode with the [[ItemTriggers]] of the receiver (the behavior that mixes in
75      * [[Delegate]] ).
76      *
77      * @return
78      *   The [[ItemTriggers]] of the receiver (the behavior that mixes in [[Delegate]] ).
79      */
80     def receiverTriggers: ItemTriggers = PartialFunction.empty
81 
82     /**
83      * Composes together all the [[ItemTriggers]], both of the receiver and of the delegate. If the
84      * receiver overwrites some of the delegate triggers, the receiver ones have the priority, and
85      * the delegate ones are not executed.
86      *
87      * @return
88      *   The combined triggers.
89      */
90     final override def triggers: ItemTriggers =
91       Seq(receiverTriggers, delegateTriggers).reduce(_ orElse _)
92   }
93 
94   override type G = BehaviorBasedGround
95 
96   /**
97    * A [[Ground]] capable to process [[GroundBehavior]] s to react to specific action-state
98    * combinations.
99    *
100    * A Ground that exploit the behavioral mechanism should extend this partial implementation. This
101    * because the handling of use* functions following this style (by exploring the available
102    * triggers) is implemented here.
103    */
104   abstract class BehaviorBasedGround extends Ground {
105     def behaviors: Seq[GroundBehavior] = Seq.empty
106 
107     override def use(action: Action)(implicit state: S): Option[Reaction] =
108       behaviors
109         .map(_.triggers)
110         .reduceOption(_ orElse _)
111         .getOrElse(PartialFunction.empty)
112         .lift((action, state))
113   }
114 
115   /**
116    * A [[PartialFunction]], that makes possible to react to specific combinations action-state,
117    * responding with a [[Reaction]]. The [[GroundBehavior]] is based into this construct.
118    */
119   type GroundTriggers = PartialFunction[(Action, S), Reaction]
120 
121   /**
122    * Makes a [[BehaviorBasedGround]] react to specific [[GroundTriggers]] with a [[Reaction]]. The
123    * [[BehaviorBasedModel]] is based into this construct.
124    */
125   abstract class GroundBehavior {
126     def triggers: GroundTriggers = PartialFunction.empty
127   }
128 }
