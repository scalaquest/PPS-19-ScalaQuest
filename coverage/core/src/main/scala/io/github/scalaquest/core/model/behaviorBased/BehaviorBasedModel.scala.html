10 import io.github.scalaquest.core.model.{Action, Model}

12 /**
13  * This is a partial implementation of [[Model]], following a <b>behavior-based fashion</b>. This is
14  * a style that enables the "decoration" of [[Model.Item]] s and the [[Model.Ground]] by the use of
15  * the <b>Behavior</b>: a sort of composable plugin that matches dynamically various
16  * action-item-state scenarios.
17  */
18 abstract class BehaviorBasedModel extends Model {
19   override type I = BehaviorBasedItem
20 
21   /**
22    * A [[Item]] capable to process [[ItemBehavior]] s to react to specific action-item-state
23    * combinations.
24    *
25    * Items that exploit the behavioral mechanism should extend this partial implementation. This
26    * because the handling of use* functions following this style (by exploring the available
27    * triggers) is implemented here.
28    */
29   abstract class BehaviorBasedItem extends Item {
30 
31     /**
32      * [[ItemBehavior]] s associated to the Item.
33      *
34      * @return
35      *   [[ItemBehavior]] s associated to the Item.
36      */
37     def behaviors: Seq[ItemBehavior] = Seq.empty
38 
39     override def use(action: Action, maybeSideItem: Option[I])(implicit
40       state: S
41     ): Option[Reaction] =
42       behaviors
43         .map(_.triggers)
44         .reduceOption(_ orElse _)
45         .getOrElse(PartialFunction.empty)
46         .lift((action, maybeSideItem, state))
47   }
48 
49   /**
50    * A [[PartialFunction]], that makes possible to react to specific combinations
51    * action-item-sideItem-state (or action-item-state) responding with a [[Reaction]]. The
52    * [[ItemBehavior]] is based into this construct.
53    */
54   type ItemTriggers = PartialFunction[(Action, Option[I], S), Reaction]
55 
56   /**
57    * Makes a [[BehaviorBasedItem]] react to specific [[ItemTriggers]] with a [[Reaction]]. The
58    * [[BehaviorBasedModel]] is based into this construct.
59    */
60   abstract class ItemBehavior {
61 
62     /**
63      * All the [[ItemTriggers]] of this [[ItemBehavior]].
64      * @return
65      *   the [[ItemTriggers]].
66      */
67     def triggers: ItemTriggers = PartialFunction.empty
68 
69     /**
70      * The <b>BehaviorBasedItem</b> that owe the behavior.
71      * @return
72      *   The <b>BehaviorBasedItem</b> that owe the behavior.
73      */
74     def subject: I
75   }
76 
77   /**
78    * A mixin for [[ItemBehavior]], that enables the possibility implement the delegation pattern for
79    * them. It means that the receiver behavior (the one that mixes in [[Delegate]] ) can owe
80    * internally another one, inheriting its [[ItemTriggers]]. If the receiver overwrites some of the
81    * delegate triggers, the receiver ones have the priority, and the delegate ones are not executed.
82    */
83   trait Delegate extends ItemBehavior {
84 
85     /**
86      * Should be overrode with the delegate [[ItemBehavior]] 's [[ItemTriggers]].
87      *
88      * @return
89      *   The delegate [[ItemBehavior]] 's [[ItemTriggers]].
90      */
91     def delegateTriggers: ItemTriggers
92 
93     /**
94      * Should be overrode with the [[ItemTriggers]] of the receiver (the behavior that mixes in
95      * [[Delegate]] ).
96      *
97      * @return
98      *   The [[ItemTriggers]] of the receiver (the behavior that mixes in [[Delegate]] ).
99      */
100     def receiverTriggers: ItemTriggers
101 
102     /**
103      * Composes together all the [[ItemTriggers]], both of the receiver and of the delegate. If the
104      * receiver overwrites some of the delegate triggers, the receiver ones have the priority, and
105      * the delegate ones are not executed.
106      *
107      * @return
108      *   The combined triggers.
109      */
110     final override def triggers: ItemTriggers =
111       Seq(receiverTriggers, delegateTriggers).reduce(_ orElse _)
112   }
113 
114   override type G = BehaviorBasedGround
115 
116   /**
117    * A [[Ground]] capable to process [[GroundBehavior]] s to react to specific action-state
118    * combinations.
119    *
120    * A Ground that exploit the behavioral mechanism should extend this partial implementation. This
121    * because the handling of use* functions following this style (by exploring the available
122    * triggers) is implemented here.
123    */
124   abstract class BehaviorBasedGround extends Ground {
125 
126     /**
127      * [[GroundBehavior]] s associated to the Ground.
128      *
129      * @return
130      *   [[GroundBehavior]] s associated to the Ground.
131      */
132     def behaviors: Seq[GroundBehavior] = Seq.empty
133 
134     override def use(action: Action)(implicit state: S): Option[Reaction] =
135       behaviors
136         .map(_.triggers)
137         .reduceOption(_ orElse _)
138         .getOrElse(PartialFunction.empty)
139         .lift((action, state))
140   }
141 
142   /**
143    * A [[PartialFunction]], that makes possible to react to specific combinations action-state,
144    * responding with a [[Reaction]]. The [[GroundBehavior]] is based into this construct.
145    */
146   type GroundTriggers = PartialFunction[(Action, S), Reaction]
147 
148   /**
149    * Makes a [[BehaviorBasedGround]] react to specific [[GroundTriggers]] with a [[Reaction]]. The
150    * [[BehaviorBasedModel]] is based into this construct.
151    */
152   abstract class GroundBehavior {
153 
154     /**
155      * [[GroundTriggers]] s associated to the Ground.
156      *
157      * @return
158      *   [[GroundTriggers]] s associated to the Ground.
159      */
160     def triggers: GroundTriggers = PartialFunction.empty
161   }
162 }
