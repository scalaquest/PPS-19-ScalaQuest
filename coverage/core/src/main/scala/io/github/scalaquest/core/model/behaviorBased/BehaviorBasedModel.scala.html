5 /**
6  * This is a partial implementation of [[Model]], following a <b>behavior-based fashion</b>. This is
7  * a style that enables the "decoration" of [[Model.Item]] s and the [[Model.Ground]] by the use of
8  * the <b>Behavior</b>: a sort of composable plugin that matches dynamically various
9  * action-item-state scenarios.
10  */
11 abstract class BehaviorBasedModel extends Model {
12   override type I = BehaviorBasedItem
13 
14   /**
15    * A [[Item]] capable to process [[ItemBehavior]] s to react to specific action-item-state
16    * combinations.
17    *
18    * Items that exploit the behavioral mechanism should extend this partial implementation. This
19    * because the handling of use* functions following this style (by exploring the available
20    * triggers) is implemented here.
21    */
22   abstract class BehaviorBasedItem extends Item {
23 
24     /**
25      * [[ItemBehavior]] s associated to the Item.
26      *
27      * @return
28      *   [[ItemBehavior]] s associated to the Item.
29      */
30     def behaviors: Seq[ItemBehavior] = Seq.empty
31 
32     override def use(action: Action, maybeSideItem: Option[I])(implicit
33       state: S
34     ): Option[Reaction] =
35       behaviors
36         .map(_.triggers)
37         .reduceOption(_ orElse _)
38         .getOrElse(PartialFunction.empty)
39         .lift((action, maybeSideItem, state))
40   }
41 
42   /**
43    * A [[PartialFunction]], that makes possible to react to specific combinations
44    * action-item-sideItem-state (or action-item-state) responding with a [[Reaction]]. The
45    * [[ItemBehavior]] is based into this construct.
46    */
47   type ItemTriggers = PartialFunction[(Action, Option[I], S), Reaction]
48 
49   /**
50    * Makes a [[BehaviorBasedItem]] react to specific [[ItemTriggers]] with a [[Reaction]]. The
51    * [[BehaviorBasedModel]] is based into this construct.
52    */
53   abstract class ItemBehavior {
54 
55     /**
56      * All the [[ItemTriggers]] of this [[ItemBehavior]].
57      * @return
58      *   the [[ItemTriggers]].
59      */
60     def triggers: ItemTriggers = PartialFunction.empty
61 
62     /**
63      * The <b>BehaviorBasedItem</b> that owe the behavior.
64      * @return
65      *   The <b>BehaviorBasedItem</b> that owe the behavior.
66      */
67     def subject: I
68   }
69 
70   /**
71    * A mixin for [[ItemBehavior]], that enables the possibility implement the delegation pattern for
72    * them. It means that the receiver behavior (the one that mixes in [[Delegate]] ) can owe
73    * internally another one, inheriting its [[ItemTriggers]]. If the receiver overwrites some of the
74    * delegate triggers, the receiver ones have the priority, and the delegate ones are not executed.
75    */
76   trait Delegate extends ItemBehavior {
77 
78     /**
79      * Should be overrode with the delegate [[ItemBehavior]] 's [[ItemTriggers]].
80      *
81      * @return
82      *   The delegate [[ItemBehavior]] 's [[ItemTriggers]].
83      */
84     def delegateTriggers: ItemTriggers
85 
86     /**
87      * Should be overrode with the [[ItemTriggers]] of the receiver (the behavior that mixes in
88      * [[Delegate]] ).
89      *
90      * @return
91      *   The [[ItemTriggers]] of the receiver (the behavior that mixes in [[Delegate]] ).
92      */
93     def receiverTriggers: ItemTriggers
94 
95     /**
96      * Composes together all the [[ItemTriggers]], both of the receiver and of the delegate. If the
97      * receiver overwrites some of the delegate triggers, the receiver ones have the priority, and
98      * the delegate ones are not executed.
99      *
100      * @return
101      *   The combined triggers.
102      */
103     final override def triggers: ItemTriggers =
104       Seq(receiverTriggers, delegateTriggers).reduce(_ orElse _)
105   }
106 
107   override type G = BehaviorBasedGround
108 
109   /**
110    * A [[Ground]] capable to process [[GroundBehavior]] s to react to specific action-state
111    * combinations.
112    *
113    * A Ground that exploit the behavioral mechanism should extend this partial implementation. This
114    * because the handling of use* functions following this style (by exploring the available
115    * triggers) is implemented here.
116    */
117   abstract class BehaviorBasedGround extends Ground {
118 
119     /**
120      * [[GroundBehavior]] s associated to the Ground.
121      *
122      * @return
123      *   [[GroundBehavior]] s associated to the Ground.
124      */
125     def behaviors: Seq[GroundBehavior] = Seq.empty
126 
127     override def use(action: Action)(implicit state: S): Option[Reaction] =
128       behaviors
129         .map(_.triggers)
130         .reduceOption(_ orElse _)
131         .getOrElse(PartialFunction.empty)
132         .lift((action, state))
133   }
134 
135   /**
136    * A [[PartialFunction]], that makes possible to react to specific combinations action-state,
137    * responding with a [[Reaction]]. The [[GroundBehavior]] is based into this construct.
138    */
139   type GroundTriggers = PartialFunction[(Action, S), Reaction]
140 
141   /**
142    * Makes a [[BehaviorBasedGround]] react to specific [[GroundTriggers]] with a [[Reaction]]. The
143    * [[BehaviorBasedModel]] is based into this construct.
144    */
145   abstract class GroundBehavior {
146 
147     /**
148      * [[GroundTriggers]] s associated to the Ground.
149      *
150      * @return
151      *   [[GroundTriggers]] s associated to the Ground.
152      */
153     def triggers: GroundTriggers = PartialFunction.empty
154   }
155 }
