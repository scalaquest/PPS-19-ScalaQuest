5 /**
6  * This is a partial implementation of [[Model]], following a <b>behavior-based fashion</b>. This is
7  * a style that enables the "decoration" of [[Model.Item]] s and the [[Model.Ground]] by the use of
8  * the <b>Behavior</b>: a sort of composable plugin that matches dynamically various
9  * action-item-state scenarios.
10  */
11 abstract class BehaviorBasedModel extends Model {
12   override type I = BehaviorBasedItem
13 
14   /**
15    * A [[Item]] capable to process [[ItemBehavior]] s to react to specific action-item-state
16    * combinations.
17    *
18    * Items that exploit the behavioral mechanism should extend this partial implementation. This
19    * because the handling of use* functions following this style (by exploring the available
20    * triggers) is implemented here.
21    */
22   abstract class BehaviorBasedItem extends Item {
23 
24     /**
25      * [[ItemBehavior]] s associated to the Item.
26      *
27      * @return
28      *   [[ItemBehavior]] s associated to the Item.
29      */
30     def behaviors: Seq[ItemBehavior] = Seq()
31 
32     override def use(action: Action, state: S, maybeSideItem: Option[I]): Option[Reaction] =
33       behaviors.map(_.triggers).reduce(_ orElse _).lift((action, this, maybeSideItem, state))
34   }
35 
36   /**
37    * A [[PartialFunction]], that makes possible to react to specific combinations
38    * action-item-sideItem-state (or action-item-state) responding with a [[Reaction]]. The
39    * [[ItemBehavior]] is based into this construct.
40    */
41   type ItemTriggers = PartialFunction[(Action, I, Option[I], S), Reaction]
42 
43   /**
44    * Makes a [[BehaviorBasedItem]] react to specific [[ItemTriggers]] with a [[Reaction]]. The
45    * [[BehaviorBasedModel]] is based into this construct.
46    */
47   abstract class ItemBehavior {
48     def triggers: ItemTriggers = PartialFunction.empty
49   }
50 
51   /**
52    * A mixin for [[ItemBehavior]], that enables the possibility implement the delegation pattern for
53    * them. It means that the receiver behavior (the one that mixes in [[Delegate]] ) can owe
54    * internally another one, inheriting its [[ItemTriggers]]. If the receiver overwrites some of the
55    * delegate triggers, the receiver ones have the priority, and the delegate ones are not executed.
56    */
57   trait Delegate extends ItemBehavior {
58 
59     /**
60      * Should be overrode with the delegate [[ItemBehavior]] 's [[ItemTriggers]].
61      *
62      * @return
63      *   The delegate [[ItemBehavior]] 's [[ItemTriggers]].
64      */
65     def delegateTriggers: ItemTriggers
66 
67     /**
68      * Should be overrode with the [[ItemTriggers]] of the receiver (the behavior that mixes in
69      * [[Delegate]] ).
70      *
71      * @return
72      *   The [[ItemTriggers]] of the receiver (the behavior that mixes in [[Delegate]] ).
73      */
74     def receiverTriggers: ItemTriggers = PartialFunction.empty
75 
76     /**
77      * Composes together all the [[ItemTriggers]], both of the receiver and of the delegate. If the
78      * receiver overwrites some of the delegate triggers, the receiver ones have the priority, and
79      * the delegate ones are not executed.
80      *
81      * @return
82      *   The combined triggers.
83      */
84     final override def triggers: ItemTriggers =
85       Seq(receiverTriggers, delegateTriggers).reduce(_ orElse _)
86   }
87 
88   override type G = BehaviorBasedGround
89 
90   /**
91    * A [[Ground]] capable to process [[GroundBehavior]] s to react to specific action-state
92    * combinations.
93    *
94    * A Ground that exploit the behavioral mechanism should extend this partial implementation. This
95    * because the handling of use* functions following this style (by exploring the available
96    * triggers) is implemented here.
97    */
98   abstract class BehaviorBasedGround extends Ground {
99     def behaviors: Seq[GroundBehavior] = Seq()
100 
101     override def use(action: Action, state: S): Option[Reaction] =
102       behaviors.map(_.triggers).reduce(_ orElse _).lift((action, state))
103   }
104 
105   /**
106    * A [[PartialFunction]], that makes possible to react to specific combinations action-state,
107    * responding with a [[Reaction]]. The [[GroundBehavior]] is based into this construct.
108    */
109   type GroundTriggers = PartialFunction[(Action, S), Reaction]
110 
111   /**
112    * Makes a [[BehaviorBasedGround]] react to specific [[GroundTriggers]] with a [[Reaction]]. The
113    * [[BehaviorBasedModel]] is based into this construct.
114    */
115   abstract class GroundBehavior {
116     def triggers: GroundTriggers = PartialFunction.empty
117   }
118 }
