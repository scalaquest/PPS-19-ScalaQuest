6 /**
7  * A pipeline component that takes a [[Statement]] (wrapped into a [[ResolverResult]] ) and returns
8  * a [[Model.Reaction]] wrapped into an [[InterpreterResult]]. The execution may fail, when the
9  * given [[Statement]] does not generate a [[Model.Reaction]].
10  * @tparam M
11  *   The concrete type of the [[Model]] in use.
12  * @tparam R
13  *   The concrete type of the [[Model.Reaction]] in use. It should be derived from [[M]].
14  */
15 trait Interpreter[M <: Model, R] {
16 
17   /**
18    * Triggers the [[Interpreter]] execution.
19    * @param resolverResult
20    *   A wrapper for the input [[Statement]].
21    * @return
22    *   An [[Either]] describing the interpreter result. If the [[Interpreter]] fails, the result is
23    *   a [[Left]] describing what went wrong. Otherwise, it is a [[Right]] with the
24    *   [[InterpreterResult]] (wrapper for a [[Model.Reaction]] ).
25    */
26   def interpret(resolverResult: ResolverResult): Either[String, InterpreterResult[R]]
27 }
28 
29 /**
30  * Companion object for the [[Interpreter]] trait. It exposes some utilities to instantiate the
31  * [[Interpreter]] with the right constraints between types.
32  *
33  * It exposes an [[Interpreter::apply()]] to build an [[Interpreter]] <b>with the right types</b>,
34  * as some constraints between the used types are defined. In addition, it provides a
35  * [[Interpreter::builder()]] utility used into the pipeline to inject the [[Model.State]] in a
36  * separate phase, compared to passing the state directly to [[Interpreter::interpret()]].
37  */
38 object Interpreter {
39 
40   /**
41    * A generator of [[Interpreter]] instances, giving the possibility to inject the [[Model.State]]
42    * separately from the [[Interpreter::interpret()]] call.
43    * @tparam M
44    *   The concrete type of the [[Model]] in use.
45    * @tparam S
46    *   The concrete type of the [[Model.State]] in use. It should be derived from [[M]].
47    * @tparam R
48    *   The concrete type of the [[Model.Reaction]] in use. It should be derived from [[M]].
49    */
50   type Builder[M <: Model, S, R] = S => Interpreter[M, R]
51 
52   /**
53    * It generates a [[Builder]] with the right type constraints.
54    * @param model
55    *   The concrete instance of the [[Model]] in use.
56    * @param itemDict
57    *   A [[Map]] that indicates the [[ItemRef]] of each ot the [[Model.Item]] in use. The
58    *   [[Model.Item]] type must be derived from the `model` passed as parameter.
59    * @param ground
60    *   The [[Model.Ground]] instance of the match. The [[Model.Ground]] type must be derived from
61    *   the `model` passed as parameter.
62    * @tparam M
63    *   The concrete type of the [[Model]] in use.
64    * @return
65    *   A [[Builder]] of [[Interpreter]] instances, with the right type constraints.
66    */
67   def builder[M <: Model](model: M): Builder[model.type, model.S, model.Reaction] =
68     state => {
69       implicit val s: model.S           = state
70       val refToItem: RefToItem[model.I] = RefToItem(model)(state.items)
71 
72       // shortcut for implementing the Interpreter, as it is a single method trait
73       (resolverResult: ResolverResult) =>
74         for {
75           maybeReaction <- resolverResult.statement match {
76             case Statement.Intransitive(action) =>
77               state.ground.use(action).toRight(s"I can't do that!")
78 
79             case Statement.Transitive(action, refToItem(item)) =>
80               item.use(action).toRight(s"I can't do that!")
81 
82             case Statement.Ditransitive(action, refToItem(directObj), refToItem(indirectObj)) =>
83               directObj.use(action, Some(indirectObj)).toRight(s"I can't do that!")
84 
85             case _ => Left(s"I can't do that!")
86           }
87         } yield InterpreterResult(model)(maybeReaction)
88     }
89 }
