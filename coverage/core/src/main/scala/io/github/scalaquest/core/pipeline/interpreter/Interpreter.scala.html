13 /**
14  * A pipeline component that takes a [[Statement]] (wrapped into a [[ResolverResult]] ) and returns
15  * a [[Model.Reaction]] wrapped into an [[InterpreterResult]]. The execution may fail, when the
16  * given [[Statement]] does not generate a [[Model.Reaction]].
17  * @tparam M
18  *   The concrete type of the [[Model]] in use.
19  * @tparam R
20  *   The concrete type of the [[Model.Reaction]] in use. It should be derived from [[M]].
21  */
22 trait Interpreter[M <: Model, R] {
23 
24   /**
25    * Triggers the [[Interpreter]] execution.
26    * @param resolverResult
27    *   A wrapper for the input [[Statement]].
28    * @return
29    *   An [[Either]] describing the interpreter result. If the [[Interpreter]] fails, the result is
30    *   a [[Left]] describing what went wrong. Otherwise, it is a [[Right]] with the
31    *   [[InterpreterResult]] (wrapper for a [[Model.Reaction]] ).
32    */
33   def interpret(resolverResult: ResolverResult): Either[String, InterpreterResult[R]]
34 }
35 
36 /**
37  * Companion object for the [[Interpreter]] trait. It exposes some utilities to instantiate the
38  * [[Interpreter]] with the right constraints between types.
39  *
40  * It exposes an [[Interpreter::apply()]] to build an [[Interpreter]] <b>with the right types</b>,
41  * as some constraints between the used types are defined. In addition, it provides a
42  * [[Interpreter::builder()]] utility used into the pipeline to inject the [[Model.State]] in a
43  * separate phase, compared to passing the state directly to [[Interpreter::interpret()]].
44  */
45 object Interpreter {
46 
47   /**
48    * A generator of [[Interpreter]] instances, giving the possibility to inject the [[Model.State]]
49    * separately from the [[Interpreter::interpret()]] call.
50    * @tparam M
51    *   The concrete type of the [[Model]] in use.
52    * @tparam S
53    *   The concrete type of the [[Model.State]] in use. It should be derived from [[M]].
54    * @tparam R
55    *   The concrete type of the [[Model.Reaction]] in use. It should be derived from [[M]].
56    */
57   type Builder[M <: Model, S, R] = S => Interpreter[M, R]
58 
59   /**
60    * It generates a [[Builder]] with the right type constraints.
61    * @param model
62    *   The concrete instance of the [[Model]] in use.
63    * @tparam M
64    *   The concrete type of the [[Model]] in use.
65    * @return
66    *   A [[Builder]] of [[Interpreter]] instances, with the right type constraints.
67    */
68   def builder[M <: Model](model: M): Builder[model.type, model.S, model.Reaction] =
69     state => {
70       implicit val s: model.S           = state
71       val refToItem: RefToItem[model.I] = RefToItem(model)(state.items)
72       val actionNotFound: String        = "I can't do that!"
73 
74       // shortcut for implementing the Interpreter, as it is a single method trait
75       (resolverResult: ResolverResult) =>
76         for {
77           maybeReaction <- resolverResult.statement match {
78             case Statement.Intransitive(action) =>
79               state.ground.use(action).toRight(actionNotFound)
80 
81             case Statement.Transitive(action, refToItem(item)) =>
82               item.use(action).toRight(actionNotFound)
83 
84             case Statement.Ditransitive(action, refToItem(directObj), refToItem(indirectObj)) =>
85               directObj.use(action, Some(indirectObj)).toRight(actionNotFound)
86 
87             case _ => Left(actionNotFound)
88           }
89         } yield InterpreterResult(model)(maybeReaction)
90     }
91 }
