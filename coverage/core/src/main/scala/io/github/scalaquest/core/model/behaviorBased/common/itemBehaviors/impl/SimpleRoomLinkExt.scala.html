12 
13   /**
14    * Standard implementation of the RoomLink.
15    *
16    * This is a behavior associated to an item that is a link between two rooms (a door, for
17    * instance), and that could be opened to pass into it. The user can "Enter" into it, resulting
18    * into move it into the connected Room. another room.
19    * @param endRoom
20    *   The room into which the user is projected after entering the object.
21    * @param openable
22    *   The Openable behavior associated to the roomLink. If not passed, the item can be entered
23    *   without opening.
24    * @param onEnterExtra
25    *   Reaction to be executed into the State when entered, after the standard Reaction. It can be
26    *   omitted.
27    */
28   case class SimpleRoomLink(
29     endRoom: RM,
30     openable: Option[Openable] = None,
31     onEnterExtra: Option[Reaction] = None
32   )(implicit
33     playerLocationLens: Lens[S, RoomRef],
34     messageLens: Lens[S, Seq[Message]]
35   ) extends RoomLink
36     with Delegate {
37 
38     /**
39      * If an openable is passed, it is passed as father behavior.
40      * @return
41      *   The father behavior triggers.
42      */
43     override def delegateTriggers: ItemTriggers =
44       openable.fold(PartialFunction.empty: ItemTriggers)(_.triggers)
45 
46     override def receiverTriggers: ItemTriggers = {
47       // "Enter the item"
48       case (Enter, item, None, state)
49           if state.isInCurrentRoom(item) && openable.fold(true)(_.isOpen) =>
50         enterRoom()
51     }
52 
53     def enterRoom(): Reaction =
54       _.applyReactions(
55         playerLocationLens.modify(_ => endRoom.ref),
56         messageLens.modify(_ :+ Navigated(endRoom)),
57         onEnterExtra.getOrElse(state => state)
58       )
59 
60     override def isAccessible: Boolean = openable.fold(true)(_.isOpen)
61   }
62 }
