6 /**
7  * Extension for the model, based on the behavior-based fashion. Adds a base implementation of the
8  * <b>BehaviorBasedGround</b> with common behaviors inside, extensible on demand with more
9  * behaviors.
10  */
11 trait CGroundExt extends BehaviorBasedModel with CGroundBehaviorsExt {
12 
13   /**
14    * A <b>BehaviorBasedGround</b> implementation that includes all the common
15    * <b>GroundBehaviors</b>.
16    * @param extraBehaviors
17    *   <b>GroundBehaviors</b> to include in addition to the common ones.
18    */
19   case class CGround(extraBehaviors: Seq[GroundBehavior] = Seq.empty) extends BehaviorBasedGround {
20 
21     override val behaviors: Seq[GroundBehavior] =
22       extraBehaviors ++ Seq(InspectableLocation(), InspectableBag(), Navigable())
23   }

27    */

31      * A <b>BehaviorBasedGround</b> with the common <b>GroundBehaviors</b>, and an extra custom
32      * additional behavior.

34      *   The single additional behavior associated to the Ground.
35      * @return
36      *   A <b>BehaviorBasedGround</b> with the common <b>GroundBehaviors</b>, and an extra custom
37      *   additional behavior.
38      */
39     def withSingleExtraBehavior(
40       extraBehavior: GroundBehavior
41     ): BehaviorBasedGround = CGround(Seq(extraBehavior))
42 
43     /**
44      * A <b>BehaviorBasedGround</b> with the common <b>GroundBehaviors</b>, and an extra custom
45      * additional behavior, created on-the-fly by some <b>GroundTriggers</b>.
46      * @param behaviorTriggers
47      *   Triggers for the single behavior associated to the item.
48      * @return
49      *   An instance of a <b>GenericItem</b> with a single behavior.
50      */
51     def withGenExtraBehavior(
52       behaviorTriggers: GroundTriggers
53     ): BehaviorBasedGround = CGround(Seq(GenericGroundBehavior(behaviorTriggers)))
54   }
