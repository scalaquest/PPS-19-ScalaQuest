13 /**
14  * Extension for the model, based on the behavior-based fashion. Adds a base implementation of the
15  * <b>BehaviorBasedGround</b> with common behaviors inside, extensible on demand with more
16  * behaviors.
17  */
18 trait CGroundExt extends BehaviorBasedModel with CGroundBehaviorsExt {
19 
20   /**
21    * A <b>BehaviorBasedGround</b> implementation that includes all the common
22    * <b>GroundBehaviors</b>.
23    * @param extraBehaviors
24    *   <b>GroundBehaviors</b> to include in addition to the common ones.
25    */
26   case class CGround(extraBehaviors: Seq[GroundBehavior] = Seq.empty) extends BehaviorBasedGround {
27 
28     override val behaviors: Seq[GroundBehavior] =
29       extraBehaviors ++ Seq(InspectableLocation(), InspectableBag(), Navigable())
30   }
31 
32   /**
33    * Companion object for <b>CGround</b>.
34    */
35   object CGround {
36 
37     /**
38      * A <b>BehaviorBasedGround</b> with the common <b>GroundBehaviors</b>, and an extra custom
39      * additional behavior.
40      * @param extraBehavior
41      *   The single additional behavior associated to the Ground.
42      * @return
43      *   A <b>BehaviorBasedGround</b> with the common <b>GroundBehaviors</b>, and an extra custom
44      *   additional behavior.
45      */
46     def withSingleExtraBehavior(
47       extraBehavior: GroundBehavior
48     ): BehaviorBasedGround = CGround(Seq(extraBehavior))
49 
50     /**
51      * A <b>BehaviorBasedGround</b> with the common <b>GroundBehaviors</b>, and an extra custom
52      * additional behavior, created on-the-fly by some <b>GroundTriggers</b>.
53      * @param behaviorTriggers
54      *   Triggers for the single behavior associated to the item.
55      * @return
56      *   An instance of a <b>GenericItem</b> with a single behavior.
57      */
58     def withGenExtraBehavior(
59       behaviorTriggers: GroundTriggers
60     ): BehaviorBasedGround = CGround(Seq(GenericGroundBehavior(behaviorTriggers)))
61   }
62 }
