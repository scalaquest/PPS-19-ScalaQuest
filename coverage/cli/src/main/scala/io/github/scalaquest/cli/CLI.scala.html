23   /** An effect that prints a prompt and asks for a command that cannot be empty. */
24   def readLine: ZIO[Console, IOException, Command] =
25     for {
26       _   <- putStr("> ") *> putStr("\u001B[3m") *> putStr(AnsiColor.GREEN)
27       in  <- getStrLn.map(Command.make)
28       _   <- putStr(AnsiColor.RESET)
29       cmd <- ZIO.succeed(in).someOrElseM(readLine)
30     } yield cmd;

32   class CLIBuilder[M <: Model](val model: M) {
33 
34     /**
35      * Creates a `CLI` with the given parameters.
36      * @param state
37      *   the initial state
38      * @param game
39      *   the game to send the commands to
40      * @param pusher
41      *   the interpreter of the messages
42      * @param initialMessages
43      *   optional initial messages
44      * @return
45      *   the command line application
46      */
47     def build(
48       state: model.S,
49       game: Game[model.type],
50       pusher: MessagePusher[String],
51       initialMessages: Seq[Message] = Seq()
52     ): CLI =
53       new CLI() {
54 
55         override def start: ZIO[Console, IOException, Unit] =
56           for {
57             _ <-
58               if (initialMessages.nonEmpty) putStrLn(pusher push initialMessages)
59               else ZIO.unit
60             _ <- gameLoop(game, pusher)(state)
61           } yield ()
62       }
63 
64     private def saveState(
65       path: String,
66       state: model.S
67     ): IO[Either[UnsupportedOperationException, IOException], Unit] = {
68       IO.fromFuture { _ =>
69         model.serializer.get.write(path, state)
70       }.mapError {
71         case _: NoSuchElementException => Left(new UnsupportedOperationException)
72         case e: IOException            => Right(e)
73         case t: Throwable              => throw t
74       }
75     }
76 
77     private def loadState(
78       path: String
79     ): IO[Either[UnsupportedOperationException, IOException], model.S] = {
80       IO
81         .fromFuture { _ =>
82           model.serializer.get.read(path)
83         }
84         .mapError {
85           case _: NoSuchElementException => Left(new UnsupportedOperationException)
86           case e: IOException            => Right(e)
87           case t: Throwable              => throw t
88         }
89     }
90 
91     private def gameLoop(
92       game: Game[model.type],
93       pusher: MessagePusher[String]
94     )(state: model.S): ZIO[Console, IOException, Unit] = {
95 
96       def onCommand(input: String): ZIO[Console, IOException, Unit] = {
97         val loop = for {
98           pipelineRes <- UIO.succeed((game send input)(state))
99           (output, updState) <- IO
100             .succeed(pipelineRes match {
101               case Left(err)                   => (err, state)
102               case Right((updState, messages)) => (pusher push messages, updState)
103             })
104           _ <- putStrLn(output)
105           _ <-
106             if (updState.ended) ZIO.unit
107             else gameLoop(game, pusher)(updState)
108         } yield ()
109         loop.refineToOrDie[IOException]
110       }
111 
112       def onMetaCommand(input: String): ZIO[Console, IOException, Unit] =
113         for {
114           nextState <- input match {
115             case Command.Parser("save", path) =>
116               for {
117                 saveResult <- saveState(path, state).either
118                 _ <- saveResult match {
119                   case Left(Left(_)) =>
120                     putStrLn("Operation not supported.")
121                   case Left(Right(e)) =>
122                     putStrLn(s"Couldn't save game: ${e.getMessage}")
123                   case Right(_) =>
124                     putStrLn("Saved game successfully!")
125                 }
126               } yield None
127             case Command.Parser("load", path) =>
128               for {
129                 loadResult <- loadState(path).either
130                 s <- loadResult match {
131                   case Left(Left(_)) =>
132                     putStrLn("Operation not supported.") *> ZIO.none
133                   case Left(Right(e)) =>
134                     putStrLn(s"Couldn't load game: ${e.getMessage}") *> ZIO.none
135                   case Right(updatedState) =>
136                     putStrLn("Loaded game save successfully!").as(updatedState).asSome
137                 }
138               } yield s
139             case Command.Parser("save", _*) =>
140               putStrLn("Usage: :save <file>") *> ZIO.none
141             case Command.Parser("load", _*) =>
142               putStrLn("Usage: :load <file>") *> ZIO.none
143             case _ =>
144               putStrLn("Error: Unrecognized command \"" + input + "\"") *> ZIO.none
145           }
146           _ <- gameLoop(game, pusher)(nextState getOrElse state)
147         } yield ()
148 
149       for {
150         command <- readLine
151         _ <- command match {
152           case GameCommand(body) => onCommand(body)
153           case MetaCommand(body) => onMetaCommand(body)
154         }
155       } yield ()
156     }
157   }
