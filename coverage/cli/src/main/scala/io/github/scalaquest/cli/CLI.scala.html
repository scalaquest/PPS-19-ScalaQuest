18 /**
19  * An generic command line application that takes no arguments.
20  */
21 trait CLI {
22   def start: ZIO[Console, IOException, Unit]
23 }
24 
25 object CLI {
26 
27   /** A builder function to create a `CLI` using dependent types on the `Model` instance. */
28   def builderFrom[M <: Model](implicit model: M) = new CLIBuilder[M](model)
29 
30   /** An effect that prints a prompt and asks for a command that cannot be empty. */
31   def readLine: ZIO[Console, IOException, Command] =
32     for {
33       _   <- putStr("> ") *> putStr("\u001B[3m") *> putStr(AnsiColor.GREEN)
34       in  <- getStrLn.map(Command.make)
35       _   <- putStr(AnsiColor.RESET)
36       cmd <- ZIO.succeed(in).someOrElseM(readLine)
37     } yield cmd;
38 
39   class CLIBuilder[M <: Model](val model: M) {
40 
41     /**
42      * Creates a `CLI` with the given parameters.
43      * @param state
44      *   the initial state
45      * @param game
46      *   the game to send the commands to
47      * @param pusher
48      *   the interpreter of the messages
49      * @param initialMessages
50      *   optional initial messages
51      * @return
52      *   the command line application
53      */
54     def build(
55       state: model.S,
56       game: Game[model.type],
57       pusher: MessagePusher[String],
58       initialMessages: Seq[Message] = Seq()
59     ): CLI =
60       new CLI() {
61 
62         override def start: ZIO[Console, IOException, Unit] =
63           for {
64             _ <-
65               if (initialMessages.nonEmpty) putStrLn(pusher push initialMessages)
66               else ZIO.unit
67             _ <- gameLoop(game, pusher)(state)
68           } yield ()
69       }
70 
71     private def saveState(
72       path: String,
73       state: model.S
74     ): IO[Either[UnsupportedOperationException, IOException], Unit] = {
75       IO.fromFuture { _ =>
76         model.serializer.get.write(path, state)
77       }.mapError {
78         case _: NoSuchElementException => Left(new UnsupportedOperationException)
79         case e: IOException            => Right(e)
80         case t: Throwable              => throw t
81       }
82     }
83 
84     private def loadState(
85       path: String
86     ): IO[Either[UnsupportedOperationException, IOException], model.S] = {
87       IO
88         .fromFuture { _ =>
89           model.serializer.get.read(path)
90         }
91         .mapError {
92           case _: NoSuchElementException => Left(new UnsupportedOperationException)
93           case e: IOException            => Right(e)
94           case t: Throwable              => throw t
95         }
96     }
97 
98     private def gameLoop(
99       game: Game[model.type],
100       pusher: MessagePusher[String]
101     )(state: model.S): ZIO[Console, IOException, Unit] = {
102 
103       def onCommand(input: String): ZIO[Console, IOException, Unit] = {
104         val loop = for {
105           pipelineRes <- UIO.succeed((game send input)(state))
106           (output, updState) <- IO
107             .succeed(pipelineRes match {
108               case Left(err)                   => (err, state)
109               case Right((updState, messages)) => (pusher push messages, updState)
110             })
111           _ <- putStrLn(output)
112           _ <-
113             if (updState.ended) ZIO.unit
114             else gameLoop(game, pusher)(updState)
115         } yield ()
116         loop.refineToOrDie[IOException]
117       }
118 
119       def onMetaCommand(input: String): ZIO[Console, IOException, Unit] =
120         for {
121           nextState <- input match {
122             case Command.Parser("save", path) =>
123               for {
124                 saveResult <- saveState(path, state).either
125                 _ <- saveResult match {
126                   case Left(Left(_)) =>
127                     putStrLn("Operation not supported.")
128                   case Left(Right(e)) =>
129                     putStrLn(s"Couldn't save game: ${e.getMessage}")
130                   case Right(_) =>
131                     putStrLn("Saved game successfully!")
132                 }
133               } yield None
134             case Command.Parser("load", path) =>
135               for {
136                 loadResult <- loadState(path).either
137                 s <- loadResult match {
138                   case Left(Left(_)) =>
139                     putStrLn("Operation not supported.") *> ZIO.none
140                   case Left(Right(e)) =>
141                     putStrLn(s"Couldn't load game: ${e.getMessage}") *> ZIO.none
142                   case Right(updatedState) =>
143                     putStrLn("Loaded game save successfully!").as(updatedState).asSome
144                 }
145               } yield s
146             case Command.Parser("save", _*) =>
147               putStrLn("Usage: :save <file>") *> ZIO.none
148             case Command.Parser("load", _*) =>
149               putStrLn("Usage: :load <file>") *> ZIO.none
150             case _ =>
151               putStrLn("Error: Unrecognized command \"" + input + "\"") *> ZIO.none
152           }
153           _ <- gameLoop(game, pusher)(nextState getOrElse state)
154         } yield ()
155 
156       for {
157         command <- readLine
158         _ <- command match {
159           case GameCommand(body) => onCommand(body)
160           case MetaCommand(body) => onMetaCommand(body)
161         }
162       } yield ()
163     }
164   }
